57static int espnow_config_func(int argc, char **argv)
58{
59 if (arg_parse(argc, argv, (void **) &espnow_config_args) != ESP_OK) {
60 arg_print_errors(stderr, espnow_config_args.end, argv[0]);
61 return ESP_FAIL;
62 }
63
64 if (espnow_config_args.info->count) {
65 int8_t power = 0;
66 uint8_t protocol_bitmap = 0;
67 uint8_t primary = 0;
68 wifi_country_t country = {0};
69 wifi_second_chan_t second = WIFI_SECOND_CHAN_NONE;
70
71 esp_wifi_get_country(&country);
72 esp_wifi_get_channel(&primary, &second);
73 esp_wifi_get_max_tx_power(&power);
74 esp_wifi_get_protocol(ESP_IF_WIFI_STA, &protocol_bitmap);
75
76 ESP_LOGI(TAG, "Channel, primary: %d, second: %d", primary, second);
77 ESP_LOGI(TAG, "Maximum transmiting power: %d", power);
78 ESP_LOGI(TAG, "Wi-Fi protocol bitmap: 0x%02x", protocol_bitmap);
79 country.cc[2] = '\0';
80 ESP_LOGI(TAG, "Wi-Fi country code: %s", country.cc);
81 return ESP_OK;
82 }
83
84 if (espnow_config_args.country_code->count) {
85 wifi_country_t country = {0};
86 const char *country_code = espnow_config_args.country_code->sval[0];
87
88 if (!strcasecmp(country_code, "US")) {
89 strcpy(country.cc, "US");
90 country.schan = 1;
91 country.nchan = 11;
92 } else if (!strcasecmp(country_code, "JP")) {
93 strcpy(country.cc, "JP");
94 country.schan = 1;
95 country.nchan = 14;
96 } else if (!strcasecmp(country_code, "CN")) {
97 strcpy(country.cc, "CN");
98 country.schan = 1;
99 country.nchan = 13;
100 } else {
101 return ESP_ERR_INVALID_ARG;
102 }
103
104 esp_wifi_set_country(&country);
105 }
106
107 if (espnow_config_args.channel->count) {
108 ESP_ERROR_CHECK(esp_wifi_set_channel(espnow_config_args.channel->ival[0], WIFI_SECOND_CHAN_NONE));
109 }
110
111 if (espnow_config_args.rate->count) {
112#if ESP_IDF_VERSION >= ESP_IDF_VERSION_VAL(5, 1, 0)
113 if (espnow_config_args.rate->count == 4) {
114 uint8_t peer[ESPNOW_ADDR_LEN] = {0};
115 esp_now_rate_config_t rate_config = {
116 .phymode = espnow_config_args.rate->ival[0],
117 .rate = espnow_config_args.rate->ival[1],
118 .ersu = espnow_config_args.rate->ival[2],
119#if ESP_IDF_VERSION >= ESP_IDF_VERSION_VAL(5, 1, 3)
120 .dcm = espnow_config_args.rate->ival[3],
121#endif
122 };
123 ESP_LOGI(TAG, "Set rate config: phymode %d, rate %d, ersu %d, dcm %d", rate_config.phymode, rate_config.rate, rate_config.ersu, espnow_config_args.rate->ival[3]);
124 ESP_ERROR_CHECK(esp_wifi_get_mac(ESP_IF_WIFI_STA, peer));
125 ESP_ERROR_CHECK(esp_now_set_peer_rate_config(peer, &rate_config));
126 } else {
127 ESP_LOGE(TAG, "Loss rate config value");
128 return ESP_FAIL;
129 }
130#else
131 ESP_ERROR_CHECK(esp_wifi_config_espnow_rate(ESP_IF_WIFI_STA, espnow_config_args.rate->ival[0]));
132#endif
133 }
134
135 if (espnow_config_args.tx_power->count) {
136 ESP_ERROR_CHECK(esp_wifi_set_max_tx_power(espnow_config_args.tx_power->ival[0]));
137 }
138
139 if (espnow_config_args.protocol->count) {
140 ESP_ERROR_CHECK(esp_wifi_set_protocol(ESP_IF_WIFI_STA, espnow_config_args.protocol->ival[0]));
141 }
142
143 return ESP_OK;
144}

171struct espnow_iperf_cfg {
172 bool finish;
173 uint16_t packet_len;
174 uint16_t transmit_time;
175 uint32_t ping_count;
176 uint16_t report_interval;
177 espnow_data_type_t type;
178 espnow_frame_head_t frame_head;
179 uint8_t addr[6];
180 int gpio_num;
181} g_iperf_cfg = {
182 .finish = true,
183 .packet_len = ESPNOW_DATA_LEN,
184 .transmit_time = 60,
185 .report_interval = 3,
186 .ping_count = 64,
187 .frame_head.group = false,
188 .type = ESPNOW_DATA_TYPE_RESERVED,
189};

191typedef enum {
192 IPERF_BANDWIDTH,
193 IPERF_BANDWIDTH_STOP,
194 IPERF_BANDWIDTH_STOP_ACK,
195 IPERF_PING,
196 IPERF_PING_ACK,
197} iperf_type_t;

199typedef struct {
200 iperf_type_t type;
201 uint32_t seq;

209 void *data;

214static esp_err_t espnow_iperf_initiator_recv(uint8_t *src_addr, void *data,
215 size_t size, wifi_pkt_rx_ctrl_t *rx_ctrl)
216{
217 ESP_PARAM_CHECK(src_addr);
218 ESP_PARAM_CHECK(data);
219 ESP_PARAM_CHECK(size);
220 ESP_PARAM_CHECK(rx_ctrl);
221
222 if (g_iperf_queue) {
223 iperf_recv_data_t iperf_data = { 0 };
224 iperf_data.data = ESP_MALLOC(size);
225 if (!iperf_data.data) {
226 return ESP_FAIL;
227 }
228 memcpy(iperf_data.data, data, size);
229 iperf_data.size = size;
230 memcpy(iperf_data.src_addr, src_addr, 6);
231 memcpy(&iperf_data.rx_ctrl, rx_ctrl, sizeof(wifi_pkt_rx_ctrl_t));
232 if (xQueueSend(g_iperf_queue, &iperf_data, 0) != pdPASS) {
233 ESP_LOGW(TAG, "[%s, %d] Send iperf recv queue failed", __func__, __LINE__);
234 ESP_FREE(iperf_data.data);
235 return ESP_FAIL;
236 }
237 }
238
239 return ESP_OK;
240}

242static void espnow_iperf_initiator_task(void *arg)
243{
244 esp_err_t ret = ESP_OK;
245 espnow_iperf_data_t *iperf_data = ESP_CALLOC(1, g_iperf_cfg.packet_len);
246 iperf_data->type = IPERF_BANDWIDTH;
247 iperf_data->seq = 0;
248
249
250 int64_t start_time = esp_timer_get_time();
251 int64_t end_time = start_time + g_iperf_cfg.transmit_time * 1000 * 1000;
252 uint32_t total_count = 0;
253
254 if (!g_iperf_cfg.frame_head.broadcast) {
255 espnow_add_peer(g_iperf_cfg.addr, NULL);
256 }
257
258 ESP_LOGI(TAG, "[ Responder MAC ] Interval Transfer Frame_rate Bandwidth");
259
260 for (int64_t report_time = start_time + g_iperf_cfg.report_interval * 1000 * 1000, report_count = 0;
261 esp_timer_get_time() < end_time && !g_iperf_cfg.finish;) {
262 ret = espnow_send(g_iperf_cfg.type, g_iperf_cfg.addr, iperf_data,
263 g_iperf_cfg.packet_len, &g_iperf_cfg.frame_head, portMAX_DELAY);
264 ESP_ERROR_CONTINUE(ret != ESP_OK && ret != ESP_ERR_WIFI_TIMEOUT, "<%s> espnow_send", esp_err_to_name(ret));
265 iperf_data->seq++;
266 ++total_count;
267
268 if (esp_timer_get_time() >= report_time) {
269 uint32_t report_time_s = (report_time - start_time) / (1000 * 1000);
270 double report_size = (iperf_data->seq - report_count) * g_iperf_cfg.packet_len / 1e6;
271
272 ESP_LOGI(TAG, "["MACSTR"] %2d-%2d sec %2.2f MBytes %0.2f Hz %0.2f Mbps",
273 MAC2STR(g_iperf_cfg.addr), report_time_s - g_iperf_cfg.report_interval, report_time_s,
274 report_size, (iperf_data->seq - report_count) * 1.0 / g_iperf_cfg.report_interval, report_size * 8 / g_iperf_cfg.report_interval);
275
276 report_time = esp_timer_get_time() + g_iperf_cfg.report_interval * 1000 * 1000;
277 report_count = iperf_data->seq;
278 }
279 }
280
281 iperf_data->type = IPERF_BANDWIDTH_STOP;
282 int retry_count = 5;
283 wifi_pkt_rx_ctrl_t rx_ctrl = {0};
284 uint32_t spend_time_ms = (esp_timer_get_time() - start_time) / 1000;
285
286 do {
287 ret = espnow_send(g_iperf_cfg.type, g_iperf_cfg.addr, iperf_data,
288 g_iperf_cfg.packet_len, &g_iperf_cfg.frame_head, portMAX_DELAY);
289 ESP_ERROR_CONTINUE(ret != ESP_OK, "<%s> espnow_send", esp_err_to_name(ret));
290
291 memset(iperf_data, 0, g_iperf_cfg.packet_len);
292 iperf_recv_data_t recv_data = { 0 };
293 if (g_iperf_queue && xQueueReceive(g_iperf_queue, &recv_data, pdMS_TO_TICKS(1000)) == pdPASS) {
294 ret = ESP_OK;
295 memcpy(iperf_data, recv_data.data, recv_data.size);
296 memcpy(g_iperf_cfg.addr, recv_data.src_addr, 6);
297 memcpy(&rx_ctrl, &recv_data.rx_ctrl, sizeof(wifi_pkt_rx_ctrl_t));
298 ESP_FREE(recv_data.data);
299 } else {
300 ret = ESP_FAIL;
301 }
302 } while (ret != ESP_OK && retry_count-- > 0 && iperf_data->type != IPERF_BANDWIDTH_STOP_ACK);
303
304 if (ret != ESP_OK) {
305 ESP_LOGW(TAG, "<%s> Receive responder response failed", esp_err_to_name(ret));
306 } else {
307 uint32_t write_count = iperf_data->seq > 0 ? iperf_data->seq - 1 : 0;
308 uint32_t lost_count = total_count - write_count;
309 double total_len = (total_count * g_iperf_cfg.packet_len) / 1e6;
310
311 if (total_count && write_count && spend_time_ms) {
312 ESP_LOGI(TAG, "initiator Report:");
313 ESP_LOGI(TAG, "[ ID] Interval Transfer Bandwidth Jitter Lost/Total Datagrams RSSI Channel");
314 ESP_LOGI(TAG, "[000] %2d-%2d sec %2.2f MBytes %0.2f Mbps %0.2f ms %d/%d (%0.2f%%) %d %d",
315 0, spend_time_ms / 1000, total_len, total_len * 8 * 1000 / spend_time_ms, spend_time_ms * 1.0 / write_count,
316 lost_count, total_count, lost_count * 100.0 / total_count, rx_ctrl.rssi, rx_ctrl.channel);
317 }
318 }
319
320 if (!g_iperf_cfg.frame_head.broadcast) {
321 espnow_del_peer(g_iperf_cfg.addr);
322 }
323
324 ESP_FREE(iperf_data);
325 g_iperf_cfg.finish = true;
326
327 espnow_set_config_for_data_type(ESPNOW_DATA_TYPE_RESERVED, 0, NULL);
328 if (g_iperf_queue) {
329 iperf_recv_data_t tmp_data = { 0 };
330
331 while (xQueueReceive(g_iperf_queue, &tmp_data, 0)) {
332 ESP_FREE(tmp_data.data);
333 }
334
335 vQueueDelete(g_iperf_queue);
336 g_iperf_queue = NULL;
337 }
338
339 vTaskDelete(NULL);
340}

342static esp_err_t espnow_iperf_responder(uint8_t *src_addr, void *data,
343 size_t size, wifi_pkt_rx_ctrl_t *rx_ctrl)
344{
345 ESP_PARAM_CHECK(src_addr);
346 ESP_PARAM_CHECK(data);
347 ESP_PARAM_CHECK(size);
348 ESP_PARAM_CHECK(rx_ctrl);
349
350 esp_err_t ret = ESP_OK;
351 espnow_iperf_data_t *iperf_data = (espnow_iperf_data_t *)data;
352 static int64_t start_time;
353 static uint32_t recv_count;
354 static int64_t report_time;
355 static uint32_t report_count;
356
357 memcpy(g_iperf_cfg.addr, src_addr, 6);
358
359 if (!g_iperf_cfg.finish) {
360 recv_count++;
361
362 if (iperf_data->seq == 0) {
363 recv_count = 0;
364 start_time = esp_timer_get_time();
365 report_time = start_time + g_iperf_cfg.report_interval * 1000 * 1000;
366 report_count = 0;
367 }
368
369 if (iperf_data->type == IPERF_BANDWIDTH && esp_timer_get_time() >= report_time) {
370 uint32_t report_time_s = (report_time - start_time) / (1000 * 1000);
371 double report_size = (recv_count - report_count) * size / 1e6;
372 ESP_LOGI(TAG, "["MACSTR"] %2d-%2d sec %2.2f MBytes %0.2f Mbps %d dbm",
373 MAC2STR(g_iperf_cfg.addr), report_time_s - g_iperf_cfg.report_interval, report_time_s,
374 report_size, report_size * 8 / g_iperf_cfg.report_interval, rx_ctrl->rssi);
375
376 report_time = esp_timer_get_time() + g_iperf_cfg.report_interval * 1000 * 1000;
377 report_count = recv_count;
378 } else if (iperf_data->type == IPERF_PING) {
379 ESP_LOGV(TAG, "Recv IPERF_PING, seq: %d, recv_count: %d", iperf_data->seq, recv_count);
380 iperf_data->type = IPERF_PING_ACK;
381
382 if (g_iperf_cfg.gpio_num >= 0) {
383 gpio_set_level(g_iperf_cfg.gpio_num, 0);
384 }
385
386 if (!g_iperf_cfg.frame_head.broadcast) {
387 espnow_add_peer(g_iperf_cfg.addr, NULL);
388 }
389
390 ret = espnow_send(g_iperf_cfg.type, g_iperf_cfg.addr, iperf_data,
391 size, &g_iperf_cfg.frame_head, portMAX_DELAY);
392
393 if (!g_iperf_cfg.frame_head.broadcast) {
394 espnow_del_peer(g_iperf_cfg.addr);
395 }
396
397 if (g_iperf_cfg.gpio_num >= 0) {
398 gpio_set_level(g_iperf_cfg.gpio_num, 1);
399 }
400
401 ESP_ERROR_RETURN(ret != ESP_OK, ret, "<%s> espnow_send", esp_err_to_name(ret));
402 } else if (iperf_data->type == IPERF_BANDWIDTH_STOP) {
403 uint32_t total_count = iperf_data->seq + 1;
404 uint32_t lost_count = total_count - recv_count;
405 double total_len = (total_count * size) / 1e6;
406 uint32_t spend_time_ms = (esp_timer_get_time() - start_time) / 1000;
407
408 ESP_LOGI(TAG, "[ ID] Interval Transfer Bandwidth Jitter Lost/Total Datagrams");
409 ESP_LOGI(TAG, "[000] %2d-%2d sec %2.2f MBytes %0.2f Mbps %0.2f ms %d/%d (%0.2f%%)",
410 0, spend_time_ms / 1000, total_len, total_len * 8 * 1000 / spend_time_ms, spend_time_ms * 1.0 / recv_count,
411 lost_count, total_count, lost_count * 100.0 / total_count);
412
413 iperf_data->seq = recv_count;
414 iperf_data->type = IPERF_BANDWIDTH_STOP_ACK;
415 ESP_LOGD(TAG, "iperf_data->seq: %d", iperf_data->seq);
416
417 espnow_frame_head_t frame_head = {
418 .filter_adjacent_channel = true,
419 };
420
421 espnow_add_peer(g_iperf_cfg.addr, NULL);
422 ret = espnow_send(g_iperf_cfg.type, g_iperf_cfg.addr, iperf_data,
423 sizeof(espnow_iperf_data_t), &frame_head, portMAX_DELAY);
424 ESP_ERROR_RETURN(ret != ESP_OK, ret, "<%s> espnow_send", esp_err_to_name(ret));
425
426 espnow_del_peer(g_iperf_cfg.addr);
427 }
428 }
429
430 return ESP_OK;
431}

439static void espnow_iperf_ping_task(void *arg)
440{
441 esp_err_t ret = ESP_OK;
442 espnow_iperf_data_t *iperf_data = ESP_CALLOC(1, g_iperf_cfg.packet_len);
443 size_t send_count = 0;
444 size_t recv_count = 0;
445 uint32_t spend_time_ms = 0;
446 int64_t s_time = esp_timer_get_time();
447
448 size_t recv_size = 0;
449 uint32_t max_time_ms = 0;
450 uint32_t min_time_ms = UINT32_MAX;
451 uint32_t res_time_ms = 0;
452 wifi_pkt_rx_ctrl_t rx_ctrl = {0};
453
454 if (!g_iperf_cfg.frame_head.broadcast) {
455 espnow_add_peer(g_iperf_cfg.addr, NULL);
456 }
457
458 for (; send_count < g_iperf_cfg.ping_count && !g_iperf_cfg.finish; send_count ++) {
459 iperf_data->seq = send_count;
460 iperf_data->type = IPERF_PING;
461
462 if (g_iperf_cfg.gpio_num >= 0) {
463 gpio_set_level(g_iperf_cfg.gpio_num, 0);
464 }
465
466 int64_t start_time = esp_timer_get_time();
467 ret = espnow_send(g_iperf_cfg.type, g_iperf_cfg.addr, iperf_data,
468 g_iperf_cfg.packet_len, &g_iperf_cfg.frame_head, portMAX_DELAY);
469 ESP_ERROR_CONTINUE(ret != ESP_OK, "<%s> espnow_send", esp_err_to_name(ret));
470
471 do {
472 iperf_recv_data_t recv_data = { 0 };
473 if (g_iperf_queue && xQueueReceive(g_iperf_queue, &recv_data, pdMS_TO_TICKS(3000)) == pdPASS) {
474 ret = ESP_OK;
475 memcpy(iperf_data, recv_data.data, recv_data.size);
476 memcpy(g_iperf_cfg.addr, recv_data.src_addr, 6);
477 memcpy(&rx_ctrl, &recv_data.rx_ctrl, sizeof(wifi_pkt_rx_ctrl_t));
478 recv_size = recv_data.size;
479 ESP_FREE(recv_data.data);
480 } else {
481 ret = ESP_FAIL;
482 }
483
484 if (ret == ESP_OK && (iperf_data->type != IPERF_PING_ACK || iperf_data->seq != send_count)) {
485 ESP_LOGW(TAG, "data_size: %d, send_seq: %d, recv_seq: %d", recv_size, send_count, iperf_data->seq);
486 }
487 } while (ret == ESP_OK && (iperf_data->type != IPERF_PING_ACK || iperf_data->seq != send_count));
488
489 if (ret != ESP_OK) {
490 ESP_LOGW(TAG, "seq=%d Destination Host Unreachable", iperf_data->seq);
491 continue;
492 }
493
494 if (g_iperf_cfg.gpio_num >= 0) {
495 gpio_set_level(g_iperf_cfg.gpio_num, 1);
496 }
497
498 res_time_ms = (uint32_t)((esp_timer_get_time() - start_time) / 1000);
499 spend_time_ms += res_time_ms;
500
501 max_time_ms = MAX(max_time_ms, res_time_ms);
502 min_time_ms = MIN(min_time_ms, res_time_ms);
503
504 recv_count++;
505 ESP_LOGI(TAG, "%d bytes from " MACSTR ": seq=%d rssi=%d time=%d ms",
506 recv_size, MAC2STR(g_iperf_cfg.addr), iperf_data->seq, rx_ctrl.rssi, res_time_ms);
507
508 ESP_ERROR_CONTINUE(ret != ESP_OK, "<%s> espnow_recv", esp_err_to_name(ret));
509 }
510
511 ESP_LOGI(TAG, "initiator ping report:");
512 ESP_LOGI(TAG, "ping statistics for " MACSTR, MAC2STR(g_iperf_cfg.addr));
513 ESP_LOGI(TAG, "%d packets transmitted, %d received, %0.2f%% packet loss, time: total %d ms, max: %d, min: %d, average %0.2f ms",
514 send_count, recv_count, (send_count - recv_count) * 100.0 / send_count,
515 (int)((esp_timer_get_time() - s_time) / 1000), max_time_ms, min_time_ms, spend_time_ms * 1.0 / recv_count);
516
517 g_iperf_cfg.finish = true;
518
519 if (!g_iperf_cfg.frame_head.broadcast) {
520 espnow_del_peer(g_iperf_cfg.addr);
521 }
522
523 espnow_set_config_for_data_type(ESPNOW_DATA_TYPE_RESERVED, 0, NULL);
524 if (g_iperf_queue) {
525 iperf_recv_data_t tmp_data = { 0 };
526

533 }

535 ESP_FREE(iperf_data);
