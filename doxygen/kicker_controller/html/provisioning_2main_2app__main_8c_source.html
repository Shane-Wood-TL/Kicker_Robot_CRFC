105static esp_err_t app_initiator_init()
106{
107 esp_err_t ret = ESP_OK;
108 wifi_pkt_rx_ctrl_t rx_ctrl = {0};
109 espnow_prov_initiator_t initiator_info = {
110 .product_id = "initiator_test",
111 };
112 espnow_addr_t responder_addr = {0};
113 espnow_prov_responder_t responder_info = {0};
114
115 for (;;) {
116 ret = espnow_prov_initiator_scan(responder_addr, &responder_info, &rx_ctrl, portMAX_DELAY);
117 ESP_ERROR_CONTINUE(ret != ESP_OK, "espnow_prov_responder_beacon");
118
119 ESP_LOGI(TAG, "MAC: "MACSTR", Channel: %d, RSSI: %d, Product_id: %s, Device Name: %s",
120 MAC2STR(responder_addr), rx_ctrl.channel, rx_ctrl.rssi,
121 responder_info.product_id, responder_info.device_name);
122
123 ret = espnow_prov_initiator_send(responder_addr, &initiator_info, app_espnow_prov_recv_cb, pdMS_TO_TICKS(3 * 1000));
124 ESP_ERROR_CONTINUE(ret != ESP_OK, "<%s> espnow_prov_initiator_send", esp_err_to_name(ret));
125
126 break;
127 }
128
129 return ESP_OK;
130}
