152static bool queue_over_write(espnow_msg_id_t msg_id, const void *const data, size_t data_len, void *arg, TickType_t xTicksToWait)
153{
154 if (msg_id == ESPNOW_EVENT_RECV_ACK) {
155 if (!g_ack_queue) {
156 return false;
157 }
158 return xQueueSend(g_ack_queue, &data, xTicksToWait);
159 } else {
160 if (!g_espnow_queue) {
161 return false;
162 }
163
164 espnow_event_ctx_t espnow_event = {
165 .msg_id = msg_id,
166 .data_len = data_len,
167 .data = (void *)data,
168 .handle = arg
169 };
170
171 return xQueueSend(g_espnow_queue, &espnow_event, xTicksToWait);

175/**< callback function of receiving ESPNOW data */
176#if ESP_IDF_VERSION >= ESP_IDF_VERSION_VAL(5, 0, 1)
177void espnow_recv_cb(const esp_now_recv_info_t *recv_info, const uint8_t *data, int size)
178#else
179void espnow_recv_cb(const uint8_t *addr, const uint8_t *data, int size)
180#endif
181{
182 espnow_data_t *espnow_data = (espnow_data_t *)data;
183 wifi_pkt_rx_ctrl_t *rx_ctrl = NULL;
184
185#if ESP_IDF_VERSION >= ESP_IDF_VERSION_VAL(5, 0, 1)
186 uint8_t * addr = recv_info->src_addr;
187 rx_ctrl = recv_info->rx_ctrl;
188#else
189 wifi_promiscuous_pkt_t *promiscuous_pkt = (wifi_promiscuous_pkt_t *)(data - sizeof(wifi_pkt_rx_ctrl_t) - sizeof(espnow_frame_format_t));
190 rx_ctrl = &promiscuous_pkt->rx_ctrl;
191#endif
192
193 ESP_LOG_BUFFER_HEXDUMP(TAG, data, size, ESP_LOG_DEBUG);
194 ESP_LOGD(TAG, "[%s, %d], " MACSTR ", rssi: %d, size: %d, total: %d - %d, type: %d, addr: %02x, g_msg_magic_cache_next: %d",
195 __func__, __LINE__, MAC2STR(addr), rx_ctrl->rssi, size, espnow_data->size, sizeof(espnow_data_t), espnow_data->type, addr[5], g_msg_magic_cache_next);
196
197 /**< Filter ESP-NOW packets not generated by this project */
198 if (espnow_data->version != ESPNOW_VERSION || (espnow_data->type >= ESPNOW_DATA_TYPE_MAX)
199 || size != espnow_data->size + sizeof(espnow_data_t)
200 || ESPNOW_ADDR_IS_SELF(espnow_data->src_addr)) {
201 ESP_LOGD(TAG, "Receive cb args error, recv_addr: "MACSTR", src_addr: " MACSTR ", data: %p, size: %d",
202 MAC2STR(addr), MAC2STR(espnow_data->src_addr), data, size);
203 return ;
204 }
205
206 espnow_frame_head_t *frame_head = &espnow_data->frame_head;
207
208 /**< Data does not need to be forwarded */
209 if (!g_recv_handle[espnow_data->type].enable
210 && (!g_espnow_config->forward_enable || !frame_head->broadcast || !frame_head->forward_ttl)) {
211 return ;
212 }
213
214 /**< Channel filtering */
215 if (frame_head->filter_adjacent_channel && frame_head->channel != rx_ctrl->channel) {
216 ESP_LOGD(TAG, "Filter adjacent channels, %d != %d", frame_head->channel, rx_ctrl->channel);
217 return ;
218 }
219
220 /**< Rssi filtering */
221 if (frame_head->filter_weak_signal && frame_head->forward_rssi > rx_ctrl->rssi) {
222 ESP_LOGD(TAG, "Filter weak signal strength, %d > %d", frame_head->forward_rssi, rx_ctrl->rssi);
223 return ;
224 }
225
226 /* Security filtering*/
227 if (g_espnow_config && !g_espnow_config->sec_enable && frame_head->security) {
228 ESP_LOGD(TAG, "Filter security frame");
229 return;
230 }
231
232 if (g_recv_handle[espnow_data->type].enable
233 && espnow_data->type != ESPNOW_DATA_TYPE_ACK && espnow_data->type != ESPNOW_DATA_TYPE_GROUP
234 && frame_head->ack && ESPNOW_ADDR_IS_SELF(espnow_data->dest_addr)) {
235 espnow_data_t *ack_data = ESP_CALLOC(1, sizeof(espnow_data_t));
236 if (!ack_data) {
237 return;
238 }
239 ack_data->version = ESPNOW_VERSION;
240 ack_data->type = ESPNOW_DATA_TYPE_ACK;
241 ack_data->size = 0;
242 memcpy(&ack_data->frame_head, frame_head, sizeof(espnow_frame_head_t));
243 memcpy(ack_data->src_addr, ESPNOW_ADDR_SELF, 6);
244 memcpy(ack_data->dest_addr, espnow_data->src_addr, 6);
245
246 ack_data->frame_head.retransmit_count = 1;
247 ack_data->frame_head.broadcast = 1;
248
249 if (!g_espnow_queue || queue_over_write(ESPNOW_EVENT_SEND_ACK, ack_data, sizeof(espnow_data_t), NULL, g_espnow_config->send_max_timeout) != pdPASS) {
250 ESP_LOGW(TAG, "[%s, %d] Send event queue failed", __func__, __LINE__);
251 ESP_FREE(ack_data);
252 }
253 }
254
255 if (!frame_head->security) {
256 for (size_t i = 0, index = g_msg_magic_cache_next; i < ESPNOW_MSG_CACHE;
257 i++, index = (g_msg_magic_cache_next + i) % ESPNOW_MSG_CACHE) {
258 if (g_msg_magic_cache[index].type == espnow_data->type
259 && g_msg_magic_cache[index].magic == frame_head->magic) {
260 return ;
261 }
262 }
263 } else {
264 for (size_t i = 0, index = g_msg_magic_cache_next; i < ESPNOW_MSG_CACHE;
265 i++, index = (g_msg_magic_cache_next + i) % ESPNOW_MSG_CACHE) {
266 if (g_msg_magic_sec_cache[index].type == espnow_data->type
267 && g_msg_magic_sec_cache[index].magic == frame_head->magic) {
268 return ;
269 }
270 }
271 }
272#if CONFIG_IDF_TARGET_ESP32C6
273 ESP_LOGD(TAG, "[%s, %d]: " MACSTR ", rssi: %d, channel: %d/%d, size: %d, %s, magic: 0x%x, ack: %d",
274 __func__, __LINE__, MAC2STR(espnow_data->dest_addr), rx_ctrl->rssi, rx_ctrl->channel,
275 rx_ctrl->second, espnow_data->size, espnow_data->payload,
276 frame_head->magic, espnow_data->frame_head.ack);
277#else
278 ESP_LOGD(TAG, "[%s, %d]: " MACSTR ", rssi: %d, channel: %d/%d, size: %d, %s, magic: 0x%x, ack: %d",
279 __func__, __LINE__, MAC2STR(espnow_data->dest_addr), rx_ctrl->rssi, rx_ctrl->channel,
280 rx_ctrl->secondary_channel, espnow_data->size, espnow_data->payload,
281 frame_head->magic, espnow_data->frame_head.ack);
282#endif
283
284 if (!g_recv_handle[espnow_data->type].enable) {
285 goto FORWARD_DATA;
286 }
287
288 if (espnow_data->type == ESPNOW_DATA_TYPE_ACK) {
289 if (!ESPNOW_ADDR_IS_SELF(espnow_data->dest_addr)) {
290#ifdef CONFIG_ESPNOW_DATA_FAST_ACK
291 if (g_recv_handle[ESPNOW_DATA_TYPE_ACK].handle) {
292 g_recv_handle[ESPNOW_DATA_TYPE_ACK].handle(espnow_data->src_addr, (void *)frame_head, sizeof(espnow_frame_head_t), NULL);
293 }
294#endif
295 goto FORWARD_DATA;
296 }
297
298 ESP_LOGD(TAG, ">[%s, %d]: broadcast: %d, dest_addr: " MACSTR, __func__, __LINE__, frame_head->broadcast,
299 MAC2STR(espnow_data->dest_addr));
300
301 uint32_t *magic = ESP_MALLOC(sizeof(uint32_t));
302 *magic = frame_head->magic;
303
304 if (!g_ack_queue || queue_over_write(ESPNOW_EVENT_RECV_ACK, magic, sizeof(uint32_t), NULL, g_espnow_config->send_max_timeout) != pdPASS) {
305 ESP_LOGW(TAG, "[%s, %d] Send event queue failed", __func__, __LINE__);
306 ESP_FREE(magic);
307 return ;
308 };
309
310 goto EXIT;
311 } else if (espnow_data->type == ESPNOW_DATA_TYPE_GROUP) {
312 espnow_group_info_t *group_info = (espnow_group_info_t *) espnow_data->payload;
313 bool set_group_flag = false;
314
315 ESP_LOGD(TAG, ">1.2< group_id: " MACSTR ", dest_addr: " MACSTR,
316 MAC2STR(group_info->group_id), MAC2STR(espnow_data->dest_addr));
317 ESP_LOGD(TAG, ">1.2< addrs_num: %d, dest_addr: " MACSTR, group_info->addrs_num,
318 MAC2STR(group_info->addrs_list[0]));
319
320 if (group_info->addrs_num == 1 && ESPNOW_ADDR_IS_BROADCAST(group_info->addrs_list[0])) {
321 set_group_flag = true;
322 } else {
323 if (espnow_data->size < (sizeof(espnow_group_info_t) + group_info->addrs_num * ESPNOW_ADDR_LEN)) {
324 ESP_LOGD(TAG, "[%s, %d] The size %d of the data must match with total size for addrs_num: %d", __func__, __LINE__, espnow_data->size, group_info->addrs_num);
325 return;
326 }
327
328 for (size_t i = 0; i < group_info->addrs_num; i++) {
329 if (ESPNOW_ADDR_IS_SELF(group_info->addrs_list[i])) {
330 set_group_flag = true;
331 break;
332 }
333 }
334 }
335
336 if (set_group_flag) {
337 if (group_info->type) {
338 espnow_add_group(group_info->group_id);
339 } else {
340 espnow_del_group(group_info->group_id);
341 }
342 }
343 } else {
344 if (!frame_head->group && frame_head->broadcast && !ESPNOW_ADDR_IS_BROADCAST(espnow_data->dest_addr)
345 && !ESPNOW_ADDR_IS_SELF(espnow_data->dest_addr)) {
346 goto FORWARD_DATA;
347 } else if (frame_head->group && !espnow_is_my_group(espnow_data->dest_addr)) {
348 ESP_LOGD(TAG, "[%s, %d]: group_num: %d, group_id: " MACSTR,
349 __func__, __LINE__, espnow_get_group_num(), MAC2STR(espnow_data->dest_addr));
350 goto FORWARD_DATA;
351 }
352
353 espnow_pkt_t *q_data = ESP_MALLOC(sizeof(espnow_pkt_t) + espnow_data->size);
354 memcpy(&q_data->rx_ctrl, rx_ctrl, sizeof(wifi_pkt_rx_ctrl_t));
355 memcpy(&q_data->data, espnow_data, size);
356
357 if (frame_head->channel && frame_head->channel != ESPNOW_CHANNEL_ALL) {
358 q_data->rx_ctrl.channel = frame_head->channel;
359 }
360
361 if (queue_over_write(ESPNOW_EVENT_RECEIVE, q_data, sizeof(espnow_pkt_t) + espnow_data->size, NULL, g_espnow_config->send_max_timeout) != pdPASS) {
362 ESP_LOGW(TAG, "[%s, %d] Send event queue failed", __func__, __LINE__);
363 ESP_FREE(q_data);
364 return ;
365 }
366 }
367
368FORWARD_DATA:
369
370 if (g_espnow_config->forward_enable && frame_head->forward_ttl > 0 && frame_head->broadcast
371 && frame_head->forward_rssi <= rx_ctrl->rssi && !ESPNOW_ADDR_IS_SELF(espnow_data->dest_addr)
372 && !ESPNOW_ADDR_IS_SELF(espnow_data->src_addr)) {
373 espnow_data_t *q_data = ESP_MALLOC(size);
374
375 if (!q_data) {
376 return ;
377 }
378
379 memcpy(q_data, espnow_data, size);
380
381 if (frame_head->forward_ttl != ESPNOW_FORWARD_MAX_COUNT) {
382 q_data->frame_head.forward_ttl--;
383 }
384
385 if (!g_espnow_queue || queue_over_write(ESPNOW_EVENT_FORWARD, q_data, size, NULL, g_espnow_config->send_max_timeout) != pdPASS) {
386 ESP_LOGW(TAG, "[%s, %d] Send event queue failed", __func__, __LINE__);
387 ESP_FREE(q_data);
388 return ;
389 }
390 }
391
392EXIT:
393 g_msg_magic_cache_next = (g_msg_magic_cache_next + 1) % ESPNOW_MSG_CACHE;
394 if (!frame_head->security) {
395 g_msg_magic_cache[g_msg_magic_cache_next].type = espnow_data->type;
396 g_msg_magic_cache[g_msg_magic_cache_next].magic = frame_head->magic;
397 } else {
398 g_msg_magic_sec_cache[g_msg_magic_cache_next].type = espnow_data->type;
399 g_msg_magic_sec_cache[g_msg_magic_cache_next].magic = frame_head->magic;
400 }
401}

465static esp_err_t espnow_send_process(int count, espnow_data_t *espnow_data, uint32_t wait_ticks, bool *ack)
466{
467 ESP_PARAM_CHECK(espnow_data);
468
469 espnow_frame_head_t *frame_head = &espnow_data->frame_head;
470
471 g_buffered_num ++;
472
473 /* Wait send cb when max buffered num or ack is enable or unicast*/
474 if (g_buffered_num >= MAX_BUFFERED_NUM || frame_head->ack || !frame_head->broadcast) {
475 /**< For unicast packet, Waiting send complete ack from mac layer */
476 /**< For broadcast packet, Waiting send ok from mac layer */
477 EventBits_t uxBits = xEventGroupWaitBits(g_event_group, SEND_CB_OK | SEND_CB_FAIL,
478 pdTRUE, pdFALSE, MIN(wait_ticks, g_espnow_config->send_max_timeout));
479 if ((uxBits & SEND_CB_OK) == SEND_CB_OK) {
480 if (!frame_head->broadcast && !frame_head->ack && ack) {
481 *ack = true;
482 return ESP_OK;
483 }
484#ifdef CONFIG_ESPNOW_LIGHT_SLEEP
485 if (ack) {
486 *ack = true;
487 return ESP_OK;
488 }
489#endif
490 } else {
491 return ESP_FAIL;
492 }
493 }
494
495 if (frame_head->ack && !ESPNOW_ADDR_IS_BROADCAST(espnow_data->dest_addr) && ack) {
496 uint32_t *ack_magic = NULL;
497
498 /* retry backoff time (2,4,8,16,32,64,100,100,...)ms */
499 uint32_t delay_ms = (count < 6 ? 1 << count : 50) * SEND_DELAY_UNIT_MSECS;
500
501 do {
502 vTaskDelay(pdMS_TO_TICKS(SEND_DELAY_UNIT_MSECS));
503 while (g_ack_queue && xQueueReceive(g_ack_queue, &ack_magic, 0) == pdPASS) {
504 if (*ack_magic == frame_head->magic) {
505 espnow_data->frame_head.ack = false;
506 ESP_FREE(ack_magic);
507 *ack = true;
508 return ESP_OK;
509 }
510
511 ESP_FREE(ack_magic);
512 }
513
514 delay_ms -= SEND_DELAY_UNIT_MSECS;
515 } while (delay_ms > 0);
516
517 return ESP_ERR_WIFI_TIMEOUT;
518 }
519
520 return ESP_OK;
521}

523esp_err_t espnow_send(espnow_data_type_t type, const espnow_addr_t dest_addr, const void *data,
524 size_t size, const espnow_frame_head_t *data_head, TickType_t wait_ticks)
525{
526 ESP_PARAM_CHECK(dest_addr);
527 ESP_PARAM_CHECK(data);
528 ESP_PARAM_CHECK(type < ESPNOW_DATA_TYPE_MAX);
529 ESP_PARAM_CHECK(size <= ESPNOW_DATA_LEN);
530 ESP_ERROR_RETURN(!g_espnow_config, ESP_ERR_ESPNOW_NOT_INIT, "ESPNOW is not initialized");
531
532 esp_err_t ret = ESP_FAIL;
533 TickType_t write_ticks = 0;
534 uint32_t start_ticks = xTaskGetTickCount();
535 uint8_t primary = 0;
536 wifi_second_chan_t second = 0;
537 espnow_frame_head_t *frame_head = NULL;
538 espnow_data_t *espnow_data = NULL;
539 bool enc = false;
540
541 if (g_espnow_config->sec_enable && (data_head ? data_head->security : g_espnow_frame_head_default.security)
542 && type != ESPNOW_DATA_TYPE_ACK && type != ESPNOW_DATA_TYPE_FORWARD
543 && type != ESPNOW_DATA_TYPE_SECURITY_STATUS && type != ESPNOW_DATA_TYPE_SECURITY) {
544 ESP_ERROR_RETURN(!(g_espnow_sec && g_espnow_sec->state == ESPNOW_SEC_OVER), ESP_FAIL, "Security key is not set");
545 size_t enc_len = 0;
546 espnow_data = ESP_MALLOC(sizeof(espnow_data_t) + size + g_espnow_sec->tag_len + IV_LEN);
547 if (!espnow_data) {
548 ESP_LOGE(TAG, "Not enough memory!");
549 return ret;
550 }
551 uint8_t key_info[APP_KEY_LEN];
552 uint8_t iv_info[IV_LEN];
553
554 ret = espnow_get_key(key_info);
555 if (ret) {
556 ESP_FREE(espnow_data);
557 ESP_LOGE(TAG, "Get security key fail for encrypt, err_name: %s", esp_err_to_name(ret));
558 return ret;
559 }
560
561 esp_fill_random(iv_info, IV_LEN);
562 memcpy(key_info + KEY_LEN, iv_info, IV_LEN);
563 espnow_set_key(key_info);
564
565 ret = espnow_sec_auth_encrypt(g_espnow_sec, data, size, espnow_data->payload, size + g_espnow_sec->tag_len, &enc_len, g_espnow_sec->tag_len);
566 espnow_data->size = enc_len + IV_LEN;
567 if (ret == ESP_OK) {
568 enc = 1;
569 memcpy(espnow_data->payload + enc_len, iv_info, IV_LEN);
570 } else {
571 ESP_FREE(espnow_data);
572 ESP_LOGE(TAG, "Security encrypt return error");
573 return ret;
574 }
575 } else {
576 espnow_data = ESP_MALLOC(sizeof(espnow_data_t) + size);
577 if (!espnow_data) {
578 ESP_LOGE(TAG, "Not enough memory!");
579 return ret;
580 }
581 espnow_data->size = size;
582 memcpy(espnow_data->payload, data, size);
583 }
584
585 if (data_head) {
586 memcpy(&espnow_data->frame_head, data_head, sizeof(espnow_frame_head_t));
587 } else {
588 memcpy(&espnow_data->frame_head, &g_espnow_frame_head_default, sizeof(espnow_frame_head_t));
589 }
590 frame_head = &espnow_data->frame_head;
591
592 if (enc) {
593 frame_head->security = true;
594 }
595
596 if (!frame_head->magic) {
597 frame_head->magic = esp_random();
598 }
599
600 if (!frame_head->broadcast && ESPNOW_ADDR_IS_BROADCAST(dest_addr)) {
601 frame_head->broadcast = true;
602 }
603
604 if (frame_head->retransmit_count == 0) {
605 frame_head->retransmit_count = 1;
606 }
607
608 espnow_data->version = ESPNOW_VERSION;
609 espnow_data->type = type;
610 memcpy(espnow_data->dest_addr, dest_addr, sizeof(espnow_data->dest_addr));
611 memcpy(espnow_data->src_addr, ESPNOW_ADDR_SELF, sizeof(espnow_data->src_addr));
612
613 ESP_LOGD(TAG, "[%s, %d] addr: " MACSTR", size: %d, count: %d, rssi: %d, data: %s, magic: 0x%x",
614 __func__, __LINE__, MAC2STR(dest_addr), espnow_data->size, frame_head->retransmit_count,
615 frame_head->forward_rssi, espnow_data->payload, frame_head->magic);
616
617 /**< Wait for other tasks to be sent before send ESP-NOW data */
618 if (xSemaphoreTake(g_send_lock, pdMS_TO_TICKS(wait_ticks)) != pdPASS) {
619 ESP_FREE(espnow_data);
620 return ESP_ERR_TIMEOUT;
621 }
622
623 ret = esp_wifi_get_channel(&primary, &second);
624 ESP_ERROR_GOTO(ret != ESP_OK, EXIT, "esp_wifi_get_channel, err_name: %s", esp_err_to_name(ret));
625
626 if (frame_head->channel == 0) {
627 frame_head->channel = primary;
628 } else if (frame_head->channel > 0 && frame_head->channel < ESPNOW_CHANNEL_ALL && frame_head->channel != primary) {
629 if (g_set_channel_flag) {
630 ESP_ERROR_GOTO(frame_head->channel >= g_self_country.schan + g_self_country.nchan, EXIT,
631 "Can't set channel %d, not allowed in country %c%c%c.",
632 frame_head->channel, g_self_country.cc[0], g_self_country.cc[1], g_self_country.cc[2]);
633 ret = esp_wifi_set_channel(frame_head->channel, WIFI_SECOND_CHAN_NONE);
634 ESP_ERROR_GOTO(ret != ESP_OK, EXIT, "esp_wifi_set_channel, err_name: %s", esp_err_to_name(ret));
635 } else {
636 ESP_LOGE(TAG, "Can't set channel %d, current is %d", frame_head->channel, primary);
637 ret = ESP_FAIL;
638 goto EXIT;
639 }
640 }
641
642 const uint8_t *addr = (frame_head->broadcast) ? ESPNOW_ADDR_BROADCAST : dest_addr;
643
644 for (int count = 0; count < frame_head->retransmit_count; ++count) {
645 for (int i = 0; i == 0
646 || (g_set_channel_flag && frame_head->channel == ESPNOW_CHANNEL_ALL && i < g_self_country.nchan); ++i) {
647
648 if (g_set_channel_flag && frame_head->channel == ESPNOW_CHANNEL_ALL) {
649 esp_wifi_set_channel(g_self_country.schan + i, WIFI_SECOND_CHAN_NONE);
650 }
651
652 xEventGroupClearBits(g_event_group, SEND_CB_OK | SEND_CB_FAIL);
653
654 if (frame_head->ack) {
655 count = 0;
656 }
657
658 do {
659 ret = esp_now_send(addr, (uint8_t *)espnow_data, espnow_data->size + sizeof(espnow_data_t));
660
661 if (ret == ESP_OK) {
662 bool ack = 0;
663 write_ticks = (wait_ticks == portMAX_DELAY) ? portMAX_DELAY :
664 xTaskGetTickCount() - start_ticks < wait_ticks ?
665 wait_ticks - (xTaskGetTickCount() - start_ticks) : 0;
666 ret = espnow_send_process(count, espnow_data, write_ticks, &ack);
667 if (ret == ESP_OK && ack) {
668 goto EXIT;
669 }
670 }
671
672 } while (frame_head->ack && ++count < frame_head->retransmit_count);
673
674 ESP_ERROR_CONTINUE(ret != ESP_OK, "[%s, %d] <%s> esp_now_send, channel: %d",
675 __func__, __LINE__, esp_err_to_name(ret), frame_head->channel == ESPNOW_CHANNEL_ALL ? g_self_country.schan + i : primary);
676 }
677 }
678
679EXIT:
680
681#ifdef CONFIG_ESPNOW_AUTO_RESTORE_CHANNEL
682 if (g_set_channel_flag && frame_head->channel != primary) {
683 esp_wifi_set_channel(primary, second);
684 }
685#endif
686
687 xSemaphoreGive(g_send_lock);
688
689 if (frame_head->ack && !ESPNOW_ADDR_IS_BROADCAST(dest_addr)) {
690 uint32_t *ack_magic = NULL;
691 write_ticks = (wait_ticks == portMAX_DELAY) ? portMAX_DELAY :
692 xTaskGetTickCount() - start_ticks < wait_ticks ?
693 wait_ticks - (xTaskGetTickCount() - start_ticks) : 0;
694
695 while (g_ack_queue && xQueueReceive(g_ack_queue, &ack_magic, MIN(write_ticks,
696 g_espnow_config->send_max_timeout)) == pdPASS) {
697 if (*ack_magic == frame_head->magic) {
698 ESP_FREE(espnow_data);
699 ESP_FREE(ack_magic);
700 return ESP_OK;
701 }
702
703 ESP_FREE(ack_magic);
704 }
705
706 ret = ESP_ERR_WIFI_TIMEOUT;
707 }
708
709 ESP_FREE(espnow_data);
710
711 return ret;
712}

714esp_err_t espnow_set_group(const uint8_t addrs_list[][ESPNOW_ADDR_LEN], size_t addrs_num,
715 const uint8_t group_id[ESPNOW_ADDR_LEN], espnow_frame_head_t *data_head,
716 bool type, TickType_t wait_ticks)
717{
718 esp_err_t ret = ESP_OK;
719 uint32_t start_ticks = xTaskGetTickCount();
720
721 espnow_data_t *espnow_data = ESP_MALLOC(addrs_num > 35 ? ESPNOW_PAYLOAD_LEN : sizeof(espnow_data_t) + sizeof(espnow_group_info_t) + addrs_num * ESPNOW_ADDR_LEN);
722 espnow_frame_head_t *frame_head = &espnow_data->frame_head;
723 espnow_group_info_t *group_info = (espnow_group_info_t *) espnow_data->payload;

725 uint8_t primary = 0;

728 espnow_data->version = ESPNOW_VERSION;
729 espnow_data->type = ESPNOW_DATA_TYPE_GROUP;
730 memcpy(espnow_data->dest_addr, ESPNOW_ADDR_BROADCAST, ESPNOW_ADDR_LEN);
731 memcpy(espnow_data->src_addr, ESPNOW_ADDR_SELF, ESPNOW_ADDR_LEN);
732
733 if (data_head) {
734 memcpy(frame_head, data_head, sizeof(espnow_frame_head_t));
735 } else {
736 memcpy(frame_head, &g_espnow_frame_head_default, sizeof(espnow_frame_head_t));
737 }
738
739 if (!frame_head->magic) {
740 frame_head->magic = esp_random();
741 }
742
743 ret = esp_wifi_get_channel(&primary, &second);

745

757 }
758
759 group_info->type = type;
760 memcpy(group_info->group_id, group_id, ESPNOW_ADDR_LEN);
761
762 for (int i = 0; addrs_num > 0; ++i) {
763 size_t send_addrs_num = (addrs_num > 32) ? 32 : addrs_num;
764 addrs_num -= send_addrs_num;
765 espnow_data->size = sizeof(espnow_group_info_t) + send_addrs_num * ESPNOW_ADDR_LEN;
766 group_info->addrs_num = send_addrs_num;
767 memcpy(group_info->addrs_list, addrs_list + i * 32, send_addrs_num * ESPNOW_ADDR_LEN);
768 frame_head->magic += i;
769
770 for (int count = 0; count < frame_head->retransmit_count; ++count) {
771 for (int i = 0; i == 0
772 || (frame_head->channel == ESPNOW_CHANNEL_ALL && i < g_self_country.nchan && g_set_channel_flag); ++i) {
773
774 if (frame_head->channel == ESPNOW_CHANNEL_ALL && g_set_channel_flag) {
775 esp_wifi_set_channel(g_self_country.schan + i, WIFI_SECOND_CHAN_NONE);
776 }
777
778 ret = esp_now_send(ESPNOW_ADDR_BROADCAST, (uint8_t *)espnow_data, espnow_data->size + sizeof(espnow_data_t));
779
780 if (ret == ESP_OK) {
781 TickType_t write_ticks = (wait_ticks == portMAX_DELAY) ? portMAX_DELAY :
782 xTaskGetTickCount() - start_ticks < wait_ticks ?
783 wait_ticks - (xTaskGetTickCount() - start_ticks) : 0;
784 ret = espnow_send_process(count, espnow_data, write_ticks, NULL);
785 }
786
787 ESP_ERROR_CONTINUE(ret != ESP_OK, "[%s, %d] <%s> esp_now_send, channel: %d",
788 __func__, __LINE__, esp_err_to_name(ret), g_self_country.schan + i);
789 }
790 }
791 }
792
793EXIT:
794
795 if (frame_head->channel != primary && g_set_channel_flag) {
796 esp_wifi_set_channel(primary, second);
797 }
798
799 ESP_FREE(espnow_data);
800 return ESP_OK;
801}

803static esp_err_t espnow_recv_process(espnow_pkt_t *q_data)
804{
805 ESP_PARAM_CHECK(q_data);
806 ESP_ERROR_RETURN(!g_espnow_config, ESP_ERR_ESPNOW_NOT_INIT, "ESPNOW is not initialized");
807
808 esp_err_t ret = 0;
809 espnow_data_t *espnow_data = &q_data->data;
810 espnow_frame_head_t *frame_head = &espnow_data->frame_head;
811 uint8_t src_addr[6] = {0};
812 size_t size = 0;
813 uint8_t *data = ESP_MALLOC(ESPNOW_PAYLOAD_LEN);
814 wifi_pkt_rx_ctrl_t rx_ctrl = {0};
815
816 ESP_LOGD(TAG, "[%s, %d], " MACSTR ", magic: 0x%04x, type: %d, size: %d, %s", __func__, __LINE__, MAC2STR(espnow_data->src_addr),
817 frame_head->magic, espnow_data->type, espnow_data->size, espnow_data->payload);
818
819 if (ret == ESP_OK) {
820 /* Check security */
821 if (frame_head->security) {
822 if (g_espnow_config->sec_enable) {
823 if (g_espnow_dec && g_espnow_dec->state == ESPNOW_SEC_OVER) {
824 uint8_t key_info[APP_KEY_LEN];
825
826 ret = espnow_get_dec_key(key_info);
827 if (ret) {
828 ESP_LOGE(TAG, "Get security key fail for decrypt, err_name: %s", esp_err_to_name(ret));
829 goto EXIT;
830 }
831 memcpy(key_info + KEY_LEN, espnow_data->payload + (espnow_data->size - IV_LEN), IV_LEN);
832 espnow_set_dec_key(key_info);
833
834 ret = espnow_sec_auth_decrypt(g_espnow_dec, espnow_data->payload, (espnow_data->size - IV_LEN), data, ESPNOW_PAYLOAD_LEN, &size, g_espnow_dec->tag_len);
835 ESP_ERROR_GOTO(ret != ESP_OK, EXIT, "espnow_sec_auth_decrypt, err_name: %s", esp_err_to_name(ret));
836 } else {
837 ESP_LOGE(TAG, "Security key is not set");
838 goto EXIT;
839 }
840 } else {
841 goto EXIT;
842 }
843 } else {
844 size = espnow_data->size;
845 memcpy(data, espnow_data->payload, espnow_data->size);
846 }
847
848 memcpy(src_addr, espnow_data->src_addr, 6);
849
850 rx_ctrl = q_data->rx_ctrl;
851
852 if (g_recv_handle[espnow_data->type].handle) {
853 g_recv_handle[espnow_data->type].handle(src_addr, (void *)data, size, &rx_ctrl);
854 }
855 }
856
857EXIT:
858 ESP_FREE(q_data);
859 ESP_FREE(data);
860 return ret;
861}

863static esp_err_t espnow_send_forward(espnow_data_t *espnow_data)
864{
865 ESP_PARAM_CHECK(espnow_data);
866 ESP_ERROR_RETURN(!g_espnow_config, ESP_ERR_ESPNOW_NOT_INIT, "ESPNOW is not initialized");
867
868 esp_err_t ret = 0;
869 uint8_t primary = 0;
870 wifi_second_chan_t second = 0;
871
872 ESP_LOGD(TAG, "[%s, %d], " MACSTR ", size: %d, %s", __func__, __LINE__,
873 MAC2STR(espnow_data->src_addr), espnow_data->size, espnow_data->payload);
874
875 espnow_frame_head_t *frame_head = &espnow_data->frame_head;
876 const uint8_t *dest_addr = (frame_head->broadcast) ? ESPNOW_ADDR_BROADCAST : espnow_data->dest_addr;
877
878#ifndef CONFIG_ESPNOW_DATA_FAST_ACK
879 if (espnow_data->type == ESPNOW_DATA_TYPE_ACK && g_recv_handle[ESPNOW_DATA_TYPE_ACK].handle) {
880 g_recv_handle[ESPNOW_DATA_TYPE_ACK].handle(espnow_data->src_addr, (void *)frame_head, sizeof(espnow_frame_head_t), NULL);
881 }
882#endif
883
884 /**< Wait for other tasks to be sent before send ESP-NOW data */
885 if (xSemaphoreTake(g_send_lock, g_espnow_config->send_max_timeout) != pdPASS) {
886 ESP_LOGW(TAG, "Wait Sem fail");
887 ESP_FREE(espnow_data);
888 return ESP_ERR_TIMEOUT;
889 }
890
891 if (frame_head->channel == ESPNOW_CHANNEL_ALL && g_set_channel_flag && g_espnow_config->forward_switch_channel) {
892 ESP_ERROR_CHECK(esp_wifi_get_channel(&primary, &second));
893 }
894
895 ESP_LOGD(TAG, "[%s, %d], " MACSTR ", total: %d, type: %d, magic: 0x%x", __func__, __LINE__,
896 MAC2STR(espnow_data->src_addr), espnow_data->size, espnow_data->type, frame_head->magic);
897
898 uint32_t start_ticks = xTaskGetTickCount();
899 uint32_t max_ticks = pdMS_TO_TICKS(g_espnow_config->send_max_timeout);
900 for (int count = 0; !count || ((count < frame_head->retransmit_count) && (max_ticks > (xTaskGetTickCount() - start_ticks))); ++count) {
901 for (int i = 0; i == 0 || (frame_head->channel == ESPNOW_CHANNEL_ALL && i < g_self_country.nchan && g_set_channel_flag && g_espnow_config->forward_switch_channel); ++i) {
902
903 if (frame_head->channel == ESPNOW_CHANNEL_ALL && g_set_channel_flag && g_espnow_config->forward_switch_channel) {
904 esp_wifi_set_channel(g_self_country.schan + i, WIFI_SECOND_CHAN_NONE);
905 }
906
907 ret = esp_now_send(dest_addr, (uint8_t *)espnow_data, sizeof(espnow_data_t) + espnow_data->size);
908
909 if (ret == ESP_OK) {
910 ret = espnow_send_process(count, espnow_data, portMAX_DELAY, NULL);
911 }
912
913
914 ESP_ERROR_CONTINUE(ret != ESP_OK, "[%s, %d] <%s> esp_now_send, channel: %d",
915 __func__, __LINE__, esp_err_to_name(ret), g_self_country.schan + i);
916 }
917 }
918
919 ESP_LOGD(TAG, "[%s, %d], " MACSTR ", size: %d, %s", __func__, __LINE__, MAC2STR(espnow_data->src_addr), espnow_data->size, espnow_data->payload);
920
921 if (frame_head->channel == ESPNOW_CHANNEL_ALL && g_set_channel_flag && g_espnow_config->forward_switch_channel) {
922 esp_wifi_set_channel(primary, second);
923 }
924
925 xSemaphoreGive(g_send_lock);
926 ESP_FREE(espnow_data);
927
928 return ret;
929}

931static void espnow_main_task(void *arg)
932{
933 espnow_event_ctx_t evt_data = { 0 };
934 bool loop_continue = true;
935
936 ESP_LOGI(TAG, "main task entry");
937
938 if (g_espnow_config && g_espnow_config->qsize) {
939 g_espnow_queue = xQueueCreate(g_espnow_config->qsize, sizeof(espnow_event_ctx_t));
940 ESP_ERROR_GOTO(!g_espnow_queue, EXIT, "Create espnow event queue fail");
941
942 if (g_recv_handle[ESPNOW_DATA_TYPE_ACK].enable) {
943 g_ack_queue = xQueueCreate(4, sizeof(uint32_t *));
944 ESP_ERROR_GOTO(!g_ack_queue, EXIT, "Create espnow ack queue fail");
945 }
946 }
947
948 while (g_espnow_config && loop_continue) {
949 if (xQueueReceive(g_espnow_queue, &evt_data, portMAX_DELAY) != pdTRUE) {
950 continue;
951 }
952
953 if (evt_data.msg_id == ESPNOW_EVENT_STOP) {
954 loop_continue = false;
955 continue;
956 }
957
958 if (evt_data.msg_id == ESPNOW_EVENT_SEND_ACK || evt_data.msg_id == ESPNOW_EVENT_FORWARD) {
959 if (espnow_send_forward((espnow_data_t *)(evt_data.data)) != ESP_OK) {
960 ESP_LOGD(TAG, "espnow_send_forward failed");
961 }
962 continue;
963 }
964
965 if (evt_data.msg_id == ESPNOW_EVENT_RECEIVE) {
966 if (espnow_recv_process((espnow_pkt_t *)(evt_data.data)) != ESP_OK) {
967 ESP_LOGD(TAG, "espnow_recv_process");
968 }
969 continue;
970 }
971 }
972
973EXIT:
974 if (g_espnow_queue) {
975 while (xQueueReceive(g_espnow_queue, &evt_data, 0)) {
976 ESP_FREE(evt_data.data);
977 }
978
979 vQueueDelete(g_espnow_queue);
980 g_espnow_queue = NULL;
981 }
982
983 if (g_ack_queue) {
984 vQueueDelete(g_ack_queue);
985 g_ack_queue = NULL;
986 }
987
988 ESP_LOGI(TAG, "main task exit");
989 vTaskDelete(NULL);
990}

992static void wifi_event_handler(void *arg, esp_event_base_t event_base,
993 int32_t event_id, void *event_data)
994{
995 static bool s_ap_staconnected_flag = false;
996 static bool s_sta_connected_flag = false;
997
998 switch (event_id) {
999 case WIFI_EVENT_AP_STACONNECTED: {
1000 wifi_event_ap_stadisconnected_t *event = (wifi_event_ap_stadisconnected_t *) event_data;
1001 ESP_LOGI(TAG, "station "MACSTR" join, AID=%d", MAC2STR(event->mac), event->aid);
1002 s_ap_staconnected_flag = true;
1003 break;
1004 }
1005
1006 case WIFI_EVENT_AP_STADISCONNECTED: {
1007 wifi_event_ap_stadisconnected_t* event = (wifi_event_ap_stadisconnected_t*) event_data;
1008 ESP_LOGI(TAG, "station "MACSTR" leave, AID=%d", MAC2STR(event->mac), event->aid);
1009 s_ap_staconnected_flag = false;
1010 break;
1011 }
1012
1013 case WIFI_EVENT_STA_CONNECTED:{
1014 wifi_event_sta_connected_t *event = (wifi_event_sta_connected_t *)event_data;
1015 ESP_LOGI(TAG, "Connected to %s (BSSID: "MACSTR", Channel: %d)", event->ssid,
1016 MAC2STR(event->bssid), event->channel);
1017 s_sta_connected_flag = true;
1018 break;
1019 }
1020
1021 case WIFI_EVENT_STA_DISCONNECTED: {
1022 ESP_LOGI(TAG, "sta disconnect");
1023 s_sta_connected_flag = false;
1024 break;
1025 }
1026
1027 default:
1028 break;
1029 }
1030
1031 wifi_mode_t mode = WIFI_MODE_NULL;
1032 esp_wifi_get_mode(&mode);
1033
1034 if (mode == WIFI_MODE_APSTA || s_ap_staconnected_flag == true || s_sta_connected_flag == true) {
1035 g_set_channel_flag = false;
1036 } else {
1037 g_set_channel_flag = true;
1038 }
1039}

1041esp_err_t espnow_init(const espnow_config_t *config)
1042{
1043 ESP_LOGI(TAG, "esp-now Version: %d.%d.%d", ESP_NOW_VER_MAJOR, ESP_NOW_VER_MINOR, ESP_NOW_VER_PATCH);
1044 wifi_ap_record_t ap_info;
1045
1046 ESP_PARAM_CHECK(config);
1047
1048 if (g_espnow_config) {
1049 return ESP_OK;
1050 }
1051
1052 g_espnow_config = ESP_MALLOC(sizeof(espnow_config_t));
1053 memcpy(g_espnow_config, config, sizeof(espnow_config_t));
1054
1055 /**< Event group for espnow sent cb */
1056 g_event_group = xEventGroupCreate();
1057 ESP_ERROR_RETURN(!g_event_group, ESP_FAIL, "Create event group fail");
1058
1059 g_send_lock = xSemaphoreCreateMutex();
1060 ESP_ERROR_RETURN(!g_send_lock, ESP_FAIL, "Create send semaphore mutex fail");
1061
1062 ESP_ERROR_CHECK(esp_event_handler_register(WIFI_EVENT, ESP_EVENT_ANY_ID, wifi_event_handler, NULL));
1063
1064 /* There may be chances that another Wi-Fi application has already connected to AP before ESP-NOW
1065 * registers wifi_event_handler. Check AP info here to avoid missing WIFI_EVENT_STA_CONNECTED action.
1066 */
1067 if (esp_wifi_sta_get_ap_info(&ap_info) == ESP_OK) {
1068 ESP_LOGI(TAG, "device already connected to SSID: %s", ap_info.ssid);
1069 g_set_channel_flag = false;
1070 }
1071
1072 if (config->sec_enable) {
1073 g_espnow_sec = ESP_MALLOC(sizeof(espnow_sec_t));
1074 g_espnow_dec = ESP_MALLOC(sizeof(espnow_sec_t));
1075 espnow_sec_init(g_espnow_sec);
1076 espnow_sec_init(g_espnow_dec);
1077 }
1078
1079 uint32_t *enable = (uint32_t *)&config->receive_enable;
1080 for (int i = 0; i < ESPNOW_DATA_TYPE_MAX; ++i) {
1081 g_recv_handle[i].enable = (*enable) & BIT(i);
1082 }
