89static int parse_next_query(void *data, int size, dns_query_t *query)
90{
91 int len;
92 int lables;
93 uint8_t *ptr;
94
95 len = 0;
96 lables = 0;
97 ptr = (uint8_t *)data;
98
99 while (true) {
100 uint8_t lable_len;
101
102 if (size <= 0) {
103 return -1;
104 }
105
106 lable_len = *ptr++;
107 size--;
108
109 if (lable_len == 0) {
110 break;
111 }
112
113 if (lables > 0) {
114 if (len == DNS_MAX_HOST_NAME_LEN) {
115 return -2;
116 }
117
118 query->name[len++] = '.';
119 }
120
121 if (lable_len > size) {
122 return -1;
123 }
124
125 if (len + lable_len >= DNS_MAX_HOST_NAME_LEN) {
126 return -2;
127 }
128
129 memcpy(&query->name[len], ptr, lable_len);
130 len += lable_len;
131 ptr += lable_len;
132 size -= lable_len;
133 lables++;
134 }
135
136 if (size < 4) {
137 return -1;
138 }
139
140 query->name[len] = 0;
141 query->type = get_uint16(ptr);
142 ptr += 2;
143 query->Class = get_uint16(ptr);
144 ptr += 2;
145 return ptr - (uint8_t *)data;
146}

148static void udp_recv_proc(void *arg, struct udp_pcb *upcb, struct pbuf *p, const ip_addr_t *addr, u16_t port)
149{
150 int len;
151 dns_header_t *header;
152 static dns_query_t query;
153 struct pbuf *out;
154 ip_addr_t host_addr;
155 dns_answer_t *answer;
156
157 (void)arg;
158
159 if (p->len <= sizeof(dns_header_t)) {
160 goto error;
161 }
162
163 header = (dns_header_t *)p->payload;
164
165 if (header->flags.qr != 0) {
166 goto error;
167 }
168
169 if (ntohs(header->n_record[0]) != 1) {
170 goto error;
171 }
172
173 len = parse_next_query(header + 1, p->len - sizeof(dns_header_t), &query);
174
175 if (len < 0) {
176 goto error;
177 }
178
179 if (!query_proc(query.name, &host_addr)) {
180 goto error;
181 }
182
183 len += sizeof(dns_header_t);
184 out = pbuf_alloc(PBUF_TRANSPORT, len + 16, PBUF_POOL);
185
186 if (out == NULL) {
187 goto error;
188 }
189
190 memcpy(out->payload, p->payload, len);
191 header = (dns_header_t *)out->payload;
192 header->flags.qr = 1;
193 header->n_record[1] = htons(1);
194 answer = (struct dns_answer *)((uint8_t *)out->payload + len);
195 answer->name = htons(0xC00C);
196 answer->type = htons(1);
197 answer->Class = htons(1);
198 answer->ttl = htonl(32);
199 answer->len = htons(4);
200#if CONFIG_LWIP_IPV6
201 answer->addr = host_addr.u_addr.ip4.addr;
202#else
203 answer->addr = host_addr.addr;
204#endif
205
206 udp_sendto(upcb, out, addr, port);
207 pbuf_free(out);
208
209error:
210 pbuf_free(p);
211}

213err_t dnserv_init(const ip_addr_t *bind, uint16_t port, dns_query_proc_t qp)
214{
215 err_t err;
216 udp_init();
217 dnserv_free();
218 pcb = udp_new();
219
220 if (pcb == NULL) {
221 return ERR_MEM;
222 }
223
224 err = udp_bind(pcb, bind, port);
225
226 if (err != ERR_OK) {
227 dnserv_free();
228 return err;
229 }
230
231 udp_recv(pcb, udp_recv_proc, NULL);
232 query_proc = qp;
233 return ERR_OK;
234}

236void dnserv_free()

242 udp_remove(pcb);
243 pcb = NULL;
