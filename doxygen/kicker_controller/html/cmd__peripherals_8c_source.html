39/**
40 * @brief A function which implements gpio command.
41 */
42static int gpio_func(int argc, char **argv)
43{
44 if (arg_parse(argc, argv, (void **) &gpio_args) != ESP_OK) {
45 arg_print_errors(stderr, gpio_args.end, argv[0]);
46 return ESP_FAIL;
47 }
48
49 if (gpio_args.config->count) {
50 uint32_t gpio_num = gpio_args.config->ival[0];
51 esp_err_t ret = gpio_reset_pin(gpio_num);
52 ESP_ERROR_RETURN(ret != ESP_OK, ret, "gpio_reset_pin, gpio_num: %d", gpio_num);
53 }
54
55 if (gpio_args.set->count && gpio_args.level->count) {
56 uint32_t gpio_num = gpio_args.set->ival[0];
57
58 gpio_set_direction(gpio_num, GPIO_MODE_OUTPUT);
59 gpio_set_level(gpio_num, gpio_args.level->ival[0]);
60 ESP_LOGI(TAG, "Set gpio num: %d, level: %d", gpio_num, gpio_args.level->ival[0]);
61 }
62
63 if (gpio_args.get->count) {
64 uint32_t gpio_num = gpio_args.get->ival[0];
65
66 gpio_set_direction(gpio_num, GPIO_MODE_INPUT);
67 ESP_LOGI(TAG, "Get gpio num: %d, level: %d", gpio_num, gpio_get_level(gpio_num));
68 }
69
70 return ESP_OK;
71}

105/**
106 * @brief A function which implements uart command.
107 */
108static int uart_func(int argc, char **argv)
109{
110 if (arg_parse(argc, argv, (void **) &uart_args) != ESP_OK) {
111 arg_print_errors(stderr, uart_args.end, argv[0]);
112 return ESP_FAIL;
113 }
114
115 esp_err_t ret = ESP_OK;
116 static uart_port_t port_num = UART_NUM_0;
117
118
119 if (uart_args.start->count) {
120 uart_config_t uart_config = {
121 .baud_rate = uart_args.baud_rate->count?uart_args.baud_rate->ival[0] : 115200,
122 .data_bits = UART_DATA_8_BITS,
123 .parity = UART_PARITY_DISABLE,
124 .stop_bits = UART_STOP_BITS_1,
125 .flow_ctrl = UART_HW_FLOWCTRL_DISABLE
126 };
127
128 int tx_io_num = UART_PIN_NO_CHANGE;
129 int rx_io_num = UART_PIN_NO_CHANGE;
130
131 if(uart_args.port_num->count) {
132 port_num = uart_args.port_num->ival[0];
133 }
134
135 if (uart_args.tx_io->count) {
136 tx_io_num = uart_args.tx_io->ival[0];
137 }
138
139 if (uart_args.rx_io->count) {
140 rx_io_num = uart_args.rx_io->ival[0];
141 }
142
143 ESP_ERROR_CHECK(uart_param_config(port_num, &uart_config));
144 ESP_ERROR_CHECK(uart_set_pin(port_num, tx_io_num, rx_io_num, UART_PIN_NO_CHANGE, UART_PIN_NO_CHANGE));
145 ESP_ERROR_CHECK(uart_driver_install(port_num, 2 * ESPNOW_DATA_LEN, 2 * ESPNOW_DATA_LEN, 0, NULL, 0));
146 } else if (uart_args.read->count) {
147 uint8_t *data = ESP_CALLOC(1, ESPNOW_DATA_LEN);
148 ret = uart_read_bytes(port_num, data, ESPNOW_DATA_LEN, pdMS_TO_TICKS(uart_args.read->ival[0]));
149 ESP_LOGI(TAG, "uart_read_bytes, size: %d, data: %s", ret, data);
150 ESP_FREE(data);
151 ESP_ERROR_RETURN(ret <= 0, ESP_FAIL, "uart_read_bytes, timeout: %d", uart_args.read->ival[0]);
152 } else if (uart_args.write->count) {
153 ret = uart_write_bytes(port_num, uart_args.write->sval[0], strlen(uart_args.write->sval[0]));
154 ESP_ERROR_RETURN(ret <= 0, ESP_FAIL, "uart_write_bytes");
155 ESP_LOGI(TAG, "uart_read_bytes, size: %d, data: %s", strlen(uart_args.write->sval[0]), uart_args.write->sval[0]);
156 }
157
158 return ESP_OK;
159}
