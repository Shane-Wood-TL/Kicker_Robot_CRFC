82static esp_err_t create_packet_file(void)
83{
84 uint32_t file_no = 0;
85 char filename[PCAP_FILE_NAME_MAX_LEN + 16];
86
87 do {
88 snprintf(filename, sizeof(filename), "%s_%d.pcap", packet_filepath, file_no);
89 ++file_no;
90 } while (0 == access(filename, F_OK));
91
92 /* Create file to write, binary format */
93 pcap_config.fp = fopen(filename, "wb");
94
95 ESP_ERROR_RETURN(!pcap_config.fp, ESP_FAIL, "Create file %s failed", filename);
96
97 ESP_LOGI(TAG, "Store packets to file: %s", filename);
98
99 return ESP_OK;
100}

102static uint32_t hash_func(const char *str, uint32_t max_num)
103{
104 uint32_t ret = 0;
105 char *p = (char *)str;
106
107 while (*p) {
108 ret += *p;
109 ++p;
110 }
111
112 return ret % max_num;
113}

115static void create_wifi_filter_hashtable()
116{
117 const char *wifi_filter_keys[SNIFFER_WLAN_FILTER_MAX] = {"mgmt", "data", "ctrl", "misc", "mpdu", "ampdu"};
118 uint32_t wifi_filter_values[SNIFFER_WLAN_FILTER_MAX] = { WIFI_PROMIS_FILTER_MASK_MGMT, WIFI_PROMIS_FILTER_MASK_DATA,
119 WIFI_PROMIS_FILTER_MASK_CTRL, WIFI_PROMIS_FILTER_MASK_MISC,
120 WIFI_PROMIS_FILTER_MASK_DATA_MPDU, WIFI_PROMIS_FILTER_MASK_DATA_AMPDU
121 };
122
123 for (int i = 0; i < SNIFFER_WLAN_FILTER_MAX; i++) {
124 uint32_t idx = hash_func(wifi_filter_keys[i], SNIFFER_WLAN_FILTER_MAX);
125
126 while (wifi_filter_hash_table[idx].filter_name) {
127 if (++idx >= SNIFFER_WLAN_FILTER_MAX) {
128 idx = 0;
129 }
130 }
131
132 wifi_filter_hash_table[idx].filter_name = wifi_filter_keys[i];
133 wifi_filter_hash_table[idx].filter_val = wifi_filter_values[i];
134 }
135}

137static uint32_t search_wifi_filter_hashtable(const char *key)
138{
139 uint32_t len = strlen(key);
140 uint32_t start_idx = hash_func(key, SNIFFER_WLAN_FILTER_MAX);
141 uint32_t idx = start_idx;
142
143 while (strncmp(wifi_filter_hash_table[idx].filter_name, key, len)) {
144 if (++idx >= SNIFFER_WLAN_FILTER_MAX) {
145 idx = 0;
146 }
147
148 ESP_ERROR_RETURN(idx == start_idx, 0, "wrong key");
149 }
150
151 return wifi_filter_hash_table[idx].filter_val;
152}

154static void wifi_sniffer_cb(void *recv_buf, wifi_promiscuous_pkt_type_t type)
155{
156 sniffer_packet_into_t packet_info;
157 wifi_promiscuous_pkt_t *sniffer = (wifi_promiscuous_pkt_t *)recv_buf;
158
159 /* prepare packet_info */
160 packet_info.seconds = sniffer->rx_ctrl.timestamp / 1000000U;
161 packet_info.microseconds = sniffer->rx_ctrl.timestamp % 1000000U;
162 packet_info.length = sniffer->rx_ctrl.sig_len;
163 wifi_promiscuous_pkt_t *backup = malloc(sniffer->rx_ctrl.sig_len);
164
165 if (backup) {
166 memcpy(backup, sniffer->payload, sniffer->rx_ctrl.sig_len);
167 packet_info.payload = backup;
168
169 if (sniffer_work_queue) {
170 /* send packet_info */
171 if (xQueueSend(sniffer_work_queue, &packet_info, 100 / portTICK_PERIOD_MS) != pdTRUE) {
172 ESP_LOGE(TAG, "sniffer work queue full");
173 }
174 }
175 } else {
176 ESP_LOGE(TAG, "No enough memory for promiscuous packet");
177 }
178}

180static void sniffer_task(void *parameters)
181{
182 static uint32_t count = 0;
183 sniffer_packet_into_t packet_info;
184 BaseType_t ret = 0;
185
186 while (sniffer_running) {
187 /* receive paclet info from queue */
188 ret = xQueueReceive(sniffer_work_queue, &packet_info, 100 / portTICK_PERIOD_MS);
189 ESP_ERROR_CONTINUE(ret != pdTRUE, "xQueueReceive timeout");
190
191 if (pcap_capture_packet(packet_info.payload, packet_info.length,
192 packet_info.seconds, packet_info.microseconds) == ESP_OK) {
193 /* truncate, create another file */
194 if (++count >= CONFIG_PCAP_FILE_MAX_PACKETS) {
195 pcap_close();
196
197 if (create_packet_file() != ESP_OK || pcap_new(&pcap_config) != ESP_OK) {
198 sniffer_running = false;
199 } else {
200 count = 0;
201 }
202 }
203 }
204
205 free(packet_info.payload);
206 }
207
208 /* notify that sniffer task is over */
209 xSemaphoreGive(sem_task_over);
210 vTaskDelete(NULL);
211}

213static esp_err_t sniffer_start(sniffer_config_t *sniffer)
214{
215 wifi_promiscuous_filter_t wifi_filter;
216 /* set sniffer running status before it starts to run */
217 sniffer_running = true;
218 sniffer_work_queue = xQueueCreate(CONFIG_SNIFFER_WORK_QUEUE_LENGTH, sizeof(sniffer_packet_into_t));
219 sem_task_over = xSemaphoreCreateBinary();
220 /* sniffer task going to run*/
221 xTaskCreate(sniffer_task, "sniffer", CONFIG_SNIFFER_TASK_STACK_SIZE, NULL, CONFIG_SNIFFER_TASK_PRIORITY, NULL);
222
223 switch (sniffer->interf) {
224 case SNIFFER_INTF_WLAN:
225 /* Set Promicuous Mode */
226 wifi_filter.filter_mask = sniffer->filter;
227 esp_wifi_set_promiscuous_filter(&wifi_filter);
228 esp_wifi_set_promiscuous_rx_cb(wifi_sniffer_cb);
229 ESP_LOGI(TAG, "Start WiFi Promicuous Mode");
230 ESP_ERROR_CHECK(esp_wifi_set_promiscuous(true));
231 /* Specify the channel */
232 esp_wifi_set_channel(sniffer->channel, WIFI_SECOND_CHAN_NONE);
233 /* Create a new pcap object */
234 pcap_config.link_type = PCAP_LINK_TYPE_802_11;
235 pcap_new(&pcap_config);
236 break;
237
238 default:
239 break;
240 }
241
242 return ESP_OK;
243}

245static esp_err_t snifer_stop(sniffer_config_t *sniffer)
246{
247 /* Do interface specific work here */
248 switch (sniffer->interf) {
249 case SNIFFER_INTF_WLAN:
250 /* Disable wifi promiscuous mode */
251 esp_wifi_set_promiscuous(false);
252 break;
253
254 default:
255 break;
256 }
257
258 /* stop sniffer local task */
259 sniffer_running = false;
260 /* wait for task over */
261 xSemaphoreTake(sem_task_over, portMAX_DELAY);
262 vSemaphoreDelete(sem_task_over);
263 sem_task_over = NULL;
264 /* make sure to free all resources in the left items */
265 UBaseType_t left_items = uxQueueMessagesWaiting(sniffer_work_queue);
266 sniffer_packet_into_t packet_info;
267
268 while (left_items--) {
269 xQueueReceive(sniffer_work_queue, &packet_info, 100 / portTICK_PERIOD_MS);
270 free(packet_info.payload);
271 }
272
273 /* delete queue */
274 vQueueDelete(sniffer_work_queue);
275 sniffer_work_queue = NULL;
276 /* Close the pcap file */
277 pcap_close();
278
279 ESP_LOGI(TAG, "Sniffer Stopped");
280 return ESP_OK;
281}

291static int wifi_sniffer_func(int argc, char **argv)
292{
293 sniffer_config_t sniffer;
294
295 int nerrors = arg_parse(argc, argv, (void **)&wifi_sniffer_args);
296
297 if (nerrors != 0) {
298 arg_print_errors(stderr, wifi_sniffer_args.end, argv[0]);
299 return 0;
300 }
301
302 memset(&sniffer, 0, sizeof(sniffer));
303
304 /* wlan interface */
305 sniffer.interf = SNIFFER_INTF_WLAN;
306
307 /* Check whether or not to stop sniffer: "--stop" option */
308 if (wifi_sniffer_args.stop->count) {
309 if (sniffer_running) {
310 /* stop sniffer */
311 snifer_stop(&sniffer);
312 } else {
313 ESP_LOGW(TAG, "sniffer already stop");
314 }
315
316 return 0;
317 }
318
319 /* Check channel: "-c" option */
320 sniffer.channel = 0;
321
322 if (wifi_sniffer_args.channel->count) {
323 sniffer.channel = wifi_sniffer_args.channel->ival[0];
324 } else {
325 sniffer.channel = SNIFFER_DEFAULT_CHANNEL;
326 }
327
328 /* set pcap file name: "-f" option */
329 if (wifi_sniffer_args.file->count) {
330 snprintf(packet_filepath, PCAP_FILE_NAME_MAX_LEN, "%s/%s",
331 CONFIG_SNIFFER_MOUNT_POINT, wifi_sniffer_args.file->sval[0]);
332 } else {
333 snprintf(packet_filepath, PCAP_FILE_NAME_MAX_LEN, "%s/%s",
334 CONFIG_SNIFFER_MOUNT_POINT, SNIFFER_DEFAULT_FILE_NAME);
335 }
336
337 /* Determin file name */
338 if (create_packet_file() != ESP_OK) {
339 return 1;
340 }
341
342 /* Check filter setting: "-F" option */
343 switch (sniffer.interf) {
344 case SNIFFER_INTF_WLAN:
345 if (wifi_sniffer_args.filter->count) {
346 for (int i = 0; i < wifi_sniffer_args.filter->count; i++) {
347 sniffer.filter += search_wifi_filter_hashtable(wifi_sniffer_args.filter->sval[i]);
348 }
349
350 /* When filter conditions are all wrong */
351 if (sniffer.filter == 0) {
352 sniffer.filter = WIFI_PROMIS_FILTER_MASK_ALL;
353 }
354 } else {
355 sniffer.filter = WIFI_PROMIS_FILTER_MASK_ALL;
356 }
357
358 break;
359
360 default:
361 break;
362 }
363
364 if (!sniffer_running) {
365 /* start sniffer */
366 sniffer_start(&sniffer);
367 } else {
368 ESP_LOGW(TAG, "Please stop sniffer firstly, then start with parameters");
369 }
370
371 return 0;
372}
