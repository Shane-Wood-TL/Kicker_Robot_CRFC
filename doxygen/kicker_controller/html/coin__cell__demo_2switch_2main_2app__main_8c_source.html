43static void set_light_sleep(uint32_t ms)
44{
45 esp_wifi_force_wakeup_release();
46 esp_sleep_enable_timer_wakeup(ms * 1000);
47 esp_light_sleep_start();
48 esp_wifi_force_wakeup_acquire();
49 esp_sleep_wakeup_cause_t cause = esp_sleep_get_wakeup_cause();
50 const char *cause_str;
51 switch (cause) {
52 case ESP_SLEEP_WAKEUP_GPIO:
53 cause_str = "GPIO";
54 break;
55 case ESP_SLEEP_WAKEUP_UART:
56 cause_str = "UART";
57 break;
58 case ESP_SLEEP_WAKEUP_TIMER:
59 cause_str = "timer";
60 break;
61 default:
62 cause_str = "unknown";
63 break;
64 }
65 ESP_LOGI(TAG, "Woke up from: %s", cause_str);
66}

83static void control_task(void *pvParameter)
84{
85 board_led_on(true);
86 board_power_lock(true);
87 set_light_sleep(SEND_GAP_TIME);
88
89 for (;;) {
90 if (task_state == ESPNOW_TASK_STATE_SEND_RECORD) {
91 uint8_t status = 0;
92 /* status = 0: OFF, 1: ON, 2: TOGGLE */
93#if CONFIG_EXAMPLE_SWITCH_STATUS_PERSISTED
94 espnow_storage_get(BULB_STATUS_KEY, &status, sizeof(status));
95 status ^= 1;
96 espnow_storage_set(BULB_STATUS_KEY, &status, sizeof(status));
97#else
98 status = 2;
99#endif
100 ESP_LOGI(TAG, "switch send press");
101 espnow_ctrl_initiator_send(ESPNOW_ATTRIBUTE_KEY_1, ESPNOW_ATTRIBUTE_POWER, status);
102
103 task_state = ESPNOW_TASK_STATE_DONE;
104 }
105
106 if (task_state == ESPNOW_TASK_STATE_DONE) {
107 set_light_sleep(SEND_GAP_TIME); //Prevent false wake-ups
108 board_power_lock(false);
109 board_led_on(false);
110 ESP_LOGW(TAG, "task_state = ESPNOW_TASK_STATE_DONE");
111 set_light_sleep(LONG_PRESS_SLEEP_TIME);
112 board_power_lock(true);
113 board_led_on(true);
114 task_state = ESPNOW_TASK_STATE_SEND_BIND;
115 }
116
117 if (task_state == ESPNOW_TASK_STATE_SEND_BIND) {
118 ESP_LOGI(TAG, "switch bind press");
119 espnow_ctrl_initiator_bind(ESPNOW_ATTRIBUTE_KEY_1, true);
120 task_state = ESPNOW_TASK_STATE_BIND_DONE;
121 }
122
123 if (task_state == ESPNOW_TASK_STATE_BIND_DONE) {
124 set_light_sleep(SEND_GAP_TIME); //Prevent false wake-ups
125 board_power_lock(false);
126 board_led_on(false);
127 ESP_LOGW(TAG, "task_state = ESPNOW_TASK_STATE_DONE");
128 set_light_sleep(LONG_PRESS_SLEEP_TIME);
129 board_power_lock(true);
130 board_led_on(true);
131 task_state = ESPNOW_TASK_STATE_SEND_UNBIND;
132 }
133
134 if (task_state == ESPNOW_TASK_STATE_SEND_UNBIND) {
135 ESP_LOGI(TAG, "switch unbind press");
136 espnow_ctrl_initiator_bind(ESPNOW_ATTRIBUTE_KEY_1, false);
137 task_state = ESPNOW_TASK_STATE_UNBIND_DONE;
138 }
139
140 if (task_state == ESPNOW_TASK_STATE_UNBIND_DONE) {
141 set_light_sleep(SEND_GAP_TIME); //Prevent false wake-ups
142 board_power_lock(false);
143 board_led_on(false);
144 set_light_sleep(LONG_PRESS_SLEEP_TIME);
145 }
146 }
147}

197static void control_task(void *pvParameter)
198{
199 bool is_running = true;
200 button_event_t evt_data;
201 uint8_t status = 0;
202 g_button_queue = xQueueCreate(5, sizeof(button_event_t));
203 if (!g_button_queue) {
204 ESP_LOGE(TAG, "Error creating queue.");
205 return;
206 }
207
208#if CONFIG_PM_ENABLE
209 gpio_wakeup_enable(CONTROL_KEY_GPIO, GPIO_INTR_LOW_LEVEL);
210 esp_sleep_enable_gpio_wakeup();
211#endif
212
213 while(is_running) {
214 if (xQueueReceive(g_button_queue, &evt_data, portMAX_DELAY) != pdTRUE) {
215 ESP_LOGI(TAG, "Nothing received");
216 continue;
217 }
218#if CONFIG_PM_ENABLE
219 power_save_set(false);
220#endif
221#if CONFIG_PM_ENABLE || CONFIG_ESPNOW_LIGHT_SLEEP
222 esp_wifi_force_wakeup_acquire();
223#endif
224
225 ESP_LOGI(TAG, "Button event: %s", button_event_table[(button_event_t)evt_data]);
226 if (evt_data == BUTTON_SINGLE_CLICK) {
227 ESP_LOGI(TAG, "switch send press");
228#if CONFIG_EXAMPLE_SWITCH_STATUS_PERSISTED
229 espnow_storage_get(BULB_STATUS_KEY, &status, sizeof(status));
230 status ^= 1;
231 espnow_storage_set(BULB_STATUS_KEY, &status, sizeof(status));
232#else
233 status = 2;
234#endif
235 /* status = 0: OFF, 1: ON, 2: TOGGLE */
236 ESP_LOGI(TAG, "key status: %d", status);
237 espnow_ctrl_initiator_send(ESPNOW_ATTRIBUTE_KEY_1, ESPNOW_ATTRIBUTE_POWER, status);
238 } else if (evt_data == BUTTON_DOUBLE_CLICK) {
239 ESP_LOGI(TAG, "switch bind press");
240 espnow_ctrl_initiator_bind(ESPNOW_ATTRIBUTE_KEY_1, true);
241 } else if (evt_data == BUTTON_LONG_PRESS_START) {
242 ESP_LOGI(TAG, "switch unbind press");
243 espnow_ctrl_initiator_bind(ESPNOW_ATTRIBUTE_KEY_1, false);
244 } else {
245 ESP_LOGI(TAG, "event not handled");
246 }
247#if CONFIG_PM_ENABLE || CONFIG_ESPNOW_LIGHT_SLEEP
248 esp_wifi_force_wakeup_release();
249#endif
250#if CONFIG_PM_ENABLE
251 power_save_set(true);
252#endif
253 }
254 if (g_button_queue) {
255 vQueueDelete(g_button_queue);
256 g_button_queue = NULL;
257 }
258 vTaskDelete(NULL);
259}
