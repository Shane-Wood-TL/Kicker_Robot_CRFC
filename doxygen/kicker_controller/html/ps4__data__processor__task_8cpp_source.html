12void ps4_data_processor(void *pv) {
13 for (;;) {
14 if (xSemaphoreTake(new_controller_data, portMAX_DELAY)) {
15 display_controller_data current_display_data = {0, 0};
16 uint8_t temp_buttons = 0;
17
18 uint8_t dpad = (controller_byte_5 & lower_four_bits);
19 if (dpad == dpad_left_byte) {
20 temp_buttons |= (1 << processed_dpad_left_bit); // left
21 } else if (dpad == dpad_down_byte) {
22 temp_buttons |= (1 << processed_dpad_down_bit); // down
23 } else if (dpad == dpad_right_byte) {
24 temp_buttons |= (1 << processed_dpad_right_bit); // right
25 } else if (dpad == dpad_up_byte) {
26 temp_buttons |= (1 << processed_dpad_up_bit); // up
27 }
28
29 if (controller_byte_5 & triangle_byte) {
30 temp_buttons |= (1 << processed_triangle_bit); // triangle
31 } else if (controller_byte_5 & square_byte) {
32 temp_buttons |= (1 << processed_square_bit); // square
33 } else if (controller_byte_5 & circle_byte) {
34 temp_buttons |= (1 << processed_circle_bit); // circle
35 } else if (controller_byte_5 & x_byte) {
36 temp_buttons |= (1 << processed_x_bit); // x
37 }
38
39 current_display_data.buttons = temp_buttons;
40
41 if (controller_byte_6 & options_byte) {
42 current_display_data.options_triggers |= (1 << processed_options_bit); // options
43 } else if (controller_byte_6 & r2_byte) {
44 current_display_data.options_triggers |= (1 << processed_r2_bit); // r2
45 } else if (controller_byte_6 & l2_byte) {
46 current_display_data.options_triggers |= (1 << processed_l2_bit); // l2
47 }
48
49
50 if (controller_byte_6 & l3_byte) { //l3 (boost)
51 boost = true;
52 } else {
53 boost = false;
54 }
55 xQueueSend(other_controller_data_queue, &current_display_data, portMAX_DELAY);
56 vTaskDelay(pdMS_TO_TICKS(ps4_data_processor_wait_time));
57 }
58 }
59}
