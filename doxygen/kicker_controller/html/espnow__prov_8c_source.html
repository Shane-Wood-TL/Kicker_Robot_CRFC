87static esp_err_t espnow_prov_recv(uint8_t *src_addr, void *data,
88 size_t size, wifi_pkt_rx_ctrl_t *rx_ctrl)
89{
90 ESP_PARAM_CHECK(src_addr);
91 ESP_PARAM_CHECK(data);
92 ESP_PARAM_CHECK(size);
93 ESP_PARAM_CHECK(rx_ctrl);
94
95 esp_err_t ret = ESP_OK;
96 uint8_t data_type = ((uint8_t *)data)[0];
97 espnow_prov_data_t *recv_data = (espnow_prov_data_t *)data;
98
99 switch (data_type) {
100 case ESPNOW_PROV_TYPE_BEACON:
101 ESP_LOGD(TAG, "ESPNOW_PROV_TYPE_BEACON");
102 if (g_prov_init && g_prov_init->beacon_en) {
103 g_prov_init->fix_ch = true;
104 if (g_prov_init->scan_info.responder_info && g_prov_init->scan_info.rx_ctrl && g_prov_init->scan_info.addr) {
105 memcpy(g_prov_init->scan_info.responder_info, &recv_data->responder_info, sizeof(espnow_prov_responder_t));
106 memcpy(g_prov_init->scan_info.rx_ctrl, rx_ctrl, sizeof(wifi_pkt_rx_ctrl_t));
107 memcpy(g_prov_init->scan_info.addr, src_addr, 6);
108 }
109 }
110 break;
111
112 case ESPNOW_PROV_TYPE_DEVICE:
113 ESP_LOGD(TAG, "ESPNOW_PROV_TYPE_DEVICE");
114 if (g_prov_resp && g_prov_resp->device_en) {
115 ret = ESP_OK;
116 if (g_prov_resp->device_cb) {
117 ret = g_prov_resp->device_cb(src_addr, &recv_data->initiator_info, size - 1, rx_ctrl);
118 }
119 if (ret == ESP_OK) {
120 espnow_addr_t initiator_addr = {0};
121 memcpy(initiator_addr, src_addr, 6);
122 ret = espnow_prov_responder_send((const espnow_addr_t *)&initiator_addr, 1, g_prov_resp->wifi_config);
123 }
124 }
125 break;
126
127 case ESPNOW_PROV_TYPE_WIFI:
128 ESP_LOGD(TAG, "ESPNOW_PROV_TYPE_WIFI");
129 if (g_prov_init && g_prov_init->wifi_en) {
130 ret = ESP_OK;
131 if (g_prov_init->wifi_cb) {
132 ret = g_prov_init->wifi_cb(src_addr, &recv_data->wifi_config, size - 1, rx_ctrl);
133 }
134 if (ret == ESP_OK) {
135 g_prov_init->config = true;
136 }
137 }
138 break;
139
140 default:
141 break;
142 }
143
144 return ret;
145}

147esp_err_t espnow_prov_initiator_scan(espnow_addr_t responder_addr, espnow_prov_responder_t *responder_info,
148 wifi_pkt_rx_ctrl_t *rx_ctrl, TickType_t wait_ticks)
149{
150 ESP_PARAM_CHECK(responder_addr);
151 ESP_PARAM_CHECK(responder_info);
152 ESP_PARAM_CHECK(rx_ctrl);
153
154 esp_err_t ret = ESP_FAIL;
155
156 wifi_country_t country = {0};
157 uint32_t start_ticks = xTaskGetTickCount();
158
159 if (!g_prov_init) {
160 g_prov_init = ESP_CALLOC(1, sizeof(espnow_prov_init_t));
161 }
162
163 g_prov_init->beacon_en = true;
164 g_prov_init->fix_ch = false;
165 g_prov_init->scan_info.responder_info = responder_info;
166 g_prov_init->scan_info.rx_ctrl = rx_ctrl;
167 g_prov_init->scan_info.addr = responder_addr;
168
169 espnow_set_config_for_data_type(ESPNOW_DATA_TYPE_PROV, 1, espnow_prov_recv);
170 esp_wifi_get_country(&country);
171
172 while (g_prov_init->fix_ch == false && (wait_ticks == portMAX_DELAY || xTaskGetTickCount() - start_ticks < wait_ticks)) {
173 for (int i = 0; i < country.nchan && g_prov_init->fix_ch == false; ++i) {
174 esp_wifi_set_channel(country.schan + i, WIFI_SECOND_CHAN_NONE);
175 ESP_LOGD(TAG, "espnow_send, channel: %d", country.schan + i);
176 vTaskDelay(pdMS_TO_TICKS(ESPNOW_PROV_BEACON_INTERVAL + 10));
177 }
178 }
179
180 if (g_prov_init->fix_ch == true) {
181 ret = ESP_OK;
182 }
183
184 espnow_set_config_for_data_type(ESPNOW_DATA_TYPE_PROV, 0, NULL);
185 ESP_FREE(g_prov_init);
186
187 return ret;
188}

190esp_err_t espnow_prov_initiator_send(const espnow_addr_t responder_addr, const espnow_prov_initiator_t *initiator_info, espnow_prov_cb_t cb, TickType_t wait_ticks)
191{
192 ESP_PARAM_CHECK(responder_addr);
193 ESP_PARAM_CHECK(initiator_info);
194
195 esp_err_t ret = ESP_ERR_WIFI_TIMEOUT;
196 espnow_prov_data_t *prov_data = ESP_MALLOC(sizeof(espnow_prov_data_t));
197 prov_data->type = ESPNOW_PROV_TYPE_DEVICE;
198 memcpy(&prov_data->initiator_info, initiator_info, sizeof(espnow_prov_initiator_t));
199 espnow_frame_head_t frame_head = {
200 .filter_adjacent_channel = true,
201 .security = CONFIG_ESPNOW_PROV_SECURITY,
202 };
203 uint32_t start_ticks = xTaskGetTickCount();
204
205 espnow_add_peer(responder_addr, NULL);
206
207 ret = espnow_send(ESPNOW_DATA_TYPE_PROV, responder_addr, prov_data,
208 sizeof(espnow_prov_data_t), &frame_head, portMAX_DELAY);
209
210 espnow_del_peer(responder_addr);
211 ESP_FREE(prov_data);
212 ESP_ERROR_RETURN(ret != ESP_OK, ret, "espnow_write");
213
214 if (!g_prov_init) {
215 g_prov_init = ESP_CALLOC(1, sizeof(espnow_prov_init_t));
216 }
217 g_prov_init->wifi_cb = cb;
218 g_prov_init->wifi_en = true;
219 g_prov_init->config = false;
220 espnow_set_config_for_data_type(ESPNOW_DATA_TYPE_PROV, 1, espnow_prov_recv);
221
222 while (g_prov_init->config == false && (wait_ticks == portMAX_DELAY || xTaskGetTickCount() - start_ticks < wait_ticks)) {
223 vTaskDelay(pdMS_TO_TICKS(100));
224 }
225
226 espnow_set_config_for_data_type(ESPNOW_DATA_TYPE_PROV, 0, NULL);
227
228 if (g_prov_init->config == false) {
229 ret = ESP_FAIL;
230 } else {
231 ret = ESP_OK;
232 }
233
234 ESP_FREE(g_prov_init);
235
236 return ret;
237}

242static void responder_beacon_timercb(TimerHandle_t timer)
243{
244 if (g_beacon_stop_tick < xTaskGetTickCount()) {
245 vTimerSetReloadMode(timer, pdFALSE);
246 xTimerStop(timer, 0);
247 xTimerDelete(timer, 0);
248 ESP_FREE(g_beacon_prov_data);
249 ESP_LOGI(TAG, "Responder beacon end");
250 g_prov_resp->device_cb = NULL;
251 g_prov_resp->device_en = false;
252 ESP_FREE(g_prov_resp->wifi_config);
253 ESP_FREE(g_prov_resp);
254 espnow_set_config_for_data_type(ESPNOW_DATA_TYPE_PROV, 0, NULL);
255 return ;
256 }
257
258 espnow_frame_head_t frame_head = {
259 .retransmit_count = 10,
260 .broadcast = true,
261 .magic = esp_random(),
262 .filter_adjacent_channel = true,
263 .security = CONFIG_ESPNOW_PROV_SECURITY,
264 };
265
266 espnow_send(ESPNOW_DATA_TYPE_PROV, ESPNOW_ADDR_BROADCAST, g_beacon_prov_data,
267 sizeof(espnow_prov_responder_t) + 1, &frame_head, portMAX_DELAY);

270esp_err_t espnow_prov_responder_start(const espnow_prov_responder_t *responder_info, TickType_t wait_ticks,
271 const espnow_prov_wifi_t *wifi_config, espnow_prov_cb_t cb)
272{
273 ESP_PARAM_CHECK(responder_info);
274 ESP_PARAM_CHECK(wifi_config);
275
276 if (!g_beacon_prov_data) {
277 g_beacon_prov_data = ESP_MALLOC(sizeof(espnow_prov_responder_t) + 1);
278 TimerHandle_t timer = xTimerCreate("responder_beacon",
279 pdMS_TO_TICKS(ESPNOW_PROV_BEACON_INTERVAL),
280 true, NULL, responder_beacon_timercb);
281 xTimerStart(timer, 0);
282 }
283
284 g_beacon_stop_tick = xTaskGetTickCount() + wait_ticks;
285 g_beacon_prov_data->type = ESPNOW_PROV_TYPE_BEACON;
286 memcpy(&g_beacon_prov_data->responder_info, responder_info, sizeof(espnow_prov_responder_t));
287 ESP_LOGI(TAG, "Responder beacon start, timer: %ds", pdTICKS_TO_MS(wait_ticks) / 1000);
288
289 if (!g_prov_resp) {
290 g_prov_resp = ESP_CALLOC(1, sizeof(espnow_prov_resp_t));
291 }
292
293 g_prov_resp->device_cb = cb;
294 g_prov_resp->device_en = true;
295
296 if (!g_prov_resp->wifi_config) {
297 g_prov_resp->wifi_config = ESP_MALLOC(sizeof(espnow_prov_wifi_t));
298 }
299 memcpy(g_prov_resp->wifi_config, wifi_config, sizeof(espnow_prov_wifi_t));
300
301 espnow_set_config_for_data_type(ESPNOW_DATA_TYPE_PROV, 1, espnow_prov_recv);
302
303 return ESP_OK;
304}

306/**
307 * @brief Responder sends WiFi configuration
308 *
309 * @param[in] initiator_addr_list mac address list of initiators
310 * @param[in] initiator_addr_num initiator address number
311 * @param[in] wifi_config WiFi configuration information
312 *
313 * @return
314 * - ESP_OK
315 * - ESP_ERR_INVALID_ARG
316 */
317static esp_err_t espnow_prov_responder_send(const espnow_addr_t *initiator_addr_list, size_t initiator_addr_num,
318 const espnow_prov_wifi_t *wifi_config)
319{
320 ESP_PARAM_CHECK(initiator_addr_list);
321 ESP_PARAM_CHECK(wifi_config);
322
323 esp_err_t ret = ESP_OK;
324 uint8_t dest_addr[6] = {0};
325 espnow_frame_head_t frame_head = {
326 .retransmit_count = 10,
327 .broadcast = true,
328 .security = CONFIG_ESPNOW_PROV_SECURITY,
329 };
330
331 ESP_LOGD(TAG, MACSTR ", num: %d", MAC2STR(initiator_addr_list[0]), initiator_addr_num);
332
333 if (initiator_addr_num > 1) {
334 espnow_set_group(initiator_addr_list, initiator_addr_num, ESPNOW_ADDR_GROUP_PROV, NULL, true, portMAX_DELAY);
335 frame_head.group = true;
336 memcpy(dest_addr, ESPNOW_ADDR_GROUP_PROV, 6);
337 } else {
338 memcpy(dest_addr, initiator_addr_list[0], 6);
339 espnow_add_peer(dest_addr, NULL);
340 }
341
342 size_t size = 1 + sizeof(espnow_prov_wifi_t) + wifi_config->custom_size;
343 espnow_prov_data_t *prov_data = ESP_MALLOC(size);
344 memcpy(&prov_data->wifi_config, wifi_config, size - 1);
345 prov_data->type = ESPNOW_PROV_TYPE_WIFI;
346
347 ret = espnow_send(ESPNOW_DATA_TYPE_PROV, dest_addr, prov_data, size, &frame_head, portMAX_DELAY);
348 if (ret != ESP_OK) {
349 ESP_FREE(prov_data);
350 ESP_LOGW(TAG, "[%s, %d] <%s> " "espnow_send", __func__, __LINE__, esp_err_to_name(ret));
351 return ret;
352 }
353
354 ESP_FREE(prov_data);
355
356 if (initiator_addr_num > 1) {
357 espnow_set_group(initiator_addr_list, initiator_addr_num, ESPNOW_ADDR_GROUP_PROV, NULL, false, portMAX_DELAY);
358 } else {
359 espnow_del_peer(dest_addr);
360 }
361
362 return ESP_OK;
363}
