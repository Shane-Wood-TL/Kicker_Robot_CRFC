33static uint32_t g_mem_count = 0;
34static espnow_mem_info_t *g_mem_info = NULL;

37void espnow_mem_add_record(void *ptr, int size, const char *tag, int line)
38{
39 if (!ptr || !size || !tag) {
40 return;
41 }
42
43 ESP_LOGV(TAG, "<%s : %d> Alloc ptr: %p, size: %d, heap free: %" PRIu32 "", tag, line,
44 ptr, (int)size, esp_get_free_heap_size());
45
46 if (!g_mem_info) {
47 g_mem_info = calloc(MEM_DBG_INFO_MAX, sizeof(espnow_mem_info_t));
48 }
49
50 if (g_mem_count >= MEM_DBG_INFO_MAX) {
51 ESP_LOGE(TAG, "The buffer space of the memory record is full");
52 espnow_mem_print_record();
53 return ;
54 }
55
56 if (!g_mem_info_lock) {
57 g_mem_info_lock = xSemaphoreCreateMutex();
58 }
59
60 xSemaphoreTake(g_mem_info_lock, portMAX_DELAY);
61
62 for (int i = 0; i < MEM_DBG_INFO_MAX; i++) {
63 if (!g_mem_info[i].size) {
64 g_mem_info[i].ptr = ptr;
65 g_mem_info[i].tag = tag;
66 g_mem_info[i].line = line;
67 g_mem_info[i].timestamp = esp_log_timestamp();
68 g_mem_info[i].size = size;
69 g_mem_count++;
70 break;
71 }
72 }
73
74 xSemaphoreGive(g_mem_info_lock);
75}

77void espnow_mem_remove_record(void *ptr, const char *tag, int line)
78{
79 if (!ptr) {
80 return;
81 }
82
83 ESP_LOGV(TAG, "<%s : %d> Free ptr: %p, heap free: %" PRIu32 "", tag, line, ptr, esp_get_free_heap_size());
84
85 if (!g_mem_info) {
86 g_mem_info = calloc(MEM_DBG_INFO_MAX, sizeof(espnow_mem_info_t));
87 }
88
89 if (!g_mem_info_lock) {
90 g_mem_info_lock = xSemaphoreCreateMutex();
91 }
92
93 xSemaphoreTake(g_mem_info_lock, portMAX_DELAY);
94
95 for (int i = 0; i < MEM_DBG_INFO_MAX; i++) {
96 if (g_mem_info[i].size && g_mem_info[i].ptr == ptr) {
97 g_mem_info[i].size = 0;
98 g_mem_count--;
99 break;
100 }
101 }
102
103 xSemaphoreGive(g_mem_info_lock);
104}

106void espnow_mem_print_record(void)
107{
108 size_t total_size = 0;
109
110 if (!ESP_MEM_DEBUG) {
111 ESP_LOGW(TAG, "Please enable memory record");
112 }
113
114 if (!g_mem_count || !g_mem_info) {
115 ESP_LOGW(TAG, "Memory record is empty");
116 return ;
117 }
118
119 for (int i = 0; i < MEM_DBG_INFO_MAX; i++) {
120 if (g_mem_info[i].size) {
121 ESP_LOGI(TAG, "(%d) <%s: %d> ptr: %p, size: %d", g_mem_info[i].timestamp, g_mem_info[i].tag, g_mem_info[i].line,
122 g_mem_info[i].ptr, g_mem_info[i].size);
123 total_size += g_mem_info[i].size;
124 }
125 }
126
127 ESP_LOGI(TAG, "Memory record, num: %d, size: %d", g_mem_count, total_size);
128}

130void espnow_mem_print_task()
131{
132#if ( ( configUSE_TRACE_FACILITY == 1 ) && ( configUSE_STATS_FORMATTING_FUNCTIONS > 0 ) )
133 TaskStatus_t *pxTaskStatusArray = NULL;
134 volatile UBaseType_t uxArraySize = 0;
135 uint32_t ulTotalRunTime = 0, ulStatsAsPercentage = 0, ulRunTimeCounte = 0;
136 const char task_status_char[] = {'r', 'R', 'B', 'S', 'D'};
137
138 /* Take a snapshot of the number of tasks in case it changes while this
139 function is executing. */
140 uxArraySize = uxTaskGetNumberOfTasks();
141 pxTaskStatusArray = malloc(uxTaskGetNumberOfTasks() * sizeof(TaskStatus_t));
142
143 if (!pxTaskStatusArray) {
144 return ;
145 }
146
147 /* Generate the (binary) data. */
148 uxArraySize = uxTaskGetSystemState(pxTaskStatusArray, uxArraySize, &ulTotalRunTime);
149 ulTotalRunTime /= 100UL;
150
151 ESP_LOGI(TAG, "---------------- The State Of Tasks ----------------");
152 ESP_LOGI(TAG, "- HWM : usage high water mark (Byte)");
153 ESP_LOGI(TAG, "- Status: blocked ('B'), ready ('R'), deleted ('D') or suspended ('S')");
154 ESP_LOGI(TAG, "TaskName\t\tStatus\tPrio\tHWM\tTaskNum\tCoreID\tRunTimeCounter\tPercentage");
155
156 for (int i = 0; i < uxArraySize; i++) {
157#if( configGENERATE_RUN_TIME_STATS == 1 )
158 ulRunTimeCounte = pxTaskStatusArray[i].ulRunTimeCounter;
159 ulStatsAsPercentage = ulRunTimeCounte / ulTotalRunTime;
160#else
161#warning configGENERATE_RUN_TIME_STATS must also be set to 1 in FreeRTOSConfig.h to use vTaskGetRunTimeStats().
162#endif
163
164 int core_id = -1;
165 char precentage_char[4] = {0};
166
167#if ( configTASKLIST_INCLUDE_COREID == 1 )
168 core_id = (int) pxTaskStatusArray[ i ].xCoreID;
169#else
170#warning configTASKLIST_INCLUDE_COREID must also be set to 1 in FreeRTOSConfig.h to use xCoreID.
171#endif
172
173 /* Write the rest of the string. */
174 ESP_LOGI(TAG, "%-16s\t%c\t%u\t%u\t%u\t%hd\t%-16u%-s%%",
175 pxTaskStatusArray[i].pcTaskName, task_status_char[pxTaskStatusArray[i].eCurrentState],
176 (uint32_t) pxTaskStatusArray[i].uxCurrentPriority,
177 (uint32_t) pxTaskStatusArray[i].usStackHighWaterMark,
178 (uint32_t) pxTaskStatusArray[i].xTaskNumber, core_id,
179 ulRunTimeCounte, (ulStatsAsPercentage <= 0) ? "<1" : itoa(ulStatsAsPercentage, precentage_char, 10));
180 }
181
182 free(pxTaskStatusArray);
183#endif /**< ( configUSE_TRACE_FACILITY == 1 ) && ( configUSE_STATS_FORMATTING_FUNCTIONS > 0 */
184}
