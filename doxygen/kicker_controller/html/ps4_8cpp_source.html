96 ESP_LOGI("USB", "HID Device, protocol DISCONNECTED");
97 ESP_ERROR_CHECK(hid_host_device_close(hid_device_handle));
98 if (xSemaphoreTake(main_menu_values_mutex, portMAX_DELAY)) {

109 ESP_LOGI("USB", "HID Device, protocol TRANSFER_ERROR");

112 ESP_LOGE("USB", "HID Device, protocol Unhandled event");
113 break;

162void ps4::usb_lib_task(void *arg) {
163 const usb_host_config_t host_config = {
164 .skip_phy_setup = false,
165 .intr_flags = ESP_INTR_FLAG_LEVEL1,
166 .enum_filter_cb = NULL};
167
168 ESP_ERROR_CHECK(usb_host_install(&host_config));
169 xTaskNotifyGive((TaskHandle_t) arg);
170
171 while (true) {
172 uint32_t event_flags;
173 usb_host_lib_handle_events(portMAX_DELAY, &event_flags);
174 // In this example, there is only one client registered
175 // So, once we deregister the client, this call must succeed with ESP_OK
176 if (event_flags & USB_HOST_LIB_EVENT_FLAGS_NO_CLIENTS) {
177 ESP_ERROR_CHECK(usb_host_device_free_all());
178 break;
179 }
180 }
181
182 ESP_LOGI("USB", "USB shutdown");
183 // Clean up USB Host
184 vTaskDelay(usb_lib_disconnect_wait_time); // Short delay to allow clients clean-up
185 ESP_ERROR_CHECK(usb_host_uninstall());
186 vTaskDelete(NULL);
187}
