48/**
49 * @brief A function which implements wifi scan command.
50 */
51static esp_err_t wifi_scan_func(int argc, char **argv)
52{
53 if (arg_parse(argc, argv, (void **) &wifi_scan_args) != ESP_OK) {
54 arg_print_errors(stderr, wifi_scan_args.end, argv[0]);
55 return ESP_FAIL;
56 }
57
58 int8_t filter_rssi = -120;
59 uint16_t ap_number = 0;
60 uint8_t bssid[6] = {0x0};
61 wifi_ap_record_t *ap_record = NULL;
62 wifi_scan_config_t scan_config = {
63 .show_hidden = true,
64 .scan_type = WIFI_SCAN_TYPE_ACTIVE,
65 };
66
67 if (wifi_scan_args.passive->count) {
68 scan_config.scan_type = WIFI_SCAN_TYPE_PASSIVE;
69 scan_config.scan_time.passive = wifi_scan_args.passive->ival[0];
70 }
71
72 if (wifi_scan_args.ssid->count) {
73 scan_config.ssid = (uint8_t *)wifi_scan_args.ssid->sval[0];
74 }
75
76 if (wifi_scan_args.bssid->count) {
77 ESP_ERROR_RETURN(!espnow_mac_str2hex(wifi_scan_args.bssid->sval[0], bssid), ESP_ERR_INVALID_ARG,
78 "The format of the address is incorrect. Please enter the format as xx:xx:xx:xx:xx:xx");
79 scan_config.bssid = bssid;
80 }
81
82 if (wifi_scan_args.rssi->count) {
83 filter_rssi = wifi_scan_args.rssi->ival[0];
84 ESP_LOGW(TAG, "filter_rssi: %d", filter_rssi);
85 }
86
87 esp_wifi_scan_stop();
88
89 int retry_count = 20;
90
91 do {
92 esp_wifi_scan_start(&scan_config, true);
93 esp_wifi_scan_get_ap_num(&ap_number);
94 } while (ap_number <= 0 && --retry_count);
95
96 ESP_ERROR_RETURN(ap_number <= 0, ESP_FAIL, "esp_wifi_scan_get_ap_num");
97 ESP_LOGI(TAG, "Get number of APs found, number: %d", ap_number);
98
99 ap_record = ESP_MALLOC(sizeof(wifi_ap_record_t) * ap_number);
100 ESP_ERROR_RETURN(ap_record == NULL, ESP_FAIL, "malloc ap_record");
101 memset(ap_record, 0, sizeof(wifi_ap_record_t) * ap_number);
102
103 int ret = esp_wifi_scan_get_ap_records(&ap_number, ap_record);
104 if (ret != ESP_OK) {
105 ESP_FREE(ap_record);
106 ESP_ERROR_RETURN(1, ret, "esp_wifi_scan_get_ap_records");
107 }
108
109 for (int i = 0; i < ap_number; i++) {
110
111 if (ap_record[i].rssi < filter_rssi) {
112 continue;
113 }
114
115 ESP_LOGI(TAG, "Router, ssid: %s, bssid: " MACSTR ", channel: %u, rssi: %d",
116 ap_record[i].ssid, MAC2STR(ap_record[i].bssid),
117 ap_record[i].primary, ap_record[i].rssi);
118 }
119
120 ESP_FREE(ap_record);
121
122 return ESP_OK;
123}

159/**
160 * @brief A function which implements wifi config command.
161 */
162static int wifi_config_func(int argc, char **argv)
163{
164 if (arg_parse(argc, argv, (void **) &wifi_config_args) != ESP_OK) {
165 arg_print_errors(stderr, wifi_config_args.end, argv[0]);
166 return ESP_FAIL;
167 }
168
169 esp_err_t ret = ESP_OK;
170 wifi_config_t wifi_config = {0x0};
171
172 if (wifi_config_args.ssid->count) {
173 strcpy((char *)wifi_config.sta.ssid, wifi_config_args.ssid->sval[0]);
174 }
175
176 if (wifi_config_args.password->count) {
177 strcpy((char *)wifi_config.sta.password, wifi_config_args.password->sval[0]);
178 }
179
180 if (wifi_config_args.bssid->count) {
181 ESP_ERROR_RETURN(!espnow_mac_str2hex(wifi_config_args.bssid->sval[0], wifi_config.sta.bssid), ESP_ERR_INVALID_ARG,
182 "The format of the address is incorrect. Please enter the format as xx:xx:xx:xx:xx:xx");
183 }
184
185 if (wifi_config_args.disconnt->count) {
186 ret = esp_wifi_disconnect();
187 ESP_ERROR_RETURN(ret != ESP_OK, ret, "esp_wifi_disconnect");
188 }
189
190 if (strlen((char *)wifi_config.sta.ssid)) {
191 ret = esp_wifi_set_config(ESP_IF_WIFI_STA, &wifi_config);
192 ESP_ERROR_RETURN(ret != ESP_OK, ret, "esp_wifi_set_config");
193
194 ret = esp_wifi_connect();
195 ESP_ERROR_RETURN(ret != ESP_OK, ret, "esp_wifi_connect");
196 }
197
198 if (wifi_config_args.country_code->count) {
199 wifi_country_t country = {0};
200 const char *country_code = wifi_config_args.country_code->sval[0];
201
202 if (!strcasecmp(country_code, "US")) {
203 strcpy(country.cc, "US");
204 country.schan = 1;
205 country.nchan = 11;
206 } else if (!strcasecmp(country_code, "JP")) {
207 strcpy(country.cc, "JP");
208 country.schan = 1;
209 country.nchan = 14;
210 } else if (!strcasecmp(country_code, "CN")) {
211 strcpy(country.cc, "CN");
212 country.schan = 1;
213 country.nchan = 13;
214 } else {
215 return ESP_ERR_INVALID_ARG;
216 }
217
218 ret = esp_wifi_set_country(&country);
219 }
220
221 if (wifi_config_args.channel->count) {
222 wifi_config.sta.channel = wifi_config_args.channel->ival[0];
223
224 if (wifi_config.sta.channel > 0 && wifi_config.sta.channel <= 14) {
225 ret = esp_wifi_set_channel(wifi_config.sta.channel, WIFI_SECOND_CHAN_NONE);
226 ESP_ERROR_RETURN(ret != ESP_OK, ret, "esp_wifi_set_channel");
227 ESP_LOGI(__func__, "Set Channel, channel: %d", wifi_config.sta.channel);
228 } else {
229 return ESP_ERR_INVALID_ARG;
230 }
231 }
232
233 if (wifi_config_args.tx_power->count) {
234 esp_wifi_set_max_tx_power(wifi_config_args.tx_power->ival[0]);
235 }
236
237 if (wifi_config_args.info->count) {
238 int8_t rx_power = 0;
239 wifi_country_t country = {0};
240 uint8_t primary = 0;
241 wifi_second_chan_t second = 0;
242 wifi_mode_t mode = 0;
243
244 esp_wifi_get_channel(&primary, &second);
245 esp_wifi_get_max_tx_power(&rx_power);
246 esp_wifi_get_country(&country);
247 esp_wifi_get_mode(&mode);
248
249 extern wifi_pkt_rx_ctrl_t g_rx_ctrl;
250
251 country.cc[2] = '\0';
252 ESP_LOGI(__func__, "rx: %d, tx_power: %d, country: %s, channel: %d, mode: %d",
253 g_rx_ctrl.rssi, rx_power, country.cc, primary, mode);
254 }
255
256 return ret;
257}

284static void cmd_ping_on_ping_success(esp_ping_handle_t hdl, void *args)
285{
286 uint8_t ttl;
287 uint16_t seqno;
288 uint32_t elapsed_time, recv_len;
289 ip_addr_t target_addr;
290 esp_ping_get_profile(hdl, ESP_PING_PROF_SEQNO, &seqno, sizeof(seqno));
291 esp_ping_get_profile(hdl, ESP_PING_PROF_TTL, &ttl, sizeof(ttl));
292 esp_ping_get_profile(hdl, ESP_PING_PROF_IPADDR, &target_addr, sizeof(target_addr));
293 esp_ping_get_profile(hdl, ESP_PING_PROF_SIZE, &recv_len, sizeof(recv_len));
294 esp_ping_get_profile(hdl, ESP_PING_PROF_TIMEGAP, &elapsed_time, sizeof(elapsed_time));
295 ESP_LOGI(TAG, "%d bytes from %s icmp_seq=%d ttl=%d time=%d ms",
296 recv_len, ipaddr_ntoa((ip_addr_t*)&target_addr), seqno, ttl, elapsed_time);
297}

308static void cmd_ping_on_ping_end(esp_ping_handle_t hdl, void *args)
309{
310 ip_addr_t target_addr;
311 uint32_t transmitted;
312 uint32_t received;
313 uint32_t total_time_ms;
314 esp_ping_get_profile(hdl, ESP_PING_PROF_REQUEST, &transmitted, sizeof(transmitted));
315 esp_ping_get_profile(hdl, ESP_PING_PROF_REPLY, &received, sizeof(received));
316 esp_ping_get_profile(hdl, ESP_PING_PROF_IPADDR, &target_addr, sizeof(target_addr));
317 esp_ping_get_profile(hdl, ESP_PING_PROF_DURATION, &total_time_ms, sizeof(total_time_ms));
318 uint32_t loss = (uint32_t)((1 - ((float)received) / transmitted) * 100);
319 if (IP_IS_V4(&target_addr)) {
320 ESP_LOGI(TAG, "\n--- %s ping statistics ---", inet_ntoa(*ip_2_ip4(&target_addr)));
321#if CONFIG_LWIP_IPV6
322 } else {
323 ESP_LOGI(TAG, "\n--- %s ping statistics ---", inet6_ntoa(*ip_2_ip6(&target_addr)));
324#endif
325 }
326 ESP_LOGI(TAG, "%d packets transmitted, %d received, %d%% packet loss, time %dms",
327 transmitted, received, loss, total_time_ms);
328 // delete the ping sessions, so that we clean up all resources and can create a new ping session
329 // we don't have to call delete function in the callback, instead we can call delete function from other tasks
330 esp_ping_delete_session(hdl);
331}

343static int do_ping_cmd(int argc, char **argv)
344{
345 esp_ping_config_t config = ESP_PING_DEFAULT_CONFIG();
346 config.task_stack_size = 3072;
347
348 int nerrors = arg_parse(argc, argv, (void **)&ping_args);
349 if (nerrors != 0) {
350 arg_print_errors(stderr, ping_args.end, argv[0]);
351 return 1;
352 }
353
354 if (ping_args.timeout->count > 0) {
355 config.timeout_ms = (uint32_t)(ping_args.timeout->dval[0] * 1000);
356 }
357
358 if (ping_args.interval->count > 0) {
359 config.interval_ms = (uint32_t)(ping_args.interval->dval[0] * 1000);
360 }
361
362 if (ping_args.data_size->count > 0) {
363 config.data_size = (uint32_t)(ping_args.data_size->ival[0]);
364 }
365
366 if (ping_args.count->count > 0) {
367 config.count = (uint32_t)(ping_args.count->ival[0]);
368 }
369
370 if (ping_args.tos->count > 0) {
371 config.tos = (uint32_t)(ping_args.tos->ival[0]);
372 }
373
374 // parse IP address
375 ip_addr_t target_addr;
376 memset(&target_addr, 0, sizeof(target_addr));
377
378#if CONFIG_LWIP_IPV6
379 struct sockaddr_in6 sock_addr6;
380 if (inet_pton(AF_INET6, ping_args.host->sval[0], &sock_addr6.sin6_addr) == 1) {
381 /* convert ip6 string to ip6 address */
382 ipaddr_aton(ping_args.host->sval[0], &target_addr);
383 } else {
384#else
385 {
386#endif
387 struct addrinfo hint;
388 struct addrinfo *res = NULL;
389 memset(&hint, 0, sizeof(hint));
390 /* convert ip4 string or hostname to ip4 or ip6 address */
391 if (getaddrinfo(ping_args.host->sval[0], NULL, &hint, &res) != 0) {
392 ESP_LOGI(TAG, "ping: unknown host %s", ping_args.host->sval[0]);
393 return 1;
394 }
395 if (res->ai_family == AF_INET) {
396 struct in_addr addr4 = ((struct sockaddr_in *) (res->ai_addr))->sin_addr;
397 inet_addr_to_ip4addr(ip_2_ip4(&target_addr), &addr4);
398#if CONFIG_LWIP_IPV6
399 } else {
400 struct in6_addr addr6 = ((struct sockaddr_in6 *) (res->ai_addr))->sin6_addr;
401 inet6_addr_to_ip6addr(ip_2_ip6(&target_addr), &addr6);
402#endif
403 }
404 freeaddrinfo(res);
405 }
406 config.target_addr = target_addr;
407
408 /* set callback functions */
409 esp_ping_callbacks_t cbs = {
410 .on_ping_success = cmd_ping_on_ping_success,
411 .on_ping_timeout = cmd_ping_on_ping_timeout,
412 .on_ping_end = cmd_ping_on_ping_end,
413 .cb_args = NULL
414 };
415 esp_ping_handle_t ping;
416 esp_ping_new_session(&config, &cbs, &ping);
417 esp_ping_start(ping);
418
419 return 0;
420}
