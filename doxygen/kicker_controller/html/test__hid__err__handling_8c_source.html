18/**
19 * @brief USB HID Host interface callback.
20 *
21 * Handle close event only.
22 *
23 * @param[in] event HID Host device event
24 * @param[in] arg Pointer to arguments, does not used
25 *
26 */
27static void test_hid_host_interface_event_close(hid_host_device_handle_t hid_device_handle,
28 const hid_host_interface_event_t event,
29 void *arg)
30{
31 switch (event) {
32 case HID_HOST_INTERFACE_EVENT_INPUT_REPORT:
33 case HID_HOST_INTERFACE_EVENT_TRANSFER_ERROR:
34 break;
35 case HID_HOST_INTERFACE_EVENT_DISCONNECTED:
36 TEST_ASSERT_EQUAL(ESP_OK, hid_host_device_close(hid_device_handle) );
37 break;
38 }
39}
