32/**
33 * @brief A function which implements sdcard command.
34 */
35static int sdcard_func(int argc, char **argv)
36{
37 if (arg_parse(argc, argv, (void **) &sdcard_args) != ESP_OK) {
38 arg_print_errors(stderr, sdcard_args.end, argv[0]);
39 return ESP_FAIL;
40 }
41
42 if (sdcard_args.list->count) {
43 sdcard_list_file(sdcard_args.list->sval[0]);
44 }
45
46 if (sdcard_args.remove->count) {
47 sdcard_remove_file(sdcard_args.remove->sval[0]);
48 }
49
50 if (sdcard_args.output->count) {
51 file_format_t type = FILE_TYPE_STRING;
52
53 if (sdcard_args.type->count) {
54 if (!strcasecmp(sdcard_args.type->sval[0], "string")) {
55 type = FILE_TYPE_STRING;
56 } else if (!strcasecmp(sdcard_args.type->sval[0], "hex")) {
57 type = FILE_TYPE_HEX;
58 } else if (!strcasecmp(sdcard_args.type->sval[0], "base64")) {
59 type = FILE_TYPE_BASE64;
60 } else if (!strcasecmp(sdcard_args.type->sval[0], "bin")) {
61 type = FILE_TYPE_BIN;
62 } else {
63 type = FILE_TYPE_NONE;
64 }
65 }
66
67 sdcard_print_file(sdcard_args.output->sval[0], type, INT32_MAX);
68 }
69
70 return ESP_OK;
71}
