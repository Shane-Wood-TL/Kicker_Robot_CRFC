58esp_err_t espnow_sec_setkey(espnow_sec_t *sec, uint8_t app_key[APP_KEY_LEN])
59{
60 ESP_PARAM_CHECK(sec);
61 ESP_PARAM_CHECK(app_key);
62 ESP_PARAM_CHECK(sec->cipher_ctx);
63
64 int ret = mbedtls_ccm_setkey((mbedtls_ccm_context *)sec->cipher_ctx, MBEDTLS_CIPHER_ID_AES, app_key, 8 * sec->key_len);
65 ESP_ERROR_RETURN(ret != ESP_OK, ret, "mbedtls_ccm_setkey %x", ret);
66
67 memcpy(sec->key, app_key, sec->key_len);
68 memcpy(sec->iv, app_key + sec->key_len, sec->iv_len);
69 sec->state = ESPNOW_SEC_OVER;
70
71 return ESP_OK;

74esp_err_t espnow_sec_auth_encrypt(espnow_sec_t *sec, const uint8_t *input, size_t ilen,
75 uint8_t *output, size_t output_len,
76 size_t *olen, size_t tag_len)
77{
78 ESP_PARAM_CHECK(sec);
79 ESP_PARAM_CHECK(input);
80 ESP_PARAM_CHECK(ilen);
81 ESP_PARAM_CHECK(output);
82 ESP_PARAM_CHECK(olen);
83 ESP_PARAM_CHECK(output_len >= ilen + tag_len);
84 ESP_PARAM_CHECK(tag_len);
85
86 if (sec->state != ESPNOW_SEC_OVER) {
87 ESP_LOGE(TAG, "Security state is not over");
88 return ESP_FAIL;
89 }
90
91 int ret = ESP_OK;
92 ret = mbedtls_ccm_encrypt_and_tag((mbedtls_ccm_context *)sec->cipher_ctx, ilen, sec->iv, sec->iv_len, NULL, 0,
93 input, output, output + ilen, tag_len);
94 *olen = ilen + tag_len;
95
96 if (ret != 0) {
97 ESP_LOGE(TAG, "Failed at mbedtls_ccm_encrypt_and_tag with error code : %d", ret);
98 return ESP_FAIL;
99 }
100
101 return ret;
102}
