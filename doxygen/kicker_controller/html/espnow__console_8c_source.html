46/* Console command history can be stored to and loaded from a file.
47 * The easiest way to do this is to use FATFS filesystem on top of
48 * wear_levelling library.

52static void initialize_filesystem(const espnow_console_config_t *config)
53{
54 esp_vfs_spiffs_conf_t conf = {
55 .base_path = config->store_history.base_path,
56 .partition_label = config->store_history.partition_label,
57 .max_files = 4,
58 .format_if_mount_failed = true
59 };
60
61 ESP_LOGD(TAG, "base_path: %s", config->store_history.base_path);
62
63 esp_err_t ret = esp_vfs_spiffs_register(&conf);
64
65 if (ret != ESP_OK) {
66 if (ret == ESP_FAIL) {
67 ESP_LOGE(TAG, "Failed to mount or format filesystem");
68 } else if (ret == ESP_ERR_NOT_FOUND) {
69 ESP_LOGE(TAG, "Failed to find SPIFFS partition");
70 } else {
71 ESP_LOGE(TAG, "Failed to initialize SPIFFS (%s)", esp_err_to_name(ret));
72 }
73
74 return;
75 }
76
77 size_t total = 0, used = 0;
78 ret = esp_spiffs_info(conf.partition_label, &total, &used);
79
80 if (ret != ESP_OK) {
81 ESP_LOGE(TAG, "Failed to get SPIFFS partition information (%s)", esp_err_to_name(ret));
82 } else {
83 ESP_LOGI(TAG, "Partition size: total: %d, used: %d", total, used);
84 }
85}

87static void initialize_console(void)
88{
89 /* Initialize the console */
90 esp_console_config_t console_config = {
91 .max_cmdline_args = 16,
92 .max_cmdline_length = 512,
93#if CONFIG_LOG_COLORS
94 .hint_color = atoi(LOG_COLOR_CYAN)
95#endif
96 };
97 ESP_ERROR_CHECK(esp_console_init(&console_config));
98
99 /* Configure linenoise line completion library */
100 /* Enable multiline editing. If not set, long commands will scroll within
101 * single line.
102 */
103 linenoiseSetMultiLine(1);
104
105 /* Tell linenoise where to get command completions and hints */
106 linenoiseSetCompletionCallback(&esp_console_get_completion);
107 linenoiseSetHintsCallback((linenoiseHintsCallback *) &esp_console_get_hint);
108
109 /* Set command history size */
110 linenoiseHistorySetMaxLen(100);
111
112 /* Don't return empty lines */
113 linenoiseAllowEmpty(false);
114}

116static void console_uart_handle_task(void *arg)
117{
118 const char *prompt = LOG_COLOR_I PROMPT_STR "> " LOG_RESET_COLOR;
119
120#if CONFIG_LOG_COLORS
121 /* Since the terminal doesn't support escape sequences,
122 * don't use color codes in the prompt.
123 */
124 prompt = PROMPT_STR "> ";
125#endif //CONFIG_LOG_COLORS
126
127 while (g_running_flag) {
128 /* Get a line using linenoise.
129 * The line is returned when ENTER is pressed.
130 */
131 char *line = linenoise(prompt);
132
133 if (line == NULL) { /* Break on EOF or error */
134 continue;
135 }
136
137 /* Add the command to the history if not empty*/
138 if (strlen(line) > 0) {
139 linenoiseHistoryAdd(line);
140
141 if (g_store_history_filename) {
142 /* Save command history to filesystem */
143 linenoiseHistorySave(g_store_history_filename);
144 }
145 }
146
147 /* Try to run the command */
148 int ret;
149 esp_err_t err = esp_console_run(line, &ret);
150
151 if (err == ESP_ERR_NOT_FOUND) {
152 ESP_LOGW(TAG, "Unrecognized command");
153 } else if (err == ESP_ERR_INVALID_ARG) {
154 // command was empty
155 } else if (err == ESP_OK && ret != ESP_OK) {
156 ESP_LOGW(TAG, "Command returned non-zero error code: 0x%x (%s)", ret, esp_err_to_name(ret));
157 } else if (err != ESP_OK) {
158 ESP_LOGW(TAG, "Internal error: %s", esp_err_to_name(err));
159 }
160
161 /* linenoise allocates line buffer on the heap, so need to free it */
162 linenoiseFree(line);
163 }
164
165 ESP_LOGE(TAG, "Error or end-of-input, terminating console");
166 vTaskDelete(NULL);
167}

169static esp_err_t console_espnow_handle(uint8_t *src_addr, void *data,
170 size_t size, wifi_pkt_rx_ctrl_t *rx_ctrl)
171{
172 ESP_PARAM_CHECK(src_addr);
173 ESP_PARAM_CHECK(data);
174 ESP_PARAM_CHECK(size);
175 ESP_PARAM_CHECK(rx_ctrl);
176
177 esp_err_t ret = ESP_OK;
178
179 extern wifi_pkt_rx_ctrl_t g_rx_ctrl;
180 extern uint8_t g_src_addr[ESPNOW_ADDR_LEN];
181
182 if (g_running_flag) {
183 memcpy(&g_rx_ctrl, rx_ctrl, sizeof(wifi_pkt_rx_ctrl_t));
184 memcpy(g_src_addr, src_addr, 6);
185
186 esp_err_t err = esp_console_run(data, &ret);
187
188 if (err == ESP_ERR_NOT_FOUND) {
189 ESP_LOGE(TAG, "Unrecognized command");
190 } else if (err == ESP_ERR_INVALID_ARG) {
191 /**< Command was empty */
192 } else if (err == ESP_OK && ret != ESP_OK) {
193 ESP_LOGW(TAG, "Command returned non-zero error code: 0x%x (%s)", ret, esp_err_to_name(ret));
194 } else if (err != ESP_OK) {
195 ESP_LOGW(TAG, "Internal error: %s", esp_err_to_name(err));
196 }
197 }
198
199 return ret;
200}

202esp_err_t espnow_console_init(const espnow_console_config_t *config)
203{
204 if (config->monitor_command.uart) {
205 /* Drain stdout before reconfiguring it */
206 fflush(stdout);
207 fsync(fileno(stdout));
208
209#if CONFIG_ESP_CONSOLE_UART
210 /* Disable buffering on stdin */
211 setvbuf(stdin, NULL, _IONBF, 0);
212
213#if ESP_IDF_VERSION >= ESP_IDF_VERSION_VAL(5, 3, 0)
214 /* Minicom, screen, idf_monitor send CR when ENTER key is pressed */
215 uart_vfs_dev_port_set_rx_line_endings(CONFIG_ESP_CONSOLE_UART_NUM, ESP_LINE_ENDINGS_CR);
216 /* Move the caret to the beginning of the next line on '\n' */
217 uart_vfs_dev_port_set_tx_line_endings(CONFIG_ESP_CONSOLE_UART_NUM, ESP_LINE_ENDINGS_CRLF);
218#else
219 /* Minicom, screen, idf_monitor send CR when ENTER key is pressed */
220 esp_vfs_dev_uart_port_set_rx_line_endings(CONFIG_ESP_CONSOLE_UART_NUM, ESP_LINE_ENDINGS_CR);
221 /* Move the caret to the beginning of the next line on '\n' */
222 esp_vfs_dev_uart_port_set_tx_line_endings(CONFIG_ESP_CONSOLE_UART_NUM, ESP_LINE_ENDINGS_CRLF);
223#endif
224
225 /* Configure UART. Note that REF_TICK is used so that the baud rate remains
226 * correct while APB frequency is changing in light sleep mode.
227 */
228 const uart_config_t uart_config = {
229 .baud_rate = CONFIG_ESP_CONSOLE_UART_BAUDRATE,
230 .data_bits = UART_DATA_8_BITS,
231 .parity = UART_PARITY_DISABLE,
232 .stop_bits = UART_STOP_BITS_1,
233#if CONFIG_IDF_TARGET_ESP32 || CONFIG_IDF_TARGET_ESP32S2
234 .source_clk = UART_SCLK_REF_TICK,
235#else
236 .source_clk = UART_SCLK_XTAL,
237#endif
238 };
239 /* Install UART driver for interrupt-driven reads and writes */
240 ESP_ERROR_CHECK(uart_driver_install(CONFIG_ESP_CONSOLE_UART_NUM,
241 256, 0, 0, NULL, 0));
242 ESP_ERROR_CHECK(uart_param_config(CONFIG_ESP_CONSOLE_UART_NUM, &uart_config));
243
244#if ESP_IDF_VERSION >= ESP_IDF_VERSION_VAL(5, 3, 0)
245 /* Tell VFS to use UART driver */
246 uart_vfs_dev_use_driver(CONFIG_ESP_CONSOLE_UART_NUM);
247#else
248 /* Tell VFS to use UART driver */
249 esp_vfs_dev_uart_use_driver(CONFIG_ESP_CONSOLE_UART_NUM);
250#endif
251
252#elif CONFIG_ESP_CONSOLE_USB_SERIAL_JTAG
253 /* Disable buffering on stdin */
254 setvbuf(stdin, NULL, _IONBF, 0);
255
256 /* Minicom, screen, idf_monitor send CR when ENTER key is pressed */
257 esp_vfs_dev_usb_serial_jtag_set_rx_line_endings(ESP_LINE_ENDINGS_CR);
258 /* Move the caret to the beginning of the next line on '\n' */
259 esp_vfs_dev_usb_serial_jtag_set_tx_line_endings(ESP_LINE_ENDINGS_CRLF);
260
261 /* Enable non-blocking mode on stdin and stdout */
262 fcntl(fileno(stdout), F_SETFL, 0);
263 fcntl(fileno(stdin), F_SETFL, 0);
264
265 usb_serial_jtag_driver_config_t usb_serial_jtag_config = USB_SERIAL_JTAG_DRIVER_CONFIG_DEFAULT();
266
267 /* Install USB-SERIAL-JTAG driver for interrupt-driven reads and writes */
268 ESP_ERROR_CHECK(usb_serial_jtag_driver_install(&usb_serial_jtag_config));
269
270 /* Tell vfs to use usb-serial-jtag driver */
271 esp_vfs_usb_serial_jtag_use_driver();
272
273#elif CONFIG_ESP_CONSOLE_USB_CDC
274 /* Minicom, screen, idf_monitor send CR when ENTER key is pressed */
275 esp_vfs_dev_cdcacm_set_rx_line_endings(ESP_LINE_ENDINGS_CR);
276 /* Move the caret to the beginning of the next line on '\n' */
277 esp_vfs_dev_cdcacm_set_tx_line_endings(ESP_LINE_ENDINGS_CRLF);
278
279 /* Enable non-blocking mode on stdin and stdout */
280 fcntl(fileno(stdout), F_SETFL, 0);
281 fcntl(fileno(stdin), F_SETFL, 0);
282#else
283 #error "Unknown/unsupported console type"
284#endif
285
286 }
287
288 initialize_console();
289
290 if (config->store_history.base_path) {
291 initialize_filesystem(config);
292 /* Load command history from filesystem */
293 asprintf(&g_store_history_filename, "%s/history.txt", config->store_history.base_path);
294
295 linenoiseHistoryLoad(g_store_history_filename);
296 ESP_LOGI(TAG, "Command history enabled");
297 }
298
299 /* Register commands */
300 esp_console_register_help_command();
301
302 g_running_flag = true;
303
304 if (config->monitor_command.uart) {
305 xTaskCreate(console_uart_handle_task, "console_uart", 1024 * 4, NULL, tskIDLE_PRIORITY + 1, NULL);
306 }
307
308 if (config->monitor_command.espnow) {
309 espnow_set_config_for_data_type(ESPNOW_DATA_TYPE_DEBUG_COMMAND, 1, console_espnow_handle);
310 }
311
312 return ESP_OK;
313}

315esp_err_t espnow_console_deinit()
316{
317 esp_err_t ret = ESP_OK;
318 g_running_flag = false;
319 espnow_set_config_for_data_type(ESPNOW_DATA_TYPE_DEBUG_COMMAND, 0, NULL);
320
321 ret = esp_console_deinit();
322 ESP_ERROR_RETURN(ret != ESP_OK, ret, "de-initialize console module");
323
324 ESP_FREE(g_store_history_filename);
325
326 return ESP_OK;
