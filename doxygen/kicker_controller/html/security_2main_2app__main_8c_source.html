42static void app_espnow_event_handler(void *handler_args, esp_event_base_t base, int32_t id, void *event_data)
43{
44 if (base != ESP_EVENT_ESPNOW) {
45 return;
46 }
47
48 switch (id) {
49 case ESP_EVENT_ESPNOW_SEC_OK:
50 ESP_LOGI(TAG, "ESP_EVENT_ESPNOW_SEC_OK [" MACSTR "]", MAC2STR((uint8_t *)event_data));
51 s_sec_flag = true;
52 break;
53
54 case ESP_EVENT_ESPNOW_SEC_FAIL:
55 ESP_LOGI(TAG, "ESP_EVENT_ESPNOW_SEC_FAIL [" MACSTR "]", MAC2STR((uint8_t *)event_data));
56 s_sec_flag = false;
57 break;
58
59 default:
60 break;
61 }
62}
63#endif

65static void app_uart_read_task(void *arg)
66{
67 esp_err_t ret = ESP_OK;
68 uint32_t count = 0;
69 size_t size = 0;
70 uint8_t *data = ESP_CALLOC(1, ESPNOW_SEC_PACKET_MAX_SIZE);
71
72 ESP_LOGI(TAG, "Uart read handle task is running");
73
74 espnow_frame_head_t frame_head = {
75 .retransmit_count = CONFIG_APP_ESPNOW_RETRY_NUM,
76 .broadcast = true,
77 .security = CONFIG_APP_ESPNOW_SEC_OPTION
78 };
79
80 for (;;) {
81 size = uart_read_bytes(UART_PORT_NUM, data, ESPNOW_SEC_PACKET_MAX_SIZE, pdMS_TO_TICKS(10));
82 ESP_ERROR_CONTINUE(size <= 0, "");
83
84 frame_head.security = s_sec_flag;
85#ifdef CONFIG_APP_ESNOW_STRESS_OPTION
86 frame_head.ack = false;
87 uint32_t stress_data = 0;
88 while (stress_data != 0xFFFFFFFF) {
89 stress_data ++;
90 ret = espnow_send(ESPNOW_DATA_TYPE_DATA, ESPNOW_ADDR_BROADCAST, &stress_data, sizeof(uint32_t), &frame_head, 0);
91 if (ret != ESP_OK) {
92 ESP_LOG_BUFFER_HEX_LEVEL(TAG, &stress_data, sizeof(uint32_t), ESP_LOG_ERROR);
93 ESP_ERROR_CONTINUE(ret != ESP_OK, "<%s> espnow_send", esp_err_to_name(ret));
94 }
95
96 vTaskDelay(pdMS_TO_TICKS(20));
97 }
98#else
99 ret = espnow_send(ESPNOW_DATA_TYPE_DATA, ESPNOW_ADDR_BROADCAST, data, size, &frame_head, portMAX_DELAY);
100 ESP_ERROR_CONTINUE(ret != ESP_OK, "<%s> espnow_send", esp_err_to_name(ret));
101
102 ESP_LOGI(TAG, "espnow_send, count: %" PRIu32 ", size: %u, %s data: %s", count++, size, s_sec_flag ? "ciphertext" : "plaintext", data);
103 memset(data, 0, ESPNOW_DATA_LEN);
104#endif
105 }
106
107 ESP_LOGI(TAG, "Uart handle task is exit");
108
109 ESP_FREE(data);
110 vTaskDelete(NULL);
111}

166void app_main()
167{
168 espnow_storage_init();
169
170 app_wifi_init();
171 app_uart_initialize();
172
173 espnow_config_t espnow_config = ESPNOW_INIT_CONFIG_DEFAULT();
174 espnow_config.sec_enable = 1;
175 espnow_init(&espnow_config);
176
177 espnow_set_config_for_data_type(ESPNOW_DATA_TYPE_DATA, true, app_uart_write_handle);
178
179#ifdef CONFIG_APP_ESPNOW_SEC_INITIATOR
180 uint8_t key_info[APP_KEY_LEN];
181
182 if (espnow_get_key(key_info) != ESP_OK) {
183 esp_fill_random(key_info, APP_KEY_LEN);
184 }
185 espnow_set_key(key_info);
186 espnow_set_dec_key(key_info);
187
188 uint32_t start_time1 = xTaskGetTickCount();
189 espnow_sec_result_t espnow_sec_result = {0};
190 espnow_sec_responder_t *info_list = NULL;
191 size_t num = 0;
192 espnow_sec_initiator_scan(&info_list, &num, pdMS_TO_TICKS(3000));
193 ESP_LOGW(TAG, "espnow wait security num: %u", num);
194
195 if (num == 0) {
196 ESP_FREE(info_list);
197 return;
198 }
199
200 espnow_addr_t *dest_addr_list = ESP_MALLOC(num * ESPNOW_ADDR_LEN);
201
202 for (size_t i = 0; i < num; i++) {
203 memcpy(dest_addr_list[i], info_list[i].mac, ESPNOW_ADDR_LEN);
204 }
205
206 espnow_sec_initiator_scan_result_free();
207
208 uint32_t start_time2 = xTaskGetTickCount();
209 esp_err_t ret = espnow_sec_initiator_start(key_info, pop_data, dest_addr_list, num, &espnow_sec_result);
210 ESP_ERROR_GOTO(ret != ESP_OK, EXIT, "<%s> espnow_sec_initiator_start", esp_err_to_name(ret));
211 s_sec_flag = true;
212
213 ESP_LOGI(TAG, "App key is sent to the device to complete, Spend time: %" PRId32 "ms, Scan time: %" PRId32 "ms",
214 (xTaskGetTickCount() - start_time1) * portTICK_PERIOD_MS,
215 (start_time2 - start_time1) * portTICK_PERIOD_MS);
216 ESP_LOGI(TAG, "Devices security completed, successed_num: %u, unfinished_num: %u",
217 espnow_sec_result.successed_num, espnow_sec_result.unfinished_num);
218
219EXIT:
220 ESP_FREE(dest_addr_list);
221 espnow_sec_initiator_result_free(&espnow_sec_result);
222#elif CONFIG_APP_ESPNOW_SEC_RESPONDER
223 uint8_t key_info[APP_KEY_LEN];
224
225 /* If espnow_set_key succeed, sending and receiving will be in security mode */
226 if (espnow_get_key(key_info) == ESP_OK) {
227 espnow_set_key(key_info);
228 espnow_set_dec_key(key_info);
229 }
230
231 /* If responder handshake with initiator succeed, espnow_set_key will be executed again. */
232 esp_event_handler_register(ESP_EVENT_ESPNOW, ESP_EVENT_ANY_ID, app_espnow_event_handler, NULL);
233 espnow_sec_responder_start(pop_data);
234#endif
235}
