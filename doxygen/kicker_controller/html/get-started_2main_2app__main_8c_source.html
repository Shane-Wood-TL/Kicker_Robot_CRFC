35static void app_uart_read_task(void *arg)
36{
37 esp_err_t ret = ESP_OK;
38 uint32_t count = 0;
39 size_t size = 0;
40 uint8_t *data = ESP_CALLOC(1, ESPNOW_DATA_LEN);
41
42 ESP_LOGI(TAG, "Uart read handle task is running");
43
44 espnow_frame_head_t frame_head = {
45 .retransmit_count = CONFIG_RETRY_NUM,
46 .broadcast = true,
47 };
48
49 for (;;) {
50 size = uart_read_bytes(UART_PORT_NUM, data, ESPNOW_DATA_LEN, pdMS_TO_TICKS(10));
51 ESP_ERROR_CONTINUE(size <= 0, "");
52
53 ret = espnow_send(ESPNOW_DATA_TYPE_DATA, ESPNOW_ADDR_BROADCAST, data, size, &frame_head, portMAX_DELAY);
54 ESP_ERROR_CONTINUE(ret != ESP_OK, "<%s> espnow_send", esp_err_to_name(ret));
55
56 ESP_LOGI(TAG, "espnow_send, count: %" PRIu32 ", size: %u, data: %s", count++, size, data);
57 memset(data, 0, ESPNOW_DATA_LEN);
58 }
59
60 ESP_LOGI(TAG, "Uart handle task is exit");
61
62 ESP_FREE(data);
63 vTaskDelete(NULL);
64}

66static void app_uart_initialize()
67{
68 uart_config_t uart_config = {
69 .baud_rate = UART_BAUD_RATE,
70 .data_bits = UART_DATA_8_BITS,
71 .parity = UART_PARITY_DISABLE,
72 .stop_bits = UART_STOP_BITS_1,
73 .flow_ctrl = UART_HW_FLOWCTRL_DISABLE,
74#if SOC_UART_SUPPORT_REF_TICK
75 .source_clk = UART_SCLK_REF_TICK,
76#elif SOC_UART_SUPPORT_XTAL_CLK
77 .source_clk = UART_SCLK_XTAL,
78#endif
79 };
80
81 ESP_ERROR_CHECK(uart_param_config(UART_PORT_NUM, &uart_config));
82 ESP_ERROR_CHECK(uart_set_pin(UART_PORT_NUM, UART_TX_IO, UART_RX_IO, UART_PIN_NO_CHANGE, UART_PIN_NO_CHANGE));
83 ESP_ERROR_CHECK(uart_driver_install(UART_PORT_NUM, 8 * ESPNOW_DATA_LEN, 8 * ESPNOW_DATA_LEN, 0, NULL, 0));
84
85 xTaskCreate(app_uart_read_task, "app_uart_read_task", 4 * 1024, NULL, tskIDLE_PRIORITY + 1, NULL);
86}
