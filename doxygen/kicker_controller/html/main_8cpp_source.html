1/**
2 * @file main.cpp
3 * @brief This is the file that contains the main function and all other tasks for the robot controller
4 * @author Shane Wood
5 * @details This is the code for the kicker controller, it was part of the 2024-2025 senior design project. The
6 * controller communicates with a ps4 controller over usb, and transmits data to the robot over esp-now. The
7 * controller allows for various parameters to be tuned using the menu interfaces on the oled display, with all
8 * input coming from the ps4 controller. The controller also displays the battery voltage of the robot on the oled
9 * display and the connection status to the robot and the controller.
10 *
11 * @date 22/12/2024
12 * @note Dates are in the format dd/mm/yyyy
13 *
14 * @note This folder, along with the code for the kicker itself both have version control, intended to be used
15 * with github desktop. A .gitignore file is included in the kicker folder to prevent the inclusion of the build
16 * folder in the repository.
17 *
18 * @note This project has the following subfolders:
19 * - drivers: contains the code that is interfacing with the
20 * hardware
21 * - setup: contains the code that is setting up the
22 * hardware, this code either just runs once or is a
23 * callback function for another task
24 * - supporting: This is anything that is not directly
25 * interfacing with the hardware, but is not a task itself
26 * - tasks: contains the code for the tasks that are running
27 * on the microcontroller
28 */

108/**
109 * @brief the main function for the program, creates all semaphores, tasks, and runs setup functions
110 */
111extern "C" void app_main(void) {
112 // create queues
113 other_controller_data_queue = xQueueCreate(size_of_other_controller_data_queue, sizeof(display_controller_data));
114
115 // create semaphores/mutuexes (espidf does not have
116 // mutexes directly)
117 update_main_display_mutex = xSemaphoreCreateMutex();
118 main_menu_values_mutex = xSemaphoreCreateMutex();
119 ramped_velocity_mutex = xSemaphoreCreateMutex();
120 motor_speeds_mutex = xSemaphoreCreateMutex();
121 servo_status_mutex = xSemaphoreCreateMutex();
122 motor_status_mutex = xSemaphoreCreateMutex();
123 network_channel_mutex = xSemaphoreCreateMutex();
124
125 new_controller_data = xSemaphoreCreateBinary();
126
127 i2c_setup();
128 esp_now_setup();
129
130
131 //wait for i2c and esp_now to setup
132 vTaskDelay(pdMS_TO_TICKS(main_task_wait_time));
133
134 // create tasks
135 xTaskCreate(send_data_task, "SendDataTask", send_data_task_stack_size, NULL, send_data_task_priority, NULL);
136 xTaskCreate(ps4_task, "ps4_task", ps4_task_stack_size, NULL, ps4_task_priority, NULL);
137 xTaskCreate(display_task, "display_task", display_task_stack_size, NULL, display_task_priority, NULL);
138 xTaskCreate(ps4_data_processor, "ps4_data_processor", ps4_data_processor_stack_sze, NULL, ps4_data_processor_priority, NULL);
139
140 while (1) {
141 vTaskDelay(pdMS_TO_TICKS(main_task_wait_time));
142 }
143}
