57static bool addrs_remove(uint8_t addrs_list[][ESPNOW_ADDR_LEN],
58 size_t *addrs_num, const uint8_t addr[6])
59{
60 if (!addrs_list || !addrs_num || !addr) {
61 ESP_LOGE(TAG, "!addrs_list: %p !addrs_num: %p !addr: %p", addrs_list, addrs_num, addr);
62 return false;
63 }
64
65 for (int i = 0; i < *addrs_num; i++) {
66 if (ESPNOW_ADDR_IS_EQUAL(addrs_list[i], addr)) {
67 if (--(*addrs_num)) {
68 memcpy(addrs_list[i], addrs_list[*addrs_num], ESPNOW_ADDR_LEN);
69 }
70
71 return true;
72 }
73 }
74
75 return false;
76}

78static esp_err_t espnow_sec_info(uint8_t *src_addr, void *data,
79 size_t size, wifi_pkt_rx_ctrl_t *rx_ctrl)
80{
81 espnow_sec_info_t *recv_data = (espnow_sec_info_t *)data;
82
83 for (int i = 0; i < g_scan_num; ++i) {
84 if (ESPNOW_ADDR_IS_EQUAL((g_info_list)[i].mac, src_addr)) {
85 return ESP_OK;
86 }
87 }
88
89 espnow_addr_t addr_self = {0};
90 esp_wifi_get_mac(ESP_IF_WIFI_STA, addr_self);
91
92 if (recv_data->sec_ver == ESPNOW_SEC_VER_V1_0
93 && !memcmp(recv_data->client_mac, addr_self, 6)) {
94 ESP_LOGD(TAG, "Device security has been configured by this client, skip.");
95 return ESP_OK;
96 }
97
98 g_info_list = ESP_REALLOC(g_info_list, (g_scan_num + 1) * sizeof(espnow_sec_responder_t));
99 (g_info_list)[g_scan_num].channel = rx_ctrl->channel;
100 (g_info_list)[g_scan_num].rssi = rx_ctrl->rssi;
101 memcpy((g_info_list)[g_scan_num].mac, src_addr, 6);
102 (g_info_list)[g_scan_num].sec_ver = recv_data->sec_ver;
103 g_scan_num++;
104
105 ESP_LOGD(TAG, "Device information:");
106 ESP_LOGD(TAG, "Device channel: %d", rx_ctrl->channel);
107 ESP_LOGD(TAG, "Device rssi: %d", rx_ctrl->rssi);
108 ESP_LOGD(TAG, "Device mac: " MACSTR "", MAC2STR(src_addr));
109 ESP_LOGD(TAG, "Security information:");
110 ESP_LOGD(TAG, "Version: %d", recv_data->sec_ver);
111 ESP_LOGD(TAG, "Client MAC: " MACSTR "", MAC2STR(recv_data->client_mac));
112
113 return ESP_OK;
114}

141esp_err_t espnow_sec_initiator_scan(espnow_sec_responder_t **info_list, size_t *num, TickType_t wait_ticks)
142{
143 esp_err_t ret = ESP_OK;
144 espnow_sec_info_t request_sec_info = {.type = ESPNOW_SEC_TYPE_REQUEST};
145
146 espnow_frame_head_t frame_head = {
147 .retransmit_count = 10,
148 .broadcast = true,
149 .magic = esp_random(),
150 .filter_adjacent_channel = true,
151 .forward_ttl = CONFIG_ESPNOW_SEC_SEND_FORWARD_TTL,
152 .forward_rssi = CONFIG_ESPNOW_SEC_SEND_FORWARD_RSSI,
153 };
154
155 g_scan_num = 0;
156 ESP_FREE(g_info_list);
157
158 espnow_set_config_for_data_type(ESPNOW_DATA_TYPE_SECURITY_STATUS, 1, espnow_sec_initiator_status_process);
159
160 for (int i = 0, start_ticks = xTaskGetTickCount(), recv_ticks = 500; i < 5 && wait_ticks - (xTaskGetTickCount() - start_ticks) > 0;
161 ++i, recv_ticks = pdMS_TO_TICKS(500)) {
162 ret = espnow_send(ESPNOW_DATA_TYPE_SECURITY, ESPNOW_ADDR_BROADCAST, &request_sec_info, 1, &frame_head, portMAX_DELAY);
163 ESP_ERROR_RETURN(ret != ESP_OK, ret, "espnow_send");
164
165 vTaskDelay(recv_ticks);
166 }
167
168 *info_list = g_info_list;
169 *num = g_scan_num;
170
171 espnow_set_config_for_data_type(ESPNOW_DATA_TYPE_SECURITY_STATUS, 0, NULL);
172
173 return ESP_OK;
174}

215 if (g_sec_queue) {

217 sec_data.data = ESP_MALLOC(size);
218 if (!sec_data.data) {
219 return ESP_FAIL;
220 }

225 ESP_LOGW(TAG, "[%s, %d] Send sec queue failed", __func__, __LINE__);
226 ESP_FREE(sec_data.data);
227 return ESP_FAIL;
228 }
229 }

234static esp_err_t protocomm_espnow_initiator_start(const protocomm_security_t *proto_sec, const void *pop,
235 const uint8_t addrs_list[][6], size_t addrs_num, espnow_sec_result_t *res)
236{
237 ESP_PARAM_CHECK(proto_sec);
238 ESP_PARAM_CHECK(addrs_list);
239 ESP_PARAM_CHECK(addrs_num);
240
241 esp_err_t ret = ESP_OK;
242 uint8_t src_addr[6] = {0};
243 espnow_sec_packet_t *req_data = ESP_MALLOC(ESPNOW_DATA_LEN);
244 espnow_sec_packet_t *response_data = ESP_MALLOC(ESPNOW_DATA_LEN);
245 espnow_sec_result_t *result = res;
246 ssize_t response_size = 0;
247 ssize_t outlen = 0;
248 uint8_t *outbuf = NULL;
249 int32_t session_id = 0;
250 espnow_frame_head_t frame_head = {
251 .retransmit_count = CONFIG_ESPNOW_SEC_SEND_RETRY_NUM,
252 .filter_adjacent_channel = true,
253 .forward_ttl = CONFIG_ESPNOW_SEC_SEND_FORWARD_TTL,
254 .forward_rssi = CONFIG_ESPNOW_SEC_SEND_FORWARD_RSSI,
255 };
256 espnow_sec_data_t sec_data = { 0 };
257
258 /* Maximum number of session a time, can be greater but process time will be long*/
259 int32_t MAX_NUM = 100;
260 int recv_ticks = pdMS_TO_TICKS(100);
261 int wait_ticks = 0;
262 int start_ticks = 0;
263 int retry_count = (addrs_num % MAX_NUM == 0) ? (addrs_num / MAX_NUM + 1) : (addrs_num / MAX_NUM + 2);
264 g_sec_initiator_flag = true;
265
266 if (!res) {
267 result = ESP_CALLOC(1, sizeof(espnow_sec_result_t));
268 }
269 result->successed_addr = ESP_CALLOC(addrs_num, ESPNOW_ADDR_LEN);
270 result->unfinished_num = addrs_num;
271 result->unfinished_addr = ESP_CALLOC(result->unfinished_num, ESPNOW_ADDR_LEN);
272 memcpy(result->unfinished_addr, addrs_list, result->unfinished_num * ESPNOW_ADDR_LEN);
273
274 for (int i = 0; i < retry_count && result->unfinished_num > 0 && g_sec_initiator_flag; i++) {
275 size_t current_addrs_num = (result->unfinished_num > MAX_NUM) ? MAX_NUM : result->unfinished_num ;
276 espnow_addr_t *current_addrs_list = ESP_CALLOC(current_addrs_num, ESPNOW_ADDR_LEN);
277 protocomm_security_handle_t *current_session_list = ESP_CALLOC(current_addrs_num, sizeof(protocomm_security_handle_t));
278 memcpy(current_addrs_list, result->unfinished_addr, current_addrs_num * ESPNOW_ADDR_LEN);
279 size_t success_addrs_num = 0;
280 wait_ticks = pdMS_TO_TICKS(1200 + 300 * current_addrs_num);
281
282 ESP_LOGI(TAG, "count: %d, Secure_initiator_send, requested_num: %d, unfinished_num: %d, successed_num: %d",
283 i, current_addrs_num, result->unfinished_num, result->successed_num);
284
285 for (int i = 0; i < current_addrs_num; i++) {
286 proto_sec->init(&current_session_list[i]);
287 proto_sec->new_transport_session(current_session_list[i], i);
288 }
289
290 /**
291 * @brief Send Command 0
292 */
293 espnow_set_group(current_addrs_list, current_addrs_num, ESPNOW_ADDR_GROUP_SEC, NULL, true, portMAX_DELAY);
294
295 ret = write_security1_command0(&outbuf, &outlen);
296 if (ret != ESP_OK || !outbuf || !outlen) {
297 ESP_LOGW(TAG, "espnow-session cm0 prepare failed");
298 ESP_FREE(outbuf);
299 goto exit_init;
300 }
301 frame_head.broadcast = true;
302 frame_head.group = true;
303 response_data->type = ESPNOW_SEC_TYPE_HANDSHAKE;
304 response_data->size = outlen;
305 memcpy(response_data->data, outbuf, outlen);
306 response_size = sizeof(espnow_sec_packet_t) + outlen;
307 ret = espnow_send(ESPNOW_DATA_TYPE_SECURITY, ESPNOW_ADDR_GROUP_SEC, response_data, response_size, &frame_head, portMAX_DELAY);
308 if (ret != ESP_OK) {
309 ESP_LOGW(TAG, "espnow-session cm0 send failed");
310 ESP_FREE(outbuf);
311 goto exit_init;
312 }
313 ESP_FREE(outbuf);
314 espnow_set_group(current_addrs_list, current_addrs_num, ESPNOW_ADDR_GROUP_SEC, NULL, false, portMAX_DELAY);
315
316 /**
317 * @brief Receive Response 0, send Command 1, receive Response 1
318 */
319 frame_head.broadcast = false;
320 frame_head.group = false;
321 start_ticks = xTaskGetTickCount();
322 while(xTaskGetTickCount() - start_ticks < wait_ticks && success_addrs_num < current_addrs_num && g_sec_initiator_flag) {
323 while (g_sec_queue && (xQueueReceive(g_sec_queue, &sec_data, recv_ticks) == pdPASS)) {
324 memcpy(src_addr, sec_data.src_addr, 6);
325 memcpy(req_data, sec_data.data, sec_data.size);
326 ESP_FREE(sec_data.data);
327 session_id = addrs_search(current_addrs_list, current_addrs_num, src_addr);
328 ESP_ERROR_CONTINUE(session_id < 0, "addr " MACSTR " not searched", MAC2STR(src_addr));
329
330 if (req_data->type == ESPNOW_SEC_TYPE_KEY_RESP) {
331 ESP_LOGD(TAG, "Session %d successful, mac "MACSTR"", session_id, MAC2STR(src_addr));
332 addrs_remove(result->unfinished_addr, &result->unfinished_num, src_addr);
333 memcpy(result->successed_addr[result->successed_num], src_addr, ESPNOW_ADDR_LEN);
334 result->successed_num ++;
335 if (++success_addrs_num == current_addrs_num) {
336 break;
337 }
338 } else if (req_data->type == ESPNOW_SEC_TYPE_HANDSHAKE){
339 ret = proto_sec->security_req_handler(current_session_list[session_id], pop, session_id, req_data->data, req_data->size, &outbuf, &outlen, NULL);
340
341 if (ret != ESP_OK) {
342 ESP_LOGE(TAG, "espnow-session handler failed");
343 ESP_FREE(outbuf);
344 continue;
345 } else if (outbuf && outlen) {
346 response_data->type = ESPNOW_SEC_TYPE_HANDSHAKE;
347 response_data->size = outlen;
348 memcpy(response_data->data, outbuf, outlen);
349 response_size = sizeof(espnow_sec_packet_t) + outlen;
350 ESP_FREE(outbuf);
351 } else {
352 /* Send APP key */
353 response_data->type = ESPNOW_SEC_TYPE_KEY;
354 if (proto_sec->encrypt) {
355 outlen = 0;
356
357#if ESP_IDF_VERSION >= ESP_IDF_VERSION_VAL(5, 0, 0)
358 uint8_t *enc_resp = NULL;
359 ret = proto_sec->encrypt(current_session_list[session_id], session_id, app_key, APP_KEY_LEN,
360 &enc_resp, &outlen);
361
362 if (ret != ESP_OK) {
363 ESP_LOGE(TAG, "Encryption of data failed for session id %d", session_id);
364 ESP_FREE(enc_resp);
365 continue;
366 }
367 memcpy(response_data->data, enc_resp, outlen);
368 ESP_FREE(enc_resp);
369#else
370 ret = proto_sec->encrypt(current_session_list[session_id], session_id, app_key, APP_KEY_LEN,
371 response_data->data, &outlen);
372
373 if (ret != ESP_OK) {
374 ESP_LOGE(TAG, "Encryption of data failed for session id %d", session_id);
375 continue;
376 }
377#endif
378 response_data->size = outlen;
379 response_size = sizeof(espnow_sec_packet_t) + outlen;
380 } else {/* will not goto here */
381 response_data->size = APP_KEY_LEN;
382 memcpy(response_data->data, app_key, APP_KEY_LEN);
383 response_size = sizeof(espnow_sec_packet_t) + APP_KEY_LEN;
384 }
385 }
386
387 espnow_add_peer(src_addr, NULL);
388
389 ret = espnow_send(ESPNOW_DATA_TYPE_SECURITY, src_addr, response_data, response_size, &frame_head, portMAX_DELAY);
390 if (ret != ESP_OK) {
391 ESP_LOGW(TAG, "espnow-session send failed");
392 }
393
394 espnow_del_peer(src_addr);
395 }
396 }
397 }
398exit_init:
399
400 for (int i = 0; i < current_addrs_num; i++) {
401 proto_sec->close_transport_session(current_session_list[i], i);
402 proto_sec->cleanup(current_session_list[i]);
403 }
404
405 ESP_FREE(current_addrs_list);
406 ESP_FREE(current_session_list);
407
408 }
409
410 g_sec_initiator_flag = false;
411
412 ESP_FREE(req_data);
413 ESP_FREE(response_data);
414 if (!res) {
415 espnow_sec_initiator_result_free(result);
416 ESP_FREE(result);
417 }
418
419 return ret;
420}

430 espnow_sec_result_t *res)
431{
432 ESP_PARAM_CHECK(key_info);
433 ESP_PARAM_CHECK(pop_data);
434 ESP_PARAM_CHECK(addrs_list);
435 ESP_PARAM_CHECK(addrs_num);

437#if ESP_IDF_VERSION >= ESP_IDF_VERSION_VAL(5, 0, 0)

449
