39esp_err_t espnow_add_group(const espnow_group_t group_id)
40{
41 espnow_group_item_t *item = NULL;
42
43 SLIST_FOREACH(item, &g_group_item_list, next) {
44 if (!memcmp(item->id, group_id, 6)) {
45 return ESP_OK;
46 }
47 }
48
49 espnow_group_item_t *list_first = SLIST_FIRST(&g_group_item_list);
50 item = ESP_MALLOC(sizeof(espnow_group_item_t));
51 memcpy(item->id, group_id, 6);
52
53 if (!list_first) {
54 SLIST_INSERT_HEAD(&g_group_item_list, item, next);
55 } else {
56 SLIST_INSERT_AFTER(list_first, item, next);
57 }
58
59 return ESP_OK;
60}

62esp_err_t espnow_del_group(const espnow_group_t group_id)
63{
64 espnow_group_item_t *item = NULL;
65
66 SLIST_FOREACH(item, &g_group_item_list, next) {
67 if (!memcmp(item->id, group_id, 6)) {
68 SLIST_REMOVE(&g_group_item_list, item, espnow_group_item, next);
69 ESP_FREE(item);
