70 def get_func_section(self, dumps, func):
71 for dump in dumps:
72 for l in dump:
73 if ' %s'%(func) in l and '*UND*' not in l:
74 m = re.match(r'(\S*)\s*([glw])\s*([F|O])\s*(\S*)\s*(\S*)\s*(\S*)\s*', l, re.M|re.I)

155def generator(library_file, object_file, function_file, sdkconfig_file, missing_function_info, objdump='riscv32-esp-elf-objdump'):
156 global espidf_objdump, espidf_missing_function_info

162 lib_paths = paths_c()

164 lib_paths.append(p['library'], '*', p['path'])

166 obj_paths = paths_c()

168 obj_paths.append(p['library'], p['object'], p['path'])

170 libraries = libraries_c()

172 if d['option'] and sdkconfig.check(d['option']) == False:
173 print('skip %s(%s)'%(d['function'], d['option']))

175 lib_path = lib_paths.index(d['library'], '*')
176 obj_path = obj_paths.index(d['library'], d['object'])
177 if not obj_path:
178 obj_path = lib_path
179 libraries.append(d['library'], lib_path[0], d['object'], obj_path, d['function'])
180 return libraries
