35static void show_system_info_timercb(TimerHandle_t timer)
36{
37 uint8_t sta_mac[6] = {0};
38 uint8_t primary = 0;
39 wifi_second_chan_t second = 0;
40 wifi_ap_record_t ap_info = {0};
41
42 esp_wifi_get_mac(ESP_IF_WIFI_STA, sta_mac);
43 esp_wifi_get_channel(&primary, &second);
44 esp_wifi_sta_get_ap_info(&ap_info);
45
46 ESP_LOGI(TAG, "System information sta_mac: " MACSTR ", channel: [%d/%d], rssi: %d, free_heap: %u, minimum_heap: %u",
47 MAC2STR(sta_mac), primary, second, ap_info.rssi, esp_get_free_heap_size(), esp_get_minimum_free_heap_size());
48
49 if (!heap_caps_check_integrity_all(true)) {
50 ESP_LOGE(TAG, "At least one heap is corrupt");
51 }
52}

61uint8_t *espnow_mac_str2hex(const char *mac_str, uint8_t *mac_hex)
62{
63 ESP_ERROR_CHECK(!mac_str);
64 ESP_ERROR_CHECK(!mac_hex);
65
66 uint32_t mac_data[6] = {0};
67
68 int ret = sscanf(mac_str, MACSTR, mac_data, mac_data + 1, mac_data + 2,
69 mac_data + 3, mac_data + 4, mac_data + 5);
70
71 for (int i = 0; i < 6; i++) {
72 mac_hex[i] = mac_data[i];
73 }
74
75 return ret == 6 ? mac_hex : NULL;
76}
