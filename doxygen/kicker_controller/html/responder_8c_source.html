72static EventGroupHandle_t s_wifi_event_group;
73static int s_retry_num = 0;

75#define WIFI_CONNECTED_BIT BIT0
76#define WIFI_FAIL_BIT BIT1
77
78static void app_wifi_event_handler(void *arg, esp_event_base_t event_base,
79 int32_t event_id, void *event_data)
80{
81 if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_STA_DISCONNECTED) {
82 if (s_retry_num < EXAMPLE_ESP_MAXIMUM_RETRY) {
83 esp_wifi_connect();
84 s_retry_num++;
85 ESP_LOGI(TAG, "retry to connect to the AP");
86 } else {
87 xEventGroupSetBits(s_wifi_event_group, WIFI_FAIL_BIT);
88 }
89 ESP_LOGI(TAG, "connect to the AP fail");
90 } else if (event_base == IP_EVENT && event_id == IP_EVENT_STA_GOT_IP) {
91 ip_event_got_ip_t *event = (ip_event_got_ip_t *) event_data;
92 ESP_LOGI(TAG, "got ip:" IPSTR, IP2STR(&event->ip_info.ip));
93 s_retry_num = 0;
94 xEventGroupSetBits(s_wifi_event_group, WIFI_CONNECTED_BIT);
95 }
96}
97#endif

99#if defined(CONFIG_APP_ESPNOW_DEBUG) && defined(CONFIG_APP_POST_LOG_TO_HTTP_SERVER)
100/**
101 * @brief http_client task creation
102 * You can modify the http address according to your needs,
103 * parameters CONFIG_FLASH_LOG_POST_URL if change.and parameters
104 * transport_type if change.
105 */
106static void app_log_read_task(void *arg)
107{
108 esp_err_t ret = ESP_OK;
109 char *log_data = ESP_MALLOC(ESPNOW_DATA_LEN);
110 size_t log_size = 0;
111 esp_http_client_handle_t client = NULL;
112 esp_http_client_config_t config = {
113 .url = CONFIG_APP_FLASH_LOG_POST_URL,
114 .transport_type = HTTP_TRANSPORT_UNKNOWN,
115 };
116
117 client = esp_http_client_init(&config);
118 esp_http_client_set_method(client, HTTP_METHOD_POST);
119
120 for (;;) {
121 vTaskDelay(pdMS_TO_TICKS(1000));
122
123 /* Waiting until either the connection is established (WIFI_CONNECTED_BIT) or connection failed for the maximum
124 * number of re-tries (WIFI_FAIL_BIT). The bits are set by event_handler() (see above) */
125 EventBits_t bits = xEventGroupWaitBits(s_wifi_event_group,
126 WIFI_CONNECTED_BIT | WIFI_FAIL_BIT,
127 pdFALSE,
128 pdFALSE,
129 portMAX_DELAY);
130
131 /* xEventGroupWaitBits() returns the bits before the call returned, hence we can test which event actually
132 * happened. */
133 if (bits & WIFI_CONNECTED_BIT) {
134 ESP_LOGD(TAG, "connected to ap");
135 } else if (bits & WIFI_FAIL_BIT) {
136 ESP_LOGD(TAG, "Failed to connect to ap");
137 continue;
138 } else {
139 ESP_LOGE(TAG, "UNEXPECTED EVENT");
140 continue;
141 }
142
143 log_size = espnow_log_flash_size();
144 ESP_ERROR_CONTINUE(!log_size, "");
145
146 ret = esp_http_client_open(client, log_size);
147 if (ret != ESP_OK) {
148 ESP_LOGE(TAG, "Failed to open HTTP connection: %s", esp_err_to_name(ret));
149 esp_http_client_close(client);
150 continue;
151 }
152
153 /**
154 * @brief Read the log data and transfer the data out via http server
155 */
156 for (size_t size = MIN(ESPNOW_DATA_LEN, log_size);
157 size > 0 && espnow_log_flash_read(log_data, &size) == ESP_OK;
158 log_size -= size, size = MIN(ESPNOW_DATA_LEN, log_size)) {
159 ret = esp_http_client_write(client, log_data, size);
160 ESP_ERROR_BREAK(ret < 0, "<%s> Failed to write HTTP data", esp_err_to_name(ret));
161 }
162
163 esp_http_client_close(client);
164 }
165
166 ESP_FREE(log_data);
167 esp_http_client_cleanup(client);
168 vTaskDelete(NULL);
169}
170#endif

194static void app_espnow_prov_initiator_init(void *arg)
195{
196 esp_err_t ret = ESP_OK;
197 wifi_pkt_rx_ctrl_t rx_ctrl = {0};
198 espnow_prov_initiator_t initiator_info = {
199 .product_id = "initiator_test",
200 };
201 espnow_addr_t responder_addr = {0};
202 espnow_prov_responder_t responder_info = {0};
203
204 for (;;) {
205#ifdef CONFIG_APP_ESPNOW_SECURITY
206 uint8_t key_info[APP_KEY_LEN];
207
208 /* Security key is not ready */
209 if (espnow_get_key(key_info) != ESP_OK) {
210 vTaskDelay(pdMS_TO_TICKS(1000));
211 continue;
212 }
213#endif
214
215 ret = espnow_prov_initiator_scan(responder_addr, &responder_info, &rx_ctrl, pdMS_TO_TICKS(3 * 1000));
216 ESP_ERROR_CONTINUE(ret != ESP_OK, "");
217
218 ESP_LOGI(TAG, "MAC: "MACSTR", Channel: %d, RSSI: %d, Product_id: %s, Device Name: %s",
219 MAC2STR(responder_addr), rx_ctrl.channel, rx_ctrl.rssi,
220 responder_info.product_id, responder_info.device_name);
221
222 ret = espnow_prov_initiator_send(responder_addr, &initiator_info, app_espnow_prov_initiator_recv_cb, pdMS_TO_TICKS(3 * 1000));
223 ESP_ERROR_CONTINUE(ret != ESP_OK, "<%s> espnow_prov_responder_add", esp_err_to_name(ret));
224
225 break;
226 }
227
228 ESP_LOGI(TAG, "provisioning initiator exit");
229 vTaskDelete(NULL);
230 s_prov_task = NULL;
231}
