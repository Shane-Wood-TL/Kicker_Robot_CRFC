42/* Set HTTP response content type according to file extension */
43static esp_err_t set_content_type_from_file(httpd_req_t *req, const char *filepath)
44{
45 const char *type = "text/plain";
46
47 if (CHECK_FILE_EXTENSION(filepath, ".html")) {
48 type = "text/html";
49 } else if (CHECK_FILE_EXTENSION(filepath, ".js")) {
50 type = "application/javascript";
51 } else if (CHECK_FILE_EXTENSION(filepath, ".css")) {
52 type = "text/css";
53 } else if (CHECK_FILE_EXTENSION(filepath, ".png")) {
54 type = "image/png";
55 } else if (CHECK_FILE_EXTENSION(filepath, ".ico")) {
56 type = "image/x-icon";
57 } else if (CHECK_FILE_EXTENSION(filepath, ".svg")) {
58 type = "text/xml";
59 }
60
61 return httpd_resp_set_type(req, type);
62}

64/* Send HTTP response with the contents of the requested file */
65static esp_err_t rest_common_get_handler(httpd_req_t *req)
66{
67 ESP_LOGD(TAG, "[%s, %d]: uri: %s", __func__, __LINE__, req->uri);
68
69 char filepath[FILE_PATH_MAX];
70 rest_server_context_t *rest_context = (rest_server_context_t *)req->user_ctx;
71
72 strlcpy(filepath, rest_context->base_path, sizeof(filepath));
73 strlcat(filepath, req->uri, sizeof(filepath));
74
75 int fd = open(filepath, O_RDONLY, 0);
76
77 if (fd == -1) {
78 ESP_LOGD(TAG, "[%s, %d] filepath: %s", __func__, __LINE__, filepath);
79
80 const char *scratch = "index.html";
81 sprintf(filepath, "%s/%s", rest_context->base_path, scratch);
82
83 fd = open(filepath, O_RDONLY, 0);
84 }
85
86 if (fd == -1) {
87 ESP_LOGE(TAG, "Failed to open file : %s", filepath);
88 /* Respond with 500 Internal Server Error */
89 httpd_resp_send_err(req, HTTPD_500_INTERNAL_SERVER_ERROR, "Failed to read existing file");
90 return ESP_FAIL;
91 }
92
93 set_content_type_from_file(req, filepath);
94
95 char *chunk = rest_context->scratch;
96 ssize_t read_bytes;
97
98 do {
99 /* Read file in chunks into the scratch buffer */
100 read_bytes = read(fd, chunk, SCRATCH_BUFSIZE);
101
102 if (read_bytes == -1) {
103 ESP_LOGE(TAG, "Failed to read file : %s", filepath);
104 } else if (read_bytes > 0) {
105 /* Send the buffer contents as HTTP response chunk */
106 if (httpd_resp_send_chunk(req, chunk, read_bytes) != ESP_OK) {
107 close(fd);
108 ESP_LOGE(TAG, "File sending failed!");
109 /* Abort sending file */
110 httpd_resp_sendstr_chunk(req, NULL);
111 /* Respond with 500 Internal Server Error */
112 httpd_resp_send_err(req, HTTPD_500_INTERNAL_SERVER_ERROR, "Failed to send file");
113 return ESP_FAIL;
114 }
115 }
116 } while (read_bytes > 0);
117
118 /* Close file after sending complete */
119 close(fd);
120
121 ESP_LOGD(TAG, "File sending complete");
122 /* Respond with an empty chunk to signal HTTP response completion */
123 httpd_resp_send_chunk(req, NULL, 0);
124 return ESP_OK;
125}

154 uint32_t timestamp = 0;
155 char log_level[2] = {0};
156 char tag[32] = {0};
157
158 ret = sscanf((char *)data + ((data[0] == '\033') ? 7 : 0), "%1s (%d) %[^:]: %[^\r\n\033]%*c",

160 log_size = asprintf(&log_data, "{\"errno\":0,\"src_addr\":\""MACSTR"\",\"rssi\":%d,\"timetamp\":%u,\"log_level\":\"%s\",\"tag\":\"%s\",\"data\":\"%.*s\"}",
161 MAC2STR(addr), rx_ctrl->rssi, timestamp, log_level, tag, strlen(buffer), buffer);
162
163 ESP_LOGD(TAG, "ret: %d, %s", ret, log_data);
164
165 httpd_ws_frame_t ws_pkt = {
166 .payload = (uint8_t *)log_data,
167 .len = log_size,
168 .type = HTTPD_WS_TYPE_TEXT,
169 };
170
171 ret = httpd_ws_send_frame_async(g_resp_handle->hd, g_resp_handle->fd, &ws_pkt);
172
173 ESP_FREE(log_data);
174 ESP_FREE(buffer);
175
176 if (ret != ESP_OK) {
177 ESP_LOGW(TAG, "<%s> httpd_ws_send_frame_async, err_str: %s", esp_err_to_name(ret), strerror(errno));
178 close(g_resp_handle->fd);
179 ESP_FREE(g_resp_handle);
180 return ret;
181 }
182
183 return ESP_OK;

230/* Simple handler for light brightness control */
231static esp_err_t debug_send_handler(httpd_req_t *req)
232{
233 esp_err_t ret = ESP_OK;
234 int total_len = req->content_len;
235 int cur_len = 0;
236 char *buf = ((rest_server_context_t *)(req->user_ctx))->scratch;
237 int received = 0;
238
239 if (total_len >= SCRATCH_BUFSIZE) {
240 /* Respond with 500 Internal Server Error */
241 httpd_resp_send_err(req, HTTPD_500_INTERNAL_SERVER_ERROR, "content too long");
242 return ESP_FAIL;
243 }
244
245 while (cur_len < total_len) {
246 received = httpd_req_recv(req, buf + cur_len, total_len);
247
248 if (received <= 0) {
249 /* Respond with 500 Internal Server Error */
250 httpd_resp_send_err(req, HTTPD_500_INTERNAL_SERVER_ERROR, "Failed to post control value");
251 return ESP_FAIL;
252 }
253
254 cur_len += received;
255 }
256
257 buf[total_len] = '\0';
258
259 ESP_LOGI(TAG, "command: %s", buf);
260
261 esp_err_t err = esp_console_run(buf, &ret);
262 char *resp_data = NULL;
263
264 if (err == ESP_ERR_NOT_FOUND) {
265 ESP_LOGE(TAG, "Unrecognized command");
266 asprintf(&resp_data, "{\"errno\":%d, \"err_msg\":\"Unrecognized command\"}", err);
267 } else if (err == ESP_ERR_INVALID_ARG) {
268 /**< Command was empty */
269 asprintf(&resp_data, "{\"errno\":%d, \"err_msg\":\"Command was empty\"}", err);
270 } else if (err == ESP_OK && ret != ESP_OK) {
271 asprintf(&resp_data, "{\"errno\":%d, \"err_msg\":\"Command returned non-zero error code: %s\"}", ret, esp_err_to_name(ret));
272 } else if (err != ESP_OK) {
273 asprintf(&resp_data, "{\"errno\":%d, \"err_msg\":\"Internal error: %s\"}", err, esp_err_to_name(err));
274 } else {
275 asprintf(&resp_data, "{\"errno\":%d, \"err_msg\":\"success\"}", ESP_OK);
276 }
277
278 ESP_LOGI(TAG, "resp_data: %s", resp_data);
279
280 httpd_resp_sendstr(req, resp_data);
281 ESP_FREE(resp_data);
282
283 return ESP_OK;
284}

303static void ota_send_task(void *arg)
304{
305 esp_err_t ret = ESP_OK;
306 size_t addrs_num = 0;
307 espnow_ota_result_t ota_result = {0};
308 uint32_t start_time = xTaskGetTickCount();
309 uint8_t (* addrs_list)[ESPNOW_ADDR_LEN] = ESP_MALLOC(ESPNOW_ADDR_LEN);
310
311 for (const char *tmp = (char *)arg;; tmp++) {
312 if (*tmp == ',' || *tmp == ' ' || *tmp == '|' || *tmp == '.' || *tmp == '\0') {
313 espnow_mac_str2hex(tmp - 17, addrs_list[addrs_num]);
314 addrs_num++;
315
316 if (*tmp == '\0' || *(tmp + 1) == '\0') {
317 break;
318 }
319
320 addrs_list = ESP_REALLOC(addrs_list, ESPNOW_ADDR_LEN * (addrs_num + 1));
321 }
322 }
323
324 uint8_t sha_256[32] = {0};
325 const esp_partition_t *data_partition = esp_ota_get_next_update_partition(NULL);
326 ret = esp_partition_get_sha256(data_partition, sha_256);
327 ESP_ERROR_GOTO(ret!= ESP_OK, EXIT, "<%s> esp_partition_get_sha256", esp_err_to_name(ret));
328
329 ret = espnow_ota_initiator_send(addrs_list, addrs_num, sha_256, g_ota_size,
330 ota_initiator_data_cb, &ota_result);
331 ESP_ERROR_GOTO(ret!= ESP_OK, EXIT, "<%s> esp_partition_get_sha256", esp_err_to_name(ret));
332
333 ESP_LOGI(TAG, "Firmware is sent to the device to complete, Spend time: %ds",
334 (xTaskGetTickCount() - start_time) * portTICK_PERIOD_MS / 1000);
335 ESP_LOGI(TAG, "Devices upgrade completed, successed_num: %d, unfinished_num: %d",
336 ota_result.successed_num, ota_result.unfinished_num);
337
338 ESP_FREE(addrs_list);
339 espnow_ota_initiator_result_free(&ota_result);
340
341EXIT:
342 ESP_FREE(arg);
343
344 vTaskDelete(NULL);
345}

347static esp_err_t ota_data_handler(httpd_req_t *req)
348{
349 esp_err_t ret = ESP_OK;
350 int total_size = req->content_len;
351 char *resp_data = NULL;
352 uint32_t start_time = xTaskGetTickCount();
353 esp_ota_handle_t ota_handle = 0;
354 uint8_t *data = ESP_MALLOC(OTA_DATA_PAYLOAD_LEN);
355
356 const esp_partition_t *updata_partition = esp_ota_get_next_update_partition(NULL);
357 /**< Commence an OTA update writing to the specified partition. */
358 ret = esp_ota_begin(updata_partition, total_size, &ota_handle);
359 ESP_ERROR_GOTO(ret != ESP_OK, EXIT, "<%s> esp_ota_begin failed, total_size", esp_err_to_name(ret));
360
361 if (total_size >= ESP_OTA_DATA_LEN) {
362 /* Respond with 500 Internal Server Error */
363 httpd_resp_send_err(req, HTTPD_500_INTERNAL_SERVER_ERROR, "content too long");
364 return ESP_FAIL;
365 }
366
367 for (ssize_t size = 0, recv_size = 0, i = 0; recv_size < total_size; ++i) {
368 size = httpd_req_recv(req, (char *)data, OTA_DATA_PAYLOAD_LEN);
369
370 if (size > 0) {
371 /**< Write OTA update data to partition */
372 ret = esp_ota_write(ota_handle, data, size);
373 ESP_ERROR_GOTO(ret != ESP_OK, EXIT, "<%s> Write firmware to flash, size: %d, data: %.*s",
374 esp_err_to_name(ret), size, size, data);
375 } else {
376 ESP_LOGW(TAG, "<%s> esp_http_client_read, str_errno: %s", esp_err_to_name(ret), strerror(errno));
377 ret = ESP_FAIL;
378 goto EXIT;
379 }
380
381 recv_size += size;
382
383 if (i % 100 == 0 || recv_size == total_size) {
384 ESP_LOGI(TAG, "Firmware download size: %d, progress rate: %d%%",
385 recv_size, recv_size * 100 / total_size);
386 }
387 }
388
389 ESP_LOGI(TAG, "The service download firmware is complete, total_size: %d, Spend time: %ds",
390 total_size, pdTICKS_TO_MS(xTaskGetTickCount() - start_time) / 1000);
391
392 ret = esp_ota_end(ota_handle);
393 ESP_ERROR_GOTO(ret != ESP_OK, EXIT, "<%s> esp_ota_end", esp_err_to_name(ret));
394
395 size_t size = httpd_req_get_hdr_value_len(req, "Device-List") + 1;
396 ESP_ERROR_GOTO(size <= 0, EXIT, "Search for %s in request headers", "Device-List");
397
398 char *addrs_list = ESP_CALLOC(1, size + 1);
399
400 if (httpd_req_get_hdr_value_str(req, "Device-List", addrs_list, size) != ESP_OK) {
401 ESP_LOGD(TAG, "Get the value string of %s from the request headers", "Device-List");
402 ESP_FREE(addrs_list);
403 goto EXIT;
404 }
405
406 g_ota_size = total_size;
407
408 xTaskCreate(ota_send_task, "ota_send", 4 * 1024, addrs_list, tskIDLE_PRIORITY + 1, NULL);
409
410EXIT:
411 if (ret != ESP_OK) {
412 asprintf(&resp_data, "{\"errno\":%d, \"err_msg\":\"Internal error: %s\"}", ret, esp_err_to_name(ret));
413 } else {
414 asprintf(&resp_data, "{\"errno\":%d, \"err_msg\":\"success\"}", ESP_OK);
415 }
416
417 ESP_LOGI(TAG, "resp_data: %s", resp_data);
418
419 httpd_resp_sendstr(req, resp_data);
420 ESP_FREE(resp_data);
421 ESP_FREE(data);
422
423 return ESP_OK;
424}
