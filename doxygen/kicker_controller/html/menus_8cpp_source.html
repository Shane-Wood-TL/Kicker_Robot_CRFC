165template<typename T>
166void menu<T>::change_selected(int8_t direction) {
167 if (direction != 1 && direction != -1 && direction != 0) {
168 return;
169 }
170
171 if (!has_selectable_values) {
172 return;
173 }
174 currently_selected -= direction;
175 if (currently_selected >= lines) {
176 currently_selected = 0;
177 } else if (currently_selected < 0) {
178 currently_selected = lines - 1;
179 }
180 while (selectable_values[currently_selected] == false) {
181 currently_selected -= direction;
182 if (currently_selected >= lines) {
183 currently_selected = 0;
184 } else if (currently_selected < 0) {
185 currently_selected = lines - 1;
186 }
187 }
188 //draw an X denoting currently selected
189 display->write_string_SSD1306("X", 0, font_rows[currently_selected]);
190}
