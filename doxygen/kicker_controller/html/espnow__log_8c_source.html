65static void espnow_log_send_task(void *arg)
66{
67 log_info_t *log_info = NULL;
68 espnow_frame_head_t frame_head = ESPNOW_FRAME_CONFIG_DEFAULT();
69 frame_head.security = CONFIG_ESPNOW_DEBUG_SECURITY;
70
71 for (; g_log_config;) {
72 if (xQueueReceive(g_log_queue, &log_info, pdMS_TO_TICKS(DEBUG_LOG_TIMEOUT_MS)) != pdPASS) {
73 continue;
74 }
75
76 if (log_info->level <= g_log_config->log_level_flash) {
77 espnow_log_flash_write(log_info->data, log_info->size, log_info->level); /**< Write log data to flash */
78 }
79
80 if (strcasecmp(log_info->tag, "espnow") && log_info->level <= g_log_config->log_level_espnow) {
81 log_info->size = MIN(ESPNOW_DATA_LEN - 1, log_info->size) + 1;
82 log_info->data[log_info->size - 1] = '\0';
83 espnow_send(ESPNOW_DATA_TYPE_DEBUG_LOG, ESPNOW_ADDR_BROADCAST, log_info->data, log_info->size, &frame_head, portMAX_DELAY);
84 }
85
86 if (log_info->level <= g_log_config->log_level_custom) {
87 if (g_log_config->log_custom_write) {
88 g_log_config->log_custom_write(log_info->data, log_info->size, log_info->tag, log_info->level); /**< Write log data to custom */
89 }
90 }
91
92 DEBUG_LOG_FREE(log_info->data);
93 DEBUG_LOG_FREE(log_info);
94 }
95
96 vTaskDelete(NULL);
97}

104void __wrap_esp_log_writev(esp_log_level_t level,
105 const char *tag,
106 const char *format,
107 va_list args)
108{
109 if (!g_log_config || level <= g_log_config->log_level_uart) {
110 __real_esp_log_writev(level, tag, format, args);
111 }
112
113 if (!g_log_config || (level > g_log_config->log_level_espnow
114 && level > g_log_config->log_level_flash
115 && level > g_log_config->log_level_custom)) {
116 return;
117 }
118
119 log_info_t *log_info = DEBUG_LOG_MALLOC(sizeof(log_info_t));
120
121 log_info->size = vasprintf(&log_info->data, format, args);
122 log_info->level = level;
123 log_info->tag = tag;
124
125 if (xQueueSend(g_log_queue, &log_info, 0) == pdFALSE) {
126 DEBUG_LOG_FREE(log_info->data);
127 DEBUG_LOG_FREE(log_info);
128 }
129}

142esp_err_t espnow_log_init(const espnow_log_config_t *config)
143{
144 if (g_log_init_flag) {
145 return ESP_FAIL;
146 }
147
148 esp_err_t ret = ESP_OK;
149
150 g_log_queue = xQueueCreate(DEBUG_LOG_QUEUE_SIZE, sizeof(log_info_t *));
151 ESP_ERROR_RETURN(!g_log_queue, ESP_FAIL, "g_log_queue create fail");
152
153 g_log_config = ESP_CALLOC(1, sizeof(espnow_log_config_t));
154 ESP_ERROR_RETURN(!g_log_config, ESP_ERR_NO_MEM, "");
155 memcpy(g_log_config, config, sizeof(espnow_log_config_t));
156
157 if (config->log_level_flash != ESP_LOG_NONE) {
158 ret = espnow_log_flash_init();
159 ESP_ERROR_RETURN(ret != ESP_OK, ret, "espnow_log_flash_init");
160 }
161
162 xTaskCreatePinnedToCore(espnow_log_send_task, "espnow_log_send", 3 * 1024,
163 NULL, CONFIG_DEBUG_TASK_DEFAULT_PRIOTY,
164 NULL, CONFIG_DEBUG_TASK_PINNED_TO_CORE);
165
166 ESP_LOGI(TAG, "log initialized successfully");
167
168 g_log_init_flag = true;
169
170 return ESP_OK;
171}

173esp_err_t espnow_log_deinit()
174{
175 if (g_log_init_flag) {
176 return ESP_FAIL;
177 }
178
179 log_info_t *log_data = NULL;
180
181 while (xQueueReceive(g_log_queue, &log_data, 0)) {
182 DEBUG_LOG_FREE(log_data);
183 DEBUG_LOG_FREE(log_data->data);
184 }
185
186 espnow_log_flash_deinit();
187
188 g_log_init_flag = false;
189 DEBUG_LOG_FREE(g_log_config);
190
191 return ESP_OK;
192}
