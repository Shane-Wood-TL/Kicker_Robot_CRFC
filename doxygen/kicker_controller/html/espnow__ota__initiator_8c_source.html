63static bool addrs_remove(uint8_t addrs_list[][ESPNOW_ADDR_LEN],
64 size_t *addrs_num, const uint8_t addr[6])
65{
66 if (!addrs_list || !addrs_num || !addr) {
67 ESP_LOGE(TAG, "!addrs_list: %p !addrs_num: %p !addr: %p", addrs_list, addrs_num, addr);
68 return false;
69 }
70
71 for (int i = 0; i < *addrs_num; i++) {
72 if (ESPNOW_ADDR_IS_EQUAL(addrs_list[i], addr)) {
73 if (--(*addrs_num)) {
74 memcpy(addrs_list[i], addrs_list[*addrs_num], ESPNOW_ADDR_LEN);
75 }
76
77 return true;
78 }
79 }
80
81 return false;
82}

84static esp_err_t espnow_ota_info(uint8_t *src_addr, void *data,
85 size_t size, wifi_pkt_rx_ctrl_t *rx_ctrl)
86{
87 espnow_ota_info_t *recv_data = (espnow_ota_info_t *)data;
88
89 for (int i = 0; i < g_scan_num; ++i) {
90 if (ESPNOW_ADDR_IS_EQUAL((g_info_list)[i].mac, src_addr)) {
91 return ESP_OK;
92 }
93 }
94
95 g_info_list = ESP_REALLOC(g_info_list, (g_scan_num + 1) * sizeof(espnow_ota_responder_t));
96 (g_info_list)[g_scan_num].channel = rx_ctrl->channel;
97 (g_info_list)[g_scan_num].rssi = rx_ctrl->rssi;
98 memcpy((g_info_list)[g_scan_num].mac, src_addr, 6);
99 memcpy(&((g_info_list)[g_scan_num].app_desc), &recv_data->app_desc, sizeof(esp_app_desc_t));
100 g_scan_num++;
101
102 ESP_LOGV(TAG, "Application information:");
103 ESP_LOGV(TAG, "Project name: %s", recv_data->app_desc.project_name);
104 ESP_LOGV(TAG, "App version: %s", recv_data->app_desc.version);
105 ESP_LOGV(TAG, "Secure version: %d", recv_data->app_desc.secure_version);
106 ESP_LOGV(TAG, "Compile time: %s %s", recv_data->app_desc.date, recv_data->app_desc.time);
107 ESP_LOGV(TAG, "ESP-IDF: %s", recv_data->app_desc.idf_ver);
108
109 return ESP_OK;
110}

119static esp_err_t espnow_ota_status_handle(uint8_t *src_addr, void *data, size_t size)
120{
121 ESP_PARAM_CHECK(src_addr);
122 ESP_PARAM_CHECK(data);
123 ESP_PARAM_CHECK(size);
124
125 if (g_ota_queue) {
126 espnow_ota_data_t ota_data = { 0 };
127 ota_data.data = ESP_MALLOC(size);
128 if (!ota_data.data) {
129 return ESP_FAIL;
130 }
131 memcpy(ota_data.data, data, size);
132 ota_data.size = size;
133 memcpy(ota_data.src_addr, src_addr, 6);
134 if (xQueueSend(g_ota_queue, &ota_data, 0) != pdPASS) {
135 ESP_LOGW(TAG, "[%s, %d] Send ota queue failed", __func__, __LINE__);
136 ESP_FREE(ota_data.data);
137 return ESP_FAIL;
138 }
139 }
140
141 return ESP_OK;
142}

144static esp_err_t espnow_ota_initiator_status_process(uint8_t *src_addr, void *data,
145 size_t size, wifi_pkt_rx_ctrl_t *rx_ctrl)
146{
147 ESP_PARAM_CHECK(src_addr);
148 ESP_PARAM_CHECK(data);
149 ESP_PARAM_CHECK(size);
150 ESP_PARAM_CHECK(rx_ctrl);
151
152 esp_err_t ret = ESP_OK;
153 uint8_t data_type = ((uint8_t *)data)[0];
154
155 switch (data_type) {
156 case ESPNOW_OTA_TYPE_INFO:
157 ESP_LOGD(TAG, "ESPNOW_OTA_TYPE_INFO");
158 if (g_info_en){
159 ret = espnow_ota_info(src_addr, data, size, rx_ctrl);
160 }
161 break;
162
163 case ESPNOW_OTA_TYPE_STATUS:
164 ESP_LOGD(TAG, "ESPNOW_OTA_TYPE_STATUS");
165 ret = espnow_ota_status_handle(src_addr, data, size);
166 break;
167
168 default:
169 break;
170 }
171
172 ESP_ERROR_RETURN(ret != ESP_OK, ret, "espnow_ota_handle");
173 return ret;
174}

176esp_err_t espnow_ota_initiator_scan(espnow_ota_responder_t **info_list, size_t *num, TickType_t wait_ticks)
177{
178 esp_err_t ret = ESP_OK;
179 espnow_ota_info_t request_ota_info = {.type = ESPNOW_OTA_TYPE_REQUEST};
180
181 espnow_frame_head_t frame_head = {
182 .retransmit_count = CONFIG_ESPNOW_OTA_RETRANSMISSION_TIMES,
183 .broadcast = true,
184 .magic = esp_random(),
185 .filter_adjacent_channel = true,
186 .forward_ttl = CONFIG_ESPNOW_OTA_SEND_FORWARD_TTL,
187 .forward_rssi = CONFIG_ESPNOW_OTA_SEND_FORWARD_RSSI,
188 .security = CONFIG_ESPNOW_OTA_SECURITY,
189 };
190
191 espnow_ota_initiator_scan_result_free();
192
193 g_info_en = true;
194 espnow_set_config_for_data_type(ESPNOW_DATA_TYPE_OTA_STATUS, 1, espnow_ota_initiator_status_process);
195
196 for (int i = 0, start_ticks = xTaskGetTickCount(), recv_ticks = 500; i < 5 && wait_ticks - (xTaskGetTickCount() - start_ticks) > 0;
197 ++i, recv_ticks = 500) {
198 ret = espnow_send(ESPNOW_DATA_TYPE_OTA_DATA, ESPNOW_ADDR_BROADCAST, &request_ota_info, 1, &frame_head, portMAX_DELAY);
199 ESP_ERROR_GOTO(ret != ESP_OK, EXIT, "espnow_send");
200
201 vTaskDelay(recv_ticks);
202 }
203
204 *info_list = g_info_list;
205 *num = g_scan_num;
206
207EXIT:
208 espnow_set_config_for_data_type(ESPNOW_DATA_TYPE_OTA_STATUS, 0, NULL);
209 g_info_en = false;
210
211 return ret;
212}

222static esp_err_t espnow_ota_request_status(uint8_t (*progress_array)[ESPNOW_OTA_PROGRESS_MAX_SIZE],
223 const espnow_ota_status_t *status, espnow_ota_result_t *result)
224{
225 esp_err_t ret = ESP_OK;
226 uint8_t src_addr[6] = {0};
227 espnow_ota_status_t *response_data = ESP_MALLOC(ESPNOW_DATA_LEN);
228 espnow_ota_data_t ota_data = { 0 };
229
230 result->requested_num = 0;
231 ESP_FREE(result->requested_addr);
232
233 /**
234 * @brief Remove the device that the firmware upgrade has completed.
235 */
236 while (g_ota_queue && (xQueueReceive(g_ota_queue, &ota_data, 0) == pdPASS)) {
237 memcpy(src_addr, ota_data.src_addr, 6);
238 memcpy(response_data, ota_data.data, ota_data.size);
239 ESP_FREE(ota_data.data);
240 if (response_data->written_size == response_data->total_size || response_data->error_code == ESP_ERR_ESPNOW_OTA_FINISH) {
241 if (!addrs_remove(result->unfinished_addr, &result->unfinished_num, src_addr)) {
242 ESP_LOGW(TAG, "The device has been removed from the list waiting for the upgrade");
243 continue;
244 }
245
246 result->successed_num++;
247 result->successed_addr = ESP_REALLOC_RETRY(result->successed_addr,
248 result->successed_num * ESPNOW_ADDR_LEN);
249 memcpy(result->successed_addr + (result->successed_num - 1), src_addr, ESPNOW_ADDR_LEN);
250 espnow_set_group((uint8_t (*)[6])src_addr, 1, ESPNOW_ADDR_GROUP_OTA, NULL, false, portMAX_DELAY);
251 } else if (response_data->error_code == ESP_ERR_ESPNOW_OTA_STOP) {
252 addrs_remove(result->unfinished_addr, &result->unfinished_num, src_addr);
253 espnow_set_group((uint8_t (*)[6])src_addr, 1, ESPNOW_ADDR_GROUP_OTA, NULL, false, portMAX_DELAY);
254 }
255
256 if (result->unfinished_num == 0) {
257 return ESP_OK;
258 }
259 }
260
261 size_t response_num = result->unfinished_num;
262 uint8_t (*response_addrs)[6] = ESP_REALLOC_RETRY(NULL, result->unfinished_num * ESPNOW_ADDR_LEN);
263 memcpy(response_addrs, result->unfinished_addr, ESPNOW_ADDR_LEN * response_num);
264
265 /**
266 * @brief Request all devices upgrade status from unfinished device.
267 */
268 espnow_frame_head_t status_frame = {
269 .group = true,
270 .broadcast = true,
271 .retransmit_count = CONFIG_ESPNOW_OTA_RETRANSMISSION_TIMES,
272 .magic = esp_random(),
273 .filter_adjacent_channel = true,
274 .forward_ttl = CONFIG_ESPNOW_OTA_SEND_FORWARD_TTL,
275 .forward_rssi = CONFIG_ESPNOW_OTA_SEND_FORWARD_RSSI,
276 .security = CONFIG_ESPNOW_OTA_SECURITY,
277 };
278
279 for (int i = 0, wait_ticks = pdMS_TO_TICKS(1000); i < 3 && response_num > 0; ++i, wait_ticks = pdMS_TO_TICKS(500)) {
280 if (espnow_send(ESPNOW_DATA_TYPE_OTA_DATA, ESPNOW_ADDR_GROUP_OTA, status,
281 sizeof(espnow_ota_status_t), &status_frame, portMAX_DELAY) != ESP_OK) {
282 ESP_LOGW(TAG, "Request devices upgrade status");
283 }
284
285 uint8_t mac_ota_wait[6] = {0};
286
287 while (response_num > 0 && g_ota_queue) {
288 ret = xQueueReceive(g_ota_queue, &ota_data, wait_ticks);
289 ESP_ERROR_BREAK(ret != pdPASS, "<%s> wait_ticks: %d", esp_err_to_name(ret), wait_ticks);
290 memcpy(src_addr, ota_data.src_addr, 6);
291 memcpy(response_data, ota_data.data, ota_data.size);
292 ESP_FREE(ota_data.data);
293 ret = response_data->error_code;
294
295 if (response_data->error_code == ESP_ERR_ESPNOW_OTA_FIRMWARE_NOT_INIT) {
296 wait_ticks = pdMS_TO_TICKS(CONFIG_ESPNOW_OTA_WAIT_RESPONSE_TIMEOUT);
297 memcpy(mac_ota_wait, src_addr, 6);
298 continue;
299 }
300
301 if (response_data->error_code == ESP_ERR_ESPNOW_OTA_STOP
302 || response_data->error_code == ESP_ERR_ESPNOW_OTA_FINISH) {
303 ESP_LOGW(TAG, "<ESP_ERR_ESPNOW_OTA_FIRMWARE_PARTITION> response_data->error_code: ");
304 addrs_remove(result->unfinished_addr, &result->unfinished_num, src_addr);
305 addrs_remove(response_addrs, &response_num, src_addr);
306 espnow_set_group((uint8_t (*)[6])src_addr, 1, ESPNOW_ADDR_GROUP_OTA, NULL, false, portMAX_DELAY);
307 continue;
308 }
309
310 ESP_LOGD(TAG, "Response, src_addr: " MACSTR ", response_num: %d, total_size: %d, written_size: %d, error_code: %s",
311 MAC2STR(src_addr), response_num, response_data->total_size,
312 response_data->written_size, esp_err_to_name(response_data->error_code));
313
314 if (wait_ticks == pdMS_TO_TICKS(CONFIG_ESPNOW_OTA_WAIT_RESPONSE_TIMEOUT)) {
315 if (!memcmp(src_addr, mac_ota_wait, 6)) {
316 wait_ticks = pdMS_TO_TICKS(100);
317 }
318 }
319
320 /**< Remove the device that upgrade status has been received */
321 if (!addrs_remove(response_addrs, &response_num, src_addr)) {
322 continue;
323 }
324
325 if (response_data->written_size == response_data->total_size) {
326 if (!addrs_remove(result->unfinished_addr, &result->unfinished_num, src_addr)) {
327 ESP_LOGW(TAG, "The device has been removed from the list waiting for the upgrade");
328 continue;
329 }
330
331 result->successed_num++;
332 result->successed_addr = ESP_REALLOC_RETRY(result->successed_addr,
333 result->successed_num * ESPNOW_ADDR_LEN);
334 memcpy(result->successed_addr + (result->successed_num - 1), src_addr, ESPNOW_ADDR_LEN);
335
336 espnow_set_group((uint8_t (*)[6])src_addr, 1, ESPNOW_ADDR_GROUP_OTA, NULL, false, portMAX_DELAY);
337 } else {
338 ESP_LOG_BUFFER_HEXDUMP(TAG, response_data->progress_array[0],
339 sizeof(espnow_ota_status_t) + ESPNOW_OTA_PROGRESS_MAX_SIZE, ESP_LOG_VERBOSE);
340
341 result->requested_num++;
342 result->requested_addr = ESP_REALLOC_RETRY(result->requested_addr, result->requested_num * ESPNOW_ADDR_LEN);
343 memcpy(result->requested_addr + (result->requested_num - 1), src_addr, ESPNOW_ADDR_LEN);
344
345 if (response_data->written_size == 0) {
346 memset(progress_array, 0x0, status->packet_num / 8 + 1);
347 } else {
348 for (int i = 0; i < ESPNOW_OTA_PROGRESS_MAX_SIZE
349 && (response_data->progress_index * ESPNOW_OTA_PROGRESS_MAX_SIZE + i) * 8 < status->packet_num; i++) {
350 progress_array[response_data->progress_index][i] &= response_data->progress_array[0][i];
351 }
352 }
353 }
354 }
355 }
356
357 ret = ESP_OK;
358
359 if (response_num && response_num == result->unfinished_num) {
360 ESP_LOGW(TAG, "ESP_ERR_ESPNOW_OTA_DEVICE_NO_EXIST");
361 ret = ESP_ERR_ESPNOW_OTA_DEVICE_NO_EXIST;
362 } else if (response_num > 0) {
363 ESP_LOGW(TAG, "ESP_ERR_ESPNOW_OTA_SEND_PACKET_LOSS");
364 ret = ESP_ERR_ESPNOW_OTA_SEND_PACKET_LOSS;
365 } else if (result->requested_num > 0) {
366 ret = ESP_ERR_ESPNOW_OTA_FIRMWARE_INCOMPLETE;
367 ESP_LOGD(TAG, "ESP_ERR_ESPNOW_OTA_FIRMWARE_INCOMPLETE");
368 }
369
370 ESP_FREE(response_addrs);
371 ESP_FREE(response_data);
372 return ret;
373}

375esp_err_t espnow_ota_initiator_send(const uint8_t addrs_list[][6], size_t addrs_num,
376 const uint8_t sha_256[ESPNOW_OTA_HASH_LEN], size_t size,
377 espnow_ota_initiator_data_cb_t ota_data_cb, espnow_ota_result_t *res)
378{
379 ESP_PARAM_CHECK(addrs_list);
380 ESP_PARAM_CHECK(addrs_num);
381
382 esp_err_t ret = ESP_OK;
383 espnow_ota_status_t status = {
384 .type = ESPNOW_OTA_TYPE_STATUS,
385 .total_size = size,
386 .packet_num = (size + ESPNOW_OTA_PACKET_MAX_SIZE - 1) / ESPNOW_OTA_PACKET_MAX_SIZE,
387 };
388 memcpy(status.sha_256, sha_256, ESPNOW_OTA_HASH_LEN);
389
390 ESP_LOGI(TAG, "[%s, %d]: total_size: %d, packet_num: %d", __func__, __LINE__, size, status.packet_num);
391
392 espnow_ota_packet_t *packet = ESP_MALLOC(sizeof(espnow_ota_packet_t));
393 uint8_t (*progress_array)[ESPNOW_OTA_PROGRESS_MAX_SIZE] = ESP_MALLOC(status.packet_num / 8 + 1);
394 espnow_ota_result_t *result = ESP_CALLOC(1, sizeof(espnow_ota_result_t));
395 g_ota_send_running_flag = true;
396
397 espnow_frame_head_t frame_head = {
398 .broadcast = true,
399 .retransmit_count = CONFIG_ESPNOW_OTA_RETRANSMISSION_TIMES,
400 .group = true,
401 .forward_ttl = CONFIG_ESPNOW_OTA_SEND_FORWARD_TTL,
402 .forward_rssi = CONFIG_ESPNOW_OTA_SEND_FORWARD_RSSI,
403 .security = CONFIG_ESPNOW_OTA_SECURITY,
404 };
405
406 if (addrs_num == 1 && ESPNOW_ADDR_IS_BROADCAST(addrs_list[0])) {
407 espnow_ota_responder_t *info_list = NULL;
408 size_t info_num = 0;
409 ret = espnow_ota_initiator_scan(&info_list, &info_num, 3000);
410 ESP_ERROR_GOTO(ret != ESP_OK, EXIT, "<%s> espnow_ota_initiator_scan", esp_err_to_name(ret));
411
412 ESP_LOGI(TAG, "Scan OTA list, num: %d", info_num);
413
414 result->unfinished_num = info_num;
415 result->unfinished_addr = ESP_MALLOC(result->unfinished_num * ESPNOW_ADDR_LEN);
416
417 for (size_t i = 0; i < info_num; i++) {
418 memcpy(result->unfinished_addr[i], info_list[i].mac, ESPNOW_ADDR_LEN);
419 }
420
421 espnow_ota_initiator_scan_result_free();
422 } else {
423 result->unfinished_num = addrs_num;
424 result->unfinished_addr = ESP_CALLOC(result->unfinished_num, ESPNOW_ADDR_LEN);
425 memcpy(result->unfinished_addr, addrs_list, result->unfinished_num * ESPNOW_ADDR_LEN);
426 }
427
428 espnow_set_group(addrs_list, addrs_num, ESPNOW_ADDR_GROUP_OTA, NULL, true, portMAX_DELAY);
429 /* Set queue size to unfinished num to avoid send queue failed */
430 g_ota_queue = xQueueCreate(result->unfinished_num, sizeof(espnow_ota_data_t));
431 ESP_ERROR_GOTO(!g_ota_queue, EXIT, "Create espnow ota queue fail");
432 espnow_set_config_for_data_type(ESPNOW_DATA_TYPE_OTA_STATUS, 1, espnow_ota_initiator_status_process);
433
434 packet->type = ESPNOW_OTA_TYPE_DATA;
435 packet->size = ESPNOW_OTA_PACKET_MAX_SIZE;
436
437 ESP_LOGD(TAG, "packet_num: %d, total_size: %d", status.packet_num, status.total_size);
438
439 for (int i = 0; i < CONFIG_ESPNOW_OTA_RETRY_COUNT && result->unfinished_num > 0 && g_ota_send_running_flag; ++i) {
440 /**
441 * @brief Request all devices upgrade status.
442 */
443 memset(progress_array, 0xff, status.packet_num / 8 + 1);
444
445 ret = espnow_ota_request_status(progress_array, &status, result);
446 ESP_ERROR_BREAK(ret == ESP_OK || ret == ESP_ERR_ESPNOW_OTA_DEVICE_NO_EXIST, "");
447
448 ESP_LOGI(TAG, "count: %d, Upgrade_initiator_send, requested_num: %d, unfinished_num: %d, successed_num: %d",
449 i, result->unfinished_num, result->requested_num, result->successed_num);
450 ESP_LOG_BUFFER_HEXDUMP(TAG, progress_array, sizeof(espnow_ota_status_t) + ESPNOW_OTA_PROGRESS_MAX_SIZE, ESP_LOG_DEBUG);
451
452 for (packet->seq = 0; result->requested_num > 0 && packet->seq < status.packet_num && g_ota_send_running_flag; ++packet->seq) {
453 if (!ESPNOW_OTA_GET_BITS(progress_array, packet->seq)) {
454 packet->size = (packet->seq == status.packet_num - 1) ? status.total_size - ESPNOW_OTA_PACKET_MAX_SIZE * packet->seq : ESPNOW_OTA_PACKET_MAX_SIZE;
455 /**
456 * @brief Read firmware data from Flash to send to unfinished device.
457 */
458 ret = ota_data_cb(packet->seq * ESPNOW_OTA_PACKET_MAX_SIZE, packet->data, packet->size);
459 ESP_ERROR_GOTO(ret != ESP_OK, EXIT, "<%s> Read data from Flash", esp_err_to_name(ret));
460
461 ESP_LOGD(TAG, "seq: %d, size: %d, addrs_num: %d", packet->seq, packet->size, result->requested_num);
462 ret = espnow_send(ESPNOW_DATA_TYPE_OTA_DATA, ESPNOW_ADDR_GROUP_OTA, packet, sizeof(espnow_ota_packet_t), &frame_head, portMAX_DELAY);
463 ESP_ERROR_CONTINUE(ret != ESP_OK, "<%s> espnow write", esp_err_to_name(ret));
464 }
465 }
466 }
467
468EXIT:
469
470 espnow_set_config_for_data_type(ESPNOW_DATA_TYPE_OTA_STATUS, 0, NULL);
471 if (g_ota_queue) {
472 espnow_ota_data_t *tmp_data = NULL;
473
474 while (xQueueReceive(g_ota_queue, &tmp_data, 0)) {
475 ESP_FREE(tmp_data->data);
476 }
477
478 vQueueDelete(g_ota_queue);
479 g_ota_queue = NULL;
480 }
481
482 if (result->unfinished_num > 0) {
483 espnow_set_group(result->unfinished_addr, result->unfinished_num, ESPNOW_ADDR_GROUP_OTA, NULL, false, portMAX_DELAY);
484 ret = ESP_ERR_ESPNOW_OTA_FIRMWARE_INCOMPLETE;
485 }
486
487 g_ota_send_running_flag = false;
488
489 if (res) {
490 memcpy(res, result, sizeof(espnow_ota_result_t));
491 } else {
492 espnow_ota_initiator_result_free(result);
493 }
494
495 ESP_FREE(packet);
496 ESP_FREE(progress_array);
497 ESP_FREE(result);
498
499 if (g_ota_send_exit_sem) {
500 xSemaphoreGive(g_ota_send_exit_sem);
501 }
502
503 return ret;
504}

506esp_err_t espnow_ota_initiator_result_free(espnow_ota_result_t *result)
507{
508 ESP_PARAM_CHECK(result);
509
510 result->unfinished_num = 0;
511 result->requested_num = 0;
512 result->successed_num = 0;
513 ESP_FREE(result->unfinished_addr);
514 ESP_FREE(result->requested_addr);
515 ESP_FREE(result->successed_addr);
516
517 return ESP_OK;
518}

520esp_err_t espnow_ota_initiator_stop()
521{
522 if (!g_ota_send_running_flag) {
523 return ESP_OK;
524 }
525
526 if (!g_ota_send_exit_sem) {
527 g_ota_send_exit_sem = xSemaphoreCreateBinary();
528 }
529
530 g_ota_send_running_flag = false;
531
532 xSemaphoreTake(g_ota_send_exit_sem, portMAX_DELAY);
533 vSemaphoreDelete(g_ota_send_exit_sem);
534 g_ota_send_exit_sem = NULL;
535
