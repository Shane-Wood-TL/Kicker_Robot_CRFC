54static esp_err_t app_espnow_debug_recv_process(uint8_t *src_addr, void *data,
55 size_t size, wifi_pkt_rx_ctrl_t *rx_ctrl)
56{
57 ESP_PARAM_CHECK(src_addr);
58 ESP_PARAM_CHECK(data);
59 ESP_PARAM_CHECK(size);
60 ESP_PARAM_CHECK(rx_ctrl);
61
62 char *recv_data = (char *)data;
63 web_server_send(src_addr, recv_data, size, rx_ctrl);
64
65 printf("[" MACSTR "][%d][%d]: %.*s", MAC2STR(src_addr), rx_ctrl->channel, rx_ctrl->rssi, size, recv_data);
66 fflush(stdout);
67
68 if (sdcard_is_mount()) {
69 char file_name[32] = {0x0};
70 sprintf(file_name, "%02x-%02x-%02x-%02x-%02x-%02x.log", MAC2STR(src_addr));
71 size = (recv_data[size - 1] == '\0') ? size - 1 : size;
72 sdcard_write_file(file_name, UINT32_MAX, recv_data, size);
73 }
74
75
76 return ESP_OK;
77}

103static uint8_t app_find_device_channel(const uint8_t addr[ESPNOW_ADDR_LEN])
104{
105 esp_err_t ret = ESP_OK;
106 char *data = "beacon";
107 static wifi_country_t country = {0};
108
109 espnow_frame_head_t frame_head = {
110 .retransmit_count = 10,
111 .broadcast = true,
112 .magic = esp_random(),
113 .filter_adjacent_channel = true,
114 };
115
116 esp_wifi_get_country(&country);
117 espnow_set_config_for_data_type(ESPNOW_DATA_TYPE_DEBUG_LOG, true, app_espnow_debug_recv_beacon);
118 device_channel = 0;
119
120 for (int i = 0; i < country.nchan; ++i) {
121 esp_wifi_set_channel(country.schan + i, WIFI_SECOND_CHAN_NONE);
122 frame_head.channel = country.schan + i;
123
124 ret = espnow_send(ESPNOW_DATA_TYPE_DEBUG_COMMAND, ESPNOW_ADDR_BROADCAST,
125 data, strlen(data) + 1, &frame_head, portMAX_DELAY);
126 ESP_ERROR_CONTINUE(ret != ESP_OK, "<%s> espnow_send", esp_err_to_name(ret));
127
128 vTaskDelay(pdMS_TO_TICKS(100));
129 if (device_channel) {
130 break;
131 }
132 }
133
134 espnow_set_config_for_data_type(ESPNOW_DATA_TYPE_DEBUG_LOG, false, NULL);
135
136 return device_channel;
137}

173static esp_err_t app_init_fs(void)
174{
175 esp_vfs_spiffs_conf_t conf = {
176 .base_path = CONFIG_APP_WEB_MOUNT_POINT,
177 .partition_label = "www",
178 .max_files = 5,
179 .format_if_mount_failed = false
180 };
181 esp_err_t ret = esp_vfs_spiffs_register(&conf);
182
183 if (ret != ESP_OK) {
184 if (ret == ESP_FAIL) {
185 ESP_LOGE(TAG, "Failed to mount or format filesystem");
186 } else if (ret == ESP_ERR_NOT_FOUND) {
187 ESP_LOGE(TAG, "Failed to find SPIFFS partition");
188 } else {
189 ESP_LOGE(TAG, "Failed to initialize SPIFFS (%s)", esp_err_to_name(ret));
190 }
191 return ESP_FAIL;
192 }
193
194 size_t total = 0, used = 0;
195 ret = esp_spiffs_info("www", &total, &used);
196 if (ret != ESP_OK) {
197 ESP_LOGE(TAG, "Failed to get SPIFFS partition information (%s)", esp_err_to_name(ret));
198 } else {
199 ESP_LOGI(TAG, "Partition size: total: %d, used: %d", total, used);
200 }
201 return ESP_OK;
202}
