151/**
152 * @brief Frame header of espnow
153 */
154typedef struct espnow_frame_head_s {
155 uint16_t magic; /**< Unique identifier of each packet. Packets with the same identifier will be filtered. 0: a random number */
156 uint8_t channel : 4; /**< Set the channel where the packet is sent, ESPNOW_CHANNEL_CURRENT or ESPNOW_CHANNEL_ALL */
157 bool filter_adjacent_channel : 1; /**< Because ESP-NOW is sent through HT20, it can receive packets from adjacent channels */
158 bool filter_weak_signal : 1; /**< When the signal received by the receiving device is lower than forward_rssi, frame_head data will be discarded */
159 bool security : 1; /**< The payload data is encrypted if security is true */
160 uint16_t : 4; /**< Reserved */
161
162 /**
163 * @brief Configure broadcast
164 */
165 bool broadcast : 1; /**< Packet sent in broadcast mode or unicast mode */
166 bool group : 1; /**< Only the group set as broadcast transmission mode is valid */
167 bool ack : 1; /**< Wait for the receiving device to return ack to ensure transmission reliability */
168 uint16_t retransmit_count : 5; /**< Too many packet retransmissions will lead to network congestion */
169 uint8_t forward_ttl : 5; /**< Number of hops in data transfer */
170 int8_t forward_rssi : 8; /**< When the data packet signal received by the receiving device is lower than forward_rssi, it will not be transferred,
171 in order to avoid network congestion caused by packet transfer */
172} ESPNOW_PACKED_STRUCT espnow_frame_head_t;
