21 last_network_channel = current_network_channel;
22 xSemaphoreGive(network_channel_mutex);
23 }
24
25 while (1) {
26 if (ps4_controller_connected) {
27 //only controller data if the ps4 controller is connected
28
29 //raw data from the joysticks
30 current_transmission.driving_speed = controller_byte_2;
31 current_transmission.turning_speed = controller_byte_3;
32
33 //ramped velocity values
34 if (xSemaphoreTake(ramped_velocity_mutex, portMAX_DELAY) == pdTRUE) {
35 current_transmission.velocity_ramp_limit = velocity_ramp_limit;
36 current_transmission.velocity_gain = velocity_gain;
37 current_transmission.velocity_integrator_gain = velocity_integrator_gain;
38 xSemaphoreGive(ramped_velocity_mutex);
39 }
40
41 //servo status value
42 if (xSemaphoreTake(servo_status_mutex, portMAX_DELAY) == pdTRUE) {
43 current_transmission.servo_and_motor_state = (servo_status & lower_four_bits);
44 xSemaphoreGive(servo_status_mutex);
45 }
46
47 //motor status value
48 if (xSemaphoreTake(motor_status_mutex, portMAX_DELAY) == pdTRUE) {
49 if (boost && motor_status == IDLE) {
50 current_transmission.servo_and_motor_state |= ((BOOSTED & lower_four_bits) left_shift_4);
51 } else {
52 current_transmission.servo_and_motor_state |= ((motor_status & lower_four_bits) left_shift_4);
53 }
54 xSemaphoreGive(motor_status_mutex);
55 }
56
57 //motor speed values
58 if (xSemaphoreTake(motor_speeds_mutex, portMAX_DELAY) == pdTRUE) {
59 current_transmission.motor_speed_setting = ((drive_motor_speed & lower_four_bits) left_shift_4) | (turn_motor_speed & lower_four_bits);
60 xSemaphoreGive(motor_speeds_mutex);
61 }
62 } else {
63 // if the controller is not connected, send 0 values and disable motors
64 current_transmission.driving_speed = 0;
65 current_transmission.turning_speed = 0;
66 current_transmission.servo_and_motor_state = (LATCHED & lower_four_bits);
67 current_transmission.servo_and_motor_state |= ((DISABLED & lower_four_bits) left_shift_4);
68 }
69
70 //update the network channel even if the ps4 controller is not connected
71 if (xSemaphoreTake(network_channel_mutex, portMAX_DELAY) == pdTRUE) {
72 current_transmission.esp_now_channel = current_network_channel;
73 if (current_network_channel != last_network_channel) {
74 //set the new network channel
75 esp_wifi_set_channel(current_network_channel, WIFI_SECOND_CHAN_NONE);
76 last_network_channel = current_network_channel;
77
78 //update peer settings
79 esp_now_del_peer(robotMacAddress);
80 esp_now_peer_info_t peer_info = {};
81 memcpy(peer_info.peer_addr, robotMacAddress, mac_address_length);
82 peer_info.channel = current_network_channel;
83 peer_info.encrypt = false;
84 ESP_ERROR_CHECK(esp_now_add_peer(&peer_info));
85 }
86 xSemaphoreGive(network_channel_mutex);
87 }
88
89 //transmit the message to the robot
90 (void) esp_now_send(robotMacAddress, (uint8_t *) &current_transmission, sizeof(esp_now_data_to_send));
91
92 vTaskDelay(pdMS_TO_TICKS(send_data_task_wait_time));
93 }
94}
