55esp_err_t pcap_capture_packet(void *payload, uint32_t length, uint32_t seconds, uint32_t microseconds)
56{
57 ESP_ERROR_RETURN(!file, ESP_FAIL, "file empty");
58
59 size_t real_write = 0;
60 pcap_packet_header_t header = {
61 .seconds = seconds,
62 .microseconds = microseconds,
63 .capture_length = length,
64 .packet_length = length
65 };
66 real_write = fwrite(&header, sizeof(header), 1, file);
67 ESP_ERROR_RETURN(real_write != 1, ESP_FAIL, "Write packet header error");
68 real_write = fwrite(payload, sizeof(uint8_t), length, file);
69 ESP_ERROR_RETURN(real_write != length, ESP_FAIL, "Write packet payload error");
70 /* Flush content in the buffer into device */
71 fflush(file);
72 return ESP_OK;
73}

75esp_err_t pcap_close(void)
76{
77 ESP_ERROR_RETURN(!file, ESP_OK, "file not exist, or already close");
78
79 if (fclose(file)) {
80 ESP_LOGE(TAG, "Close pcap file failed");
81 file = NULL;
82 return ESP_FAIL;
83 }
84
85 ESP_LOGI(TAG, "Close Pcap file OK");
86 file = NULL;
87 return ESP_OK;
88}

90esp_err_t pcap_new(pcap_config_t *config)
91{
92 file = config->fp;
93 /* Write Pcap File header */
94 pcap_file_header_t header = {
95 .magic = PCAP_MAGIC_BIG_ENDIAN,
96 .major = PCAP_VERSION_MAJOR,
97 .minor = PCAP_VERSION_MINOR,
98 .zone = PCAP_TIME_ZONE_GMT,
99 .sigfigs = 0,
100 .snaplen = 0x40000,
101 .link_type = config->link_type
102 };
103 size_t real_write = fwrite(&header, sizeof(header), 1, file);
104
105 ESP_ERROR_GOTO(real_write != 1, err_write, "Write Pcap file header error");
106
107 /* Flush content in the buffer into device */
108 fflush(file);
109 return ESP_OK;
110
111 /* Error Handling */
112err_write:
113 fclose(file);
114 return ESP_FAIL;
115}
