58static void *log_info_storage_get(const char *key)
59{
60 esp_err_t err = ESP_FAIL;
61 nvs_handle_t handle = 0;
62 void *value = NULL;
63 size_t required_size = 0;
64
65 if ((err = nvs_open_from_partition(CONFIG_ESPNOW_DEBUG_LOG_PARTITION_LABEL_NVS, LOG_FLASH_STORE_NAMESPACE,
66 NVS_READONLY, &handle)) != ESP_OK) {
67 ESP_LOGW(TAG, "<%s> NVS open for %s %s %s failed",
68 esp_err_to_name(err), CONFIG_ESPNOW_DEBUG_LOG_PARTITION_LABEL_NVS, LOG_FLASH_STORE_NAMESPACE, key);
69 return NULL;
70 }
71
72 if ((err = nvs_get_blob(handle, key, NULL, &required_size)) == ESP_OK) {
73 value = ESP_CALLOC(required_size + 1, 1); /* + 1 for NULL termination */
74 nvs_get_blob(handle, key, value, &required_size);
75 } else if ((err = nvs_get_str(handle, key, NULL, &required_size)) == ESP_OK) {
76 value = ESP_CALLOC(required_size + 1, 1); /* + 1 for NULL termination */
77 nvs_get_str(handle, key, value, &required_size);
78 }
79
80 nvs_close(handle);
81 return value;
82}

84static esp_err_t log_info_storage_set(const char *key, void *data, size_t len)
85{
86 nvs_handle_t handle;
87 esp_err_t err;
88
89 if ((err = nvs_open_from_partition(CONFIG_ESPNOW_DEBUG_LOG_PARTITION_LABEL_NVS, LOG_FLASH_STORE_NAMESPACE,
90 NVS_READWRITE, &handle)) != ESP_OK) {
91 ESP_LOGE(TAG, "NVS open failed with error %d", err);
92 return ESP_FAIL;
93 }
94
95 if ((err = nvs_set_blob(handle, key, data, len)) != ESP_OK) {
96 ESP_LOGE(TAG, "Failed to write key %s with error %d size %d", key, err, len);
97 nvs_close(handle);
98 return ESP_FAIL;
99 }
100
101 nvs_commit(handle);
102 nvs_close(handle);
103 return ESP_OK;
104}

106static esp_err_t log_info_storage_erase(const char *key)
107{
108 nvs_handle_t handle;
109 esp_err_t err;
110
111 if ((err = nvs_open_from_partition(CONFIG_ESPNOW_DEBUG_LOG_PARTITION_LABEL_NVS, LOG_FLASH_STORE_NAMESPACE,
112 NVS_READWRITE, &handle)) != ESP_OK) {
113 ESP_LOGE(TAG, "NVS open failed with error %d", err);
114 return ESP_FAIL;
115 }
116
117 /**
118 * @brief If key is LOG_FLASH_STORE_NAMESPACE, erase all info in LOG_FLASH_STORE_NAMESPACE
119 */
120 if (!strcmp(key, LOG_FLASH_STORE_NAMESPACE)) {
121 err = nvs_erase_all(handle);
122 } else {
123 err = nvs_erase_key(handle, key);
124 }
125
126 nvs_commit(handle);
127 nvs_close(handle);
128 return err;
129}

131esp_err_t espnow_log_flash_init()
132{
133 if (g_espnow_log_flash_init_flag) {
134 return ESP_OK;
135 }
136
137 esp_err_t err = ESP_OK;
138
139 err = log_info_storage_init();
140 ESP_ERROR_RETURN(err != ESP_OK, err, "log_info_storage_init");
141
142 esp_partition_iterator_t part_itra = esp_partition_find(ESP_PARTITION_TYPE_DATA,
143 ESP_PARTITION_SUBTYPE_ANY, CONFIG_ESPNOW_DEBUG_LOG_PARTITION_LABEL_DATA);
144 ESP_ERROR_RETURN(!part_itra, ESP_ERR_NOT_SUPPORTED, "partition no find, subtype: 0x%x, label: %s",
145 ESP_PARTITION_SUBTYPE_ANY, CONFIG_ESPNOW_DEBUG_LOG_PARTITION_LABEL_DATA);
146
147 g_log_part = esp_partition_get(part_itra);
148
149 ESP_ERROR_RETURN(!g_log_part, ESP_ERR_NOT_SUPPORTED, "esp_partition_get");
150 ESP_ERROR_RETURN(g_log_part->size < LOG_FLASH_FILE_MAX_SIZE, ESP_ERR_NOT_SUPPORTED,
151 "Log file (%d Byte) size must be smaller than partition size (%d Byte).",
152 LOG_FLASH_FILE_MAX_SIZE, g_log_part->size);
153 ESP_ERROR_RETURN(LOG_FLASH_FILE_MAX_SIZE / LOG_FLASH_FILE_MAX_NUM % 4096 != 0, ESP_ERR_NOT_SUPPORTED,
154 "The size of the log partition must be an integer of %d KB.", LOG_FLASH_FILE_MAX_NUM * 4);
155
156 g_log_info = log_info_storage_get(LOG_FLASH_STORE_KEY);
157
158 if (!g_log_info) {
159 g_log_info = ESP_CALLOC(LOG_FLASH_FILE_MAX_NUM, sizeof(flash_log_info_t));
160 err = esp_partition_erase_range(g_log_part, CONFIG_ESPNOW_DEBUG_LOG_PARTITION_OFFSET, LOG_FLASH_FILE_MAX_SIZE);
161 ESP_ERROR_RETURN(err != ESP_OK, err, "esp_partition_erase_range");
162 }
163
164 /**< Create two files */
165 for (size_t i = 0, min_size = LOG_FLASH_FILE_MAX_SIZE / LOG_FLASH_FILE_MAX_NUM; i < LOG_FLASH_FILE_MAX_NUM; i++) {
166 (g_log_info + i)->addr = CONFIG_ESPNOW_DEBUG_LOG_PARTITION_OFFSET + LOG_FLASH_FILE_MAX_SIZE / LOG_FLASH_FILE_MAX_NUM * i;
167
168 /**< Whether file is full */
169 if ((g_log_info + i)->size < LOG_FLASH_FILE_MAX_SIZE / LOG_FLASH_FILE_MAX_NUM && (g_log_info + i)->size < min_size) {
170 min_size = (g_log_info + i)->size;
171 g_log_index = i; /**< File index */
172 }
173 }
174
175 g_espnow_log_flash_init_flag = true;
176 ESP_LOGI(TAG, "LOG flash initialized successfully");
177 ESP_LOGI(TAG, "Log save partition subtype: label: %s, addr:0x%x, offset: %d, size: %d",
178 CONFIG_ESPNOW_DEBUG_LOG_PARTITION_LABEL_DATA, g_log_part->address, CONFIG_ESPNOW_DEBUG_LOG_PARTITION_OFFSET, g_log_part->size);
179
180 return ESP_OK;
181}

196esp_err_t espnow_log_flash_write(const char *data, size_t size, esp_log_level_t level)
197{
198 esp_err_t err = ESP_OK;
199 static bool flag_timestamp = true;
200
201 ESP_PARAM_CHECK(data);
202 ESP_PARAM_CHECK(size > 0);
203
204 if (!g_espnow_log_flash_init_flag) {
205 return ESP_FAIL;
206 }
207
208 /**
209 * @brief Get the address of the pointer to the next file and clear the file to be written
210 * and clear offset size.
211 */
212 if ((g_log_info + g_log_index)->size + size > LOG_FLASH_FILE_MAX_SIZE / LOG_FLASH_FILE_MAX_NUM) {
213 g_log_index = (g_log_index + 1) % LOG_FLASH_FILE_MAX_NUM;
214 (g_log_info + g_log_index)->size = 0;
215 (g_log_info + g_log_index)->offset = 0;
216
217 err = esp_partition_erase_range(g_log_part, (g_log_info + g_log_index)->addr, LOG_FLASH_FILE_MAX_SIZE / LOG_FLASH_FILE_MAX_NUM);
218 ESP_ERROR_RETURN(err != ESP_OK, err, "esp_partition_erase_range, addr: %x", (g_log_info + g_log_index)->addr);
219 }
220
221 static uint32_t s_event_send_tick = 0;
222
223 if (espnow_log_flash_size() > LOG_FLASH_FILE_MAX_SIZE / LOG_FLASH_FILE_MAX_NUM
224 && (xTaskGetTickCount() - s_event_send_tick > 30000 || !s_event_send_tick)) {
225 s_event_send_tick = xTaskGetTickCount();
226 esp_event_post(ESP_EVENT_ESPNOW, ESP_EVENT_ESPNOW_LOG_FLASH_FULL, NULL, 0, portMAX_DELAY);
227 }
228
229 DEBUG_LOG_PRINTF("esp_partition_write, addr: %d, offset: %d, log_index: %d, size: %d\n",
230 (g_log_info + g_log_index)->addr, (g_log_info + g_log_index)->size, g_log_index, size);
231
232 if (flag_timestamp) {
233 flag_timestamp = false;
234
235 /**< Get the current timestamp */
236 time_t now = 0;
237 struct tm log_time = {0};
238 char strtime_buf[32] = {0};
239
240 time(&now);
241 localtime_r(&now, &log_time);
242 strftime(strtime_buf, sizeof(strtime_buf), "[%Y-%m-%d %H:%M:%S] ", &log_time);
243
244 /**
245 * @brief Change the write file address, then write timestamp data
246 */
247 err = esp_partition_write(g_log_part, (g_log_info + g_log_index)->addr + (g_log_info + g_log_index)->size, strtime_buf, strlen(strtime_buf));
248 ESP_ERROR_RETURN(err != ESP_OK, err, "esp_partition_write");
249 (g_log_info + g_log_index)->size += strlen(strtime_buf);
250 }
251
252 /**
253 * @brief Change the write file address, then write log data after the timestamp.
254 *
255 * @note First need to get the length of the timestamp, then write the log data after the timestamp
256 */
257 err = esp_partition_write(g_log_part, (g_log_info + g_log_index)->addr + (g_log_info + g_log_index)->size, data, size);
258 ESP_ERROR_RETURN(err != ESP_OK, err, "esp_partition_write");
259 (g_log_info + g_log_index)->size += size;
260
261 /**
262 * @brief Here in order to read the data address written by the file,Get where the data address is written.
263 *
264 */
265 log_info_storage_set(LOG_FLASH_STORE_KEY, g_log_info, sizeof(flash_log_info_t) * LOG_FLASH_FILE_MAX_NUM);
266
267 if (data[size - 1] == '\n') {
268 flag_timestamp = true;
269 }
270
271 return ESP_OK;
272}

274esp_err_t espnow_log_flash_read(char *data, size_t *size)
275{
276 esp_err_t err = ESP_OK;
277 size_t read_size = 0;
278 ESP_PARAM_CHECK(data);
279 ESP_PARAM_CHECK(size && *size > 0);
280
281 if (!g_espnow_log_flash_init_flag) {
282 *size = 0;
283 return ESP_FAIL;
284 }
285
286 /**
287 * @brief First find Target file index address, then read the this files data
288 *
289 * @note The file address read is different from the file address written, so that the same address
290 * can be prevented from being written and read.
291 */
292 for (int i = 0; i < LOG_FLASH_FILE_MAX_NUM && *size > read_size; ++i, data += read_size) {
293 flash_log_info_t *log_info = g_log_info + (g_log_index + 1 + i) % LOG_FLASH_FILE_MAX_NUM;
294 ssize_t data_size = MIN(*size - read_size, log_info->size - log_info->offset);
295
296 if (data_size > 0) {
297 err = esp_partition_read(g_log_part, log_info->addr + log_info->offset, data, data_size);
298
299 if (err != ESP_OK) {
300 ESP_LOGW(TAG, "<%s> fread packet, ret: %d, *size: %d, read_size: %d, log_info->offset: %d, log_info->size: %d, index: %d",
301 strerror(err), err, *size, read_size, log_info->offset, log_info->size, i);
302 log_info->offset = log_info->size = 0;
303 break;
304 }
305
306 read_size += data_size;
307 log_info->offset += data_size;
308 DEBUG_LOG_PRINTF("esp_partition_read, data_size: %d, read_size: %d, log_info->offset: %d, log_info->size: %d, ret: %d, index: %d, data: %.*s\n",
309 data_size, read_size, log_info->offset, log_info->size, err, i, data_size, data);
310
311 log_info_storage_set(LOG_FLASH_STORE_KEY, g_log_info, sizeof(flash_log_info_t) * LOG_FLASH_FILE_MAX_NUM);
312 }
313 }
314
315 *size = read_size;
316 return read_size > 0 ? ESP_OK : ESP_FAIL;
317}
