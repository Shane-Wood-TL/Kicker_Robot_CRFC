83/**
84 * @enum motor_status_list
85 * @brief Contains the different types of states for the Odrive motor controllers
86 *
87 * @details motor_status_list has the following options: ENABLED, DISABLED, ERRORLESS, CALIBRATING, IDLE
88 * ENABLED = send messages to the motor controller to enable the motors
89 * DISABLED = motor set to idle
90 * ERRORLESS = clears motor errors
91 * CALIBRATING = motor is running it's calibration sequence
92 * IDLE = motor state does not change (if it was active it remains active, if it was disabled it remains disabled)
93 * BOOSTED = motor is in a boosted state, this is used to increase the speed of the robot when the button is pressed
94 */
95enum motor_status_list { ENABLED,
96 DISABLED,
97 ERRORLESS,
98 CALIBRATING,
99 IDLE,
100 BOOSTED };
101
102/**
103 * @enum servo_status_list
104 * @brief Contains the different types of states for the servo motors
105 *
106 * @details servo_status_list has the following options: LATCHED, RELEASED, DETACHED
107 * LATCHED = servo is in position to keep the kicking leg from moving
108 * RELEASED = servo is in position to release the kicking leg
109 * DETACHED = state servo is in on boot, will update to the expected state when controller connects
110 */
111enum servo_status_list { LATCHED,
112 RELEASED,
113 DETACHED };
