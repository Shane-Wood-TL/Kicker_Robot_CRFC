41/**
42 * Macro which can be used to check the error code,
43 * and terminate the program in case the code is not ESP_OK.
44 * Prints the error code, error location, and the failed statement to serial output.
45 *
46 * Disabled if assertions are disabled.
47 */
48#define ESP_ERROR_RETURN(con, err, format, ...) do { \
49 if (con) { \
50 if(*format != '\0') \
51 ESP_LOGW(TAG, "[%s, %d] <%s> " format, __func__, __LINE__,esp_err_to_name(err), ##__VA_ARGS__); \
52 return err; \
53 } \
54 } while(0)

56/**
57 * @brief Macro serves similar purpose as ``assert``, except that it checks `esp_err_t`
58 * value rather than a `bool` condition. If the argument of `ESP_ERROR_ASSERT`
59 * is not equal `ESP_OK`, then an error message is printed on the console,
60 * and `abort()` is called.
61 *
62 * @note If `IDF monitor` is used, addresses in the backtrace will be converted
63 * to file names and line numbers.
64 *
65 * @param[in] err [description]
66 * @return [description]
67 */
68#define ESP_ERROR_ASSERT(err) do { \
69 esp_err_t __err_rc = (err); \
70 if (__err_rc != ESP_OK) { \
71 ESP_LOGW(TAG, "[%s, %d] <%s> ESP_ERROR_ASSERT failed, at 0x%08x, expression: %s", \
72 __func__, __LINE__, esp_err_to_name(__err_rc), (intptr_t)__builtin_return_address(0) - 3, __ASSERT_FUNC); \
73 assert(0 && #err); \
74 } \
75 } while(0)

77#define ESP_ERROR_GOTO(con, lable, format, ...) do { \
78 if (con) { \

85#define ESP_ERROR_CONTINUE(con, format, ...) { \
86 if (con) { \
