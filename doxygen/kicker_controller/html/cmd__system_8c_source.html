285static int deep_sleep(int argc, char **argv)
286{
287 int nerrors = arg_parse(argc, argv, (void **) &deep_sleep_args);
288 if (nerrors != 0) {
289 arg_print_errors(stderr, deep_sleep_args.end, argv[0]);
290 return 1;
291 }
292 if (deep_sleep_args.wakeup_time->count) {
293 uint64_t timeout = 1000ULL * deep_sleep_args.wakeup_time->ival[0];
294 ESP_LOGI(TAG, "Enabling timer wakeup, timeout=%lluus", timeout);
295 ESP_ERROR_CHECK( esp_sleep_enable_timer_wakeup(timeout) );
296 }
297
298#if SOC_PM_SUPPORT_EXT_WAKEUP
299 if (deep_sleep_args.wakeup_gpio_num->count) {
300 int io_num = deep_sleep_args.wakeup_gpio_num->ival[0];
301 if (!esp_sleep_is_valid_wakeup_gpio(io_num)) {
302 ESP_LOGE(TAG, "GPIO %d is not an RTC IO", io_num);
303 return 1;
304 }
305 int level = 0;
306 if (deep_sleep_args.wakeup_gpio_level->count) {
307 level = deep_sleep_args.wakeup_gpio_level->ival[0];
308 if (level != 0 && level != 1) {
309 ESP_LOGE(TAG, "Invalid wakeup level: %d", level);
310 return 1;
311 }
312 }
313 ESP_LOGI(TAG, "Enabling wakeup on GPIO%d, wakeup on %s level",
314 io_num, level ? "HIGH" : "LOW");
315
316 ESP_ERROR_CHECK( esp_sleep_enable_ext1_wakeup(1ULL << io_num, level) );
317 ESP_LOGE(TAG, "GPIO wakeup from deep sleep currently unsupported on ESP32-C3");
318 }
319#endif // SOC_PM_SUPPORT_EXT_WAKEUP
320
321#if CONFIG_IDF_TARGET_ESP32
322 rtc_gpio_isolate(GPIO_NUM_12);
323#endif //CONFIG_IDF_TARGET_ESP32
324
325 esp_deep_sleep_start();
326 return 0;
327}

360/** 'light_sleep' command puts the chip into light sleep mode */
361
362static struct {
363 struct arg_int *wakeup_time;
364 struct arg_int *wakeup_gpio_num;
365 struct arg_int *wakeup_gpio_level;
366 struct arg_end *end;
367} light_sleep_args;
368
369static int light_sleep(int argc, char **argv)
370{
371 int nerrors = arg_parse(argc, argv, (void **) &light_sleep_args);
372 if (nerrors != 0) {
373 arg_print_errors(stderr, light_sleep_args.end, argv[0]);
374 return 1;
375 }
376 esp_sleep_disable_wakeup_source(ESP_SLEEP_WAKEUP_ALL);
377 if (light_sleep_args.wakeup_time->count) {
378 uint64_t timeout = 1000ULL * light_sleep_args.wakeup_time->ival[0];
379 ESP_LOGI(TAG, "Enabling timer wakeup, timeout=%lluus", timeout);
380 ESP_ERROR_CHECK( esp_sleep_enable_timer_wakeup(timeout) );
381 }
382 int io_count = light_sleep_args.wakeup_gpio_num->count;
383 if (io_count != light_sleep_args.wakeup_gpio_level->count) {
384 ESP_LOGE(TAG, "Should have same number of 'io' and 'io_level' arguments");
385 return 1;
386 }
387 for (int i = 0; i < io_count; ++i) {
388 int io_num = light_sleep_args.wakeup_gpio_num->ival[i];
389 int level = light_sleep_args.wakeup_gpio_level->ival[i];
390 if (level != 0 && level != 1) {
391 ESP_LOGE(TAG, "Invalid wakeup level: %d", level);
392 return 1;
393 }
394 ESP_LOGI(TAG, "Enabling wakeup on GPIO%d, wakeup on %s level",
395 io_num, level ? "HIGH" : "LOW");
396
397 ESP_ERROR_CHECK( gpio_wakeup_enable(io_num, level ? GPIO_INTR_HIGH_LEVEL : GPIO_INTR_LOW_LEVEL) );
398 }
399 if (io_count > 0) {
400 ESP_ERROR_CHECK( esp_sleep_enable_gpio_wakeup() );
401 }
402 if (CONFIG_ESP_CONSOLE_UART_NUM >= 0 && CONFIG_ESP_CONSOLE_UART_NUM <= UART_NUM_1) {
403 ESP_LOGI(TAG, "Enabling UART wakeup (press ENTER to exit light sleep)");
404 ESP_ERROR_CHECK( uart_set_wakeup_threshold(CONFIG_ESP_CONSOLE_UART_NUM, 3) );
405 ESP_ERROR_CHECK( esp_sleep_enable_uart_wakeup(CONFIG_ESP_CONSOLE_UART_NUM) );
406 }
407 fflush(stdout);
408 fsync(fileno(stdout));
409 esp_light_sleep_start();
410 esp_sleep_wakeup_cause_t cause = esp_sleep_get_wakeup_cause();
411 const char *cause_str;
412 switch (cause) {
413 case ESP_SLEEP_WAKEUP_GPIO:
414 cause_str = "GPIO";
415 break;
416 case ESP_SLEEP_WAKEUP_UART:
417 cause_str = "UART";
418 break;
419 case ESP_SLEEP_WAKEUP_TIMER:
420 cause_str = "timer";
421 break;
422 default:
423 cause_str = "unknown";
424 printf("%d\n", cause);
425 }
426 ESP_LOGI(TAG, "Woke up from: %s", cause_str);
427 return 0;
428}

474 ESP_ERROR_RETURN(ret != ESP_OK, ret, "Erase part of the nvs partition");
475
476 ESP_LOGI(__func__, "Restarting");
477 esp_restart();
478}

531/**
532 * @brief A function which implements coredump command.
533 */
534static int coredump_func(int argc, char **argv)
535{
536 esp_err_t ret = ESP_OK;
537 ssize_t coredump_size = 0;
538 const esp_partition_t *coredump_part = NULL;
539
540 if (arg_parse(argc, argv, (void **)&coredump_args) != ESP_OK) {
541 arg_print_errors(stderr, coredump_args.end, argv[0]);
542 return ESP_FAIL;
543 }
544
545 coredump_part = esp_partition_find_first(ESP_PARTITION_TYPE_DATA,
546 ESP_PARTITION_SUBTYPE_DATA_COREDUMP, NULL);
547 ESP_ERROR_RETURN(coredump_part == NULL, ESP_ERR_NOT_SUPPORTED, "No core dump partition found!");
548
549 ret = esp_partition_read(coredump_part, 4, &coredump_size, sizeof(size_t));
550 ESP_ERROR_RETURN(coredump_part == NULL, ESP_ERR_NOT_SUPPORTED, "Core dump read length!");
551
552 if (coredump_args.length->count) {
553 ESP_LOGI(__func__, "Core dump is length: %d Bytes", coredump_size);
554 }
555
556 if (coredump_args.output->count && coredump_size > 0) {
557#define COREDUMP_BUFFER_SIZE 150
558 uint8_t *buffer = ESP_REALLOC_RETRY(NULL, COREDUMP_BUFFER_SIZE);
559 ESP_LOGI(__func__, "\n================= CORE DUMP START =================\n");
560
561 for (int offset = 4; offset < coredump_size; offset += COREDUMP_BUFFER_SIZE) {
562 size_t size = MIN(COREDUMP_BUFFER_SIZE, coredump_size - offset);
563 esp_partition_read(coredump_part, offset, buffer, size);
564 size_t dlen = (size + 2) / 3 * 4; //base64 encode maximum length = n / 3 * 4
565 size_t olen = 0;
566 uint8_t *b64_buf = ESP_MALLOC(dlen);
567 mbedtls_base64_encode(b64_buf, dlen, &olen, buffer, size);
568 ESP_LOGI(__func__, "%s", b64_buf);
569 ESP_FREE(b64_buf);
570 }
571
572 ESP_LOGI(__func__, "================= CORE DUMP END ===================\n");
573
574 ESP_LOGI(__func__, "1. Save core dump text body to some file manually");
575 ESP_LOGI(__func__, "2. Run the following command: \n"
576 "python esp-idf/components/espcoredump/espcoredump.py info_corefile -t b64 -c </path/to/saved/base64/text> </path/to/program/elf/file>");
577 ESP_FREE(buffer);
578 }
579
580 if (coredump_args.erase->count) {
581 ret = esp_partition_erase_range(coredump_part, 0, coredump_part->size);
582 ESP_ERROR_RETURN(ret != ESP_OK, ESP_FAIL, "Core dump erase fail");
583 ESP_LOGI(__func__, "Core dump erase successful");
584 }
585
586 return ESP_OK;
587}

617/**
618 * @brief A function which implements time command.
619 */
620static int time_func(int argc, char **argv)
621{
622 if (arg_parse(argc, argv, (void **) &time_args) != ESP_OK) {
623 arg_print_errors(stderr, time_args.end, argv[0]);
624 return ESP_FAIL;
625 }
626
627 const char *time_zone = "CST-8";
628
629 if (time_args.time_zone->count) {
630 time_zone = time_args.time_zone->sval[0];
631 }
632
633 if (time_args.set->count) {
634 setenv("TZ", time_zone, 1);
635 tzset();
636 uint64_t tv_sec = 0;
637 sscanf(time_args.set->sval[0], "%lld", (long long int *)&tv_sec);
638
639 struct timeval now = { .tv_sec = tv_sec};
640 settimeofday(&now, NULL);
641 }
642
643 if (time_args.get->count) {
644 struct timeval now = {0};
645 struct tm timeinfo = {0};
646 char strftime_buf[64] = {0};
647
648 setenv("TZ", time_zone, 1);
649 tzset();
650 gettimeofday(&now, NULL);
651 localtime_r(&now.tv_sec, &timeinfo);
652 strftime(strftime_buf, sizeof(strftime_buf), "%c", &timeinfo);
653 ESP_LOGI(TAG, "time_zone: %s, strftime: %s, sec: %lld",
654 time_zone, strftime_buf, (long long)now.tv_sec);
655 }
656
657 return ESP_OK;
658}
