21void ssd1306::send_data_SSD1306(uint8_t *data, size_t len) {
22 uint8_t data_to_send[len + 1] = {0};
23 data_to_send[0] = SSD1306_DATA_PREFIX; // 0x40 is for data

56void ssd1306::clear_SSD1306() {
57 uint8_t emptyBuffer[SSD1306_horizontal_resolution] = {0}; // each page is 128 x 8
58 // display is 128 x 64, each page is 8 pixels high
59 for (uint8_t i = 0; i < (SSD1306_pages); i++) {
60 send_command_SSD1306(SSD1306_PAGE_OFFSET + i); // 0B0 - 0B7 are the 8 vertical pages
61
62 // sets the column address to 0
63 send_command_SSD1306(SSD1306_LOWER_COLUMN_START);
64 send_command_SSD1306(SSD1306_UPPER_COLUMN_START);
65
66 // write the empty buffer to the display
67 send_data_SSD1306(emptyBuffer, sizeof(emptyBuffer));
68 }
69}

73void ssd1306::draw_pixel_SSD1306(uint8_t x, uint8_t y) {
74 if ((x >= SSD1306_horizontal_resolution) || (y >= SSD1306_vertical_resolution)) {
75 return; // Out of bounds
76 }
77 uint8_t pixelInPage = (1 << (y % (SSD1306_pages))); //limit the pixel from 0 to 7 with a 1 in the position 0000_0001, 0000_0010, etc
78 //buffer is indexed as such
79 // x straight across 0-127
80 //each one of these is 8 pixels bits tall, or 1 page
81 //repeat for pages 0-8
82 //leading to a total size of 128*8 = 1024 page "strips" each containing the data for 8 pixels
83 uint16_t bufferIndex = uint16_t(x + ((y / (SSD1306_page_height)) * SSD1306_horizontal_resolution));
84 buffer[bufferIndex] |= pixelInPage; //write to the buffer based on page and column
85}

89void ssd1306::write_buffer_SSD1306() {
90 for (uint8_t i = 0; i < SSD1306_pages; i++) { // Send each page
91 send_command_SSD1306(SSD1306_PAGE_OFFSET + i); // Set page start address
92 send_command_SSD1306(SSD1306_LOWER_COLUMN_START); // Set lower column address
93 send_command_SSD1306(SSD1306_UPPER_COLUMN_START); // Set higher column address
94
95 send_data_SSD1306(&buffer[SSD1306_horizontal_resolution * i], SSD1306_horizontal_resolution);
96 }
97}
