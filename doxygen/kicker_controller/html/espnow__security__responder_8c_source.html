67static esp_err_t espnow_sec_handle(const char *ep_name, uint8_t resp_type, const espnow_addr_t src_addr, const uint8_t *data, size_t size)
68{
69 esp_err_t ret = ESP_OK;
70 protocomm_t *pc = g_espnow_pc;
71 espnow_sec_packet_t *req_data = (espnow_sec_packet_t *)data;
72 espnow_sec_packet_t *response_data = (espnow_sec_packet_t *)ESP_MALLOC(ESPNOW_DATA_LEN);
73 ssize_t response_size = 0;
74 ssize_t outlen = 0;
75 uint8_t *outbuf = NULL;
76 uint32_t session_id = src_addr[5];
77 espnow_frame_head_t frame_head = {
78 .retransmit_count = 1,
79 .broadcast = false,
80 .filter_adjacent_channel = true,
81 .forward_ttl = 0,
82 };
83
84 /* Config or not */
85 if (g_sec_info.sec_ver != ESPNOW_SEC_VER_NONE)
86 return ESP_OK;
87
88 /* Message from the same client */
89 if (!ESPNOW_ADDR_IS_EMPTY(g_sec_info.client_mac)
90 && !ESPNOW_ADDR_IS_EQUAL(src_addr, g_sec_info.client_mac)) {
91 return ESP_OK;
92 }
93
94 /* Record the client */
95 if (ESPNOW_ADDR_IS_EMPTY(g_sec_info.client_mac)) {
96 memcpy(g_sec_info.client_mac, src_addr, 6);
97 protocomm_open_session(pc, session_id);
98 }
99
100 ret = protocomm_req_handle(pc, ep_name, session_id, req_data->data, req_data->size, &outbuf, &outlen);
101 if (ret != ESP_OK) {
102 ESP_LOGE(TAG, "espnow-session handler failed");
103 esp_event_post(ESP_EVENT_ESPNOW, ESP_EVENT_ESPNOW_SEC_FAIL, g_sec_info.client_mac, sizeof(espnow_addr_t), 0);
104 memset(g_sec_info.client_mac, 0, 6);
105 protocomm_close_session(pc, session_id);
106 } else {
107 response_data->type = resp_type;
108 response_data->size = outlen;
109 memcpy(response_data->data, outbuf, outlen);
110 response_size = sizeof(espnow_sec_packet_t) + outlen;
111 ret = espnow_send(ESPNOW_DATA_TYPE_SECURITY, src_addr, response_data, response_size, &frame_head, portMAX_DELAY);
112 if (ret != ESP_OK) {
113 ESP_LOGW(TAG, "espnow-session send failed");
114 }
115 }
116
117 ESP_FREE(outbuf);
118 ESP_FREE(response_data);
119
120 return ret;
121}

123static esp_err_t espnow_sec_responder_process(uint8_t *src_addr, void *data,
124 size_t size, wifi_pkt_rx_ctrl_t *rx_ctrl)
125{
126 ESP_PARAM_CHECK(src_addr);
127 ESP_PARAM_CHECK(data);
128 ESP_PARAM_CHECK(size);
129 ESP_PARAM_CHECK(rx_ctrl);
130
131 esp_err_t ret = ESP_OK;
132 uint8_t data_type = ((uint8_t *)data)[0];
133 espnow_add_peer(src_addr, NULL);
134
135 switch (data_type) {
136 case ESPNOW_SEC_TYPE_REQUEST:
137 ESP_LOGD(TAG, "ESPNOW_SEC_TYPE_INFO");
138 ret = espnow_sec_info(src_addr);
139 break;
140
141 case ESPNOW_SEC_TYPE_REST:
142 ESP_LOGD(TAG, "ESPNOW_SEC_TYPE_REST");
143 ret = espnow_sec_reset_info(src_addr);
144 break;
145
146 case ESPNOW_SEC_TYPE_HANDSHAKE:
147 ESP_LOGD(TAG, "ESPNOW_SEC_TYPE_HANDSHAKE");
148 ret = espnow_sec_handle("espnow-session", ESPNOW_SEC_TYPE_HANDSHAKE, src_addr, data, size);
149 break;
150
151 case ESPNOW_SEC_TYPE_KEY:
152 ESP_LOGD(TAG, "ESPNOW_SEC_TYPE_KEY");
153 ret = espnow_sec_handle("espnow-config", ESPNOW_SEC_TYPE_KEY_RESP, src_addr, data, size);
154 break;
155
156 default:
157 break;
158 }
159
160 espnow_del_peer(src_addr);
161 ESP_ERROR_RETURN(ret != ESP_OK, ret, "espnow_sec_handle");
162 return ret;
163}

173static esp_err_t espnow_config_data_handler(uint32_t session_id, const uint8_t *inbuf, ssize_t inlen,
174 uint8_t **outbuf, ssize_t *outlen, void *priv_data)
175{
176 esp_err_t ret = ESP_OK;
177
178 ESP_PARAM_CHECK(inlen >= APP_KEY_LEN);
179
180 memcpy(app_key, inbuf, APP_KEY_LEN);
181 /* Mark as configured */
182 g_sec_info.sec_ver = ESPNOW_SEC_VER_V1_0;
183
184 /* return the origin message */
185 *outlen = inlen;
186 *outbuf = (uint8_t *) ESP_MALLOC(*outlen);
187 if (!*outbuf) {
188 ESP_LOGE(TAG, "System out of memory");
189 return ESP_ERR_NO_MEM;
190 }
191 memcpy(*outbuf, inbuf, inlen);
192
193 ESP_LOGI(TAG, "Get APP key");
194
195 ret = espnow_set_key(app_key);
196 ret |= espnow_set_dec_key(app_key);
197
198
199 esp_event_post(ESP_EVENT_ESPNOW, ret ? ESP_EVENT_ESPNOW_SEC_FAIL : ESP_EVENT_ESPNOW_SEC_OK, g_sec_info.client_mac, sizeof(espnow_addr_t), 0);
200
201 return ret;
202}

211esp_err_t espnow_sec_responder_start(const char *pop_data)
212{
213 ESP_PARAM_CHECK(pop_data);
214
215 if (g_espnow_pc) {
216 ESP_LOGW(TAG, "Have create the new protocomm instance");
217 return ESP_FAIL;
218 }
219
220 esp_err_t ret;
221#if ESP_IDF_VERSION >= ESP_IDF_VERSION_VAL(5, 0, 0)
222 protocomm_security1_params_t pop = {
223#else
224 protocomm_security_pop_t pop = {
225#endif
226 .data = (const uint8_t *)pop_data,
227 .len = strlen(pop_data)
228 };
229 protocomm_t *pc = protocomm_new();
230 if (pc == NULL) {
231 ESP_LOGE(TAG, "Failed to create new protocomm instance");
232 return ESP_FAIL;
233 }
234
235 /* Set version */
236 ret = protocomm_set_version(pc, "espnow-ver", "v0.1");
237 if (ret != ESP_OK) {
238 ESP_LOGE(TAG, "Failed to set version endpoint");
239 protocomm_delete(pc);
240 return ret;
241 }
242
243 /* Set protocomm security type for endpoint */
244 ret = protocomm_set_security(pc, "espnow-session", &protocomm_security1, &pop);
245 if (ret != ESP_OK) {
246 ESP_LOGE(TAG, "Failed to set security endpoint");
247 protocomm_delete(pc);
248 return ret;
249 }
250
251 /* Add protocomm endpoint for security key */
252 ret = protocomm_add_endpoint(pc, "espnow-config", espnow_config_data_handler, NULL);
253 if (ret != ESP_OK) {
254 ESP_LOGE(TAG, "Failed to set security key endpoint");
255 protocomm_delete(pc);
256 return ret;
257 }
258
259 protocomm_espnow_responder_start(pc);
260
261 return ESP_OK;
262}
