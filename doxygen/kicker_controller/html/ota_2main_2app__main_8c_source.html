30static size_t app_firmware_download(const char *url)
31{
32#define OTA_DATA_PAYLOAD_LEN 1024
33
34 esp_err_t ret = ESP_OK;
35 esp_ota_handle_t ota_handle = 0;
36 uint8_t *data = ESP_MALLOC(OTA_DATA_PAYLOAD_LEN);
37 size_t total_size = 0;
38 uint32_t start_time = xTaskGetTickCount();
39
40 esp_http_client_config_t config = {
41 .url = url,
42 .transport_type = HTTP_TRANSPORT_UNKNOWN,
43 };
44
45 /**
46 * @brief 1. Connect to the server
47 */
48 esp_http_client_handle_t client = esp_http_client_init(&config);
49 ESP_ERROR_GOTO(!client, EXIT, "Initialise HTTP connection");
50
51 ESP_LOGI(TAG, "Open HTTP connection: %s", url);
52
53 /**
54 * @brief First, the firmware is obtained from the http server and stored
55 */
56 do {
57 ret = esp_http_client_open(client, 0);
58
59 if (ret != ESP_OK) {
60 vTaskDelay(pdMS_TO_TICKS(1000));
61 ESP_LOGW(TAG, "<%s> Connection service failed", esp_err_to_name(ret));
62 }
63 } while (ret != ESP_OK);
64
65 total_size = esp_http_client_fetch_headers(client);
66
67 if (total_size <= 0) {
68 ESP_LOGW(TAG, "Please check the address of the server");
69 ret = esp_http_client_read(client, (char *)data, OTA_DATA_PAYLOAD_LEN);
70 ESP_ERROR_GOTO(ret < 0, EXIT, "<%s> Read data from http stream", esp_err_to_name(ret));
71
72 ESP_LOGW(TAG, "Recv data: %.*s", ret, data);
73 goto EXIT;
74 }
75
76 /**
77 * @brief 2. Read firmware from the server and write it to the flash of the root node
78 */
79
80 const esp_partition_t *updata_partition = esp_ota_get_next_update_partition(NULL);
81 /**< Commence an OTA update writing to the specified partition. */
82 ret = esp_ota_begin(updata_partition, total_size, &ota_handle);
83 ESP_ERROR_GOTO(ret != ESP_OK, EXIT, "<%s> esp_ota_begin failed, total_size", esp_err_to_name(ret));
84
85 for (ssize_t size = 0, recv_size = 0; recv_size < total_size; recv_size += size) {
86 size = esp_http_client_read(client, (char *)data, OTA_DATA_PAYLOAD_LEN);
87 ESP_ERROR_GOTO(size < 0, EXIT, "<%s> Read data from http stream", esp_err_to_name(ret));
88
89 if (size > 0) {
90 /**< Write OTA update data to partition */
91 ret = esp_ota_write(ota_handle, data, OTA_DATA_PAYLOAD_LEN);
92 ESP_ERROR_GOTO(ret != ESP_OK, EXIT, "<%s> Write firmware to flash, size: %u, data: %.*s",
93 esp_err_to_name(ret), size, size, data);
94 } else {
95 ESP_LOGW(TAG, "<%s> esp_http_client_read", esp_err_to_name(ret));
96 goto EXIT;
97 }
98 }
99
100 ESP_LOGI(TAG, "The service download firmware is complete, Spend time: %" PRIu32 "s",
101 (xTaskGetTickCount() - start_time) * portTICK_PERIOD_MS / 1000);
102
103 ret = esp_ota_end(ota_handle);
104 ESP_ERROR_GOTO(ret != ESP_OK, EXIT, "<%s> esp_ota_end", esp_err_to_name(ret));
105
106EXIT:
107 ESP_FREE(data);
108 esp_http_client_close(client);
109 esp_http_client_cleanup(client);
110
111 return total_size;
112}

125static void app_firmware_send(size_t firmware_size, uint8_t sha[ESPNOW_OTA_HASH_LEN])
126{
127 esp_err_t ret = ESP_OK;
128 uint32_t start_time = xTaskGetTickCount();
129 espnow_ota_result_t espnow_ota_result = {0};
130 espnow_ota_responder_t *info_list = NULL;
131 espnow_addr_t *dest_addr_list = NULL;
132 size_t num = 0;
133
134 espnow_ota_initiator_scan(&info_list, &num, pdMS_TO_TICKS(3000));
135 ESP_LOGW(TAG, "espnow wait ota num: %u", num);
136
137 if (!num) {
138 goto EXIT;
139 }
140
141 dest_addr_list = ESP_MALLOC(num * ESPNOW_ADDR_LEN);
142
143 for (size_t i = 0; i < num; i++) {
144 memcpy(dest_addr_list[i], info_list[i].mac, ESPNOW_ADDR_LEN);
145 }
146
147 espnow_ota_initiator_scan_result_free();
148
149 ret = espnow_ota_initiator_send(dest_addr_list, num, sha, firmware_size, app_ota_initiator_data_cb, &espnow_ota_result);
150 ESP_ERROR_GOTO(ret != ESP_OK, EXIT, "<%s> espnow_ota_initiator_send", esp_err_to_name(ret));
151
152 if (espnow_ota_result.successed_num == 0) {
153 ESP_LOGW(TAG, "Devices upgrade failed, unfinished_num: %u", espnow_ota_result.unfinished_num);
154 goto EXIT;
155 }
156
157 ESP_LOGI(TAG, "Firmware is sent to the device to complete, Spend time: %" PRIu32 "s",
158 (xTaskGetTickCount() - start_time) * portTICK_PERIOD_MS / 1000);
159 ESP_LOGI(TAG, "Devices upgrade completed, successed_num: %u, unfinished_num: %u",
160 espnow_ota_result.successed_num, espnow_ota_result.unfinished_num);
161
162EXIT:
163 ESP_FREE(dest_addr_list);
164 espnow_ota_initiator_result_free(&espnow_ota_result);
165}
