61/**
62 * @brief A function which implements `command` command.
63 */
64static int command_func(int argc, char **argv)
65{
66 if (arg_parse(argc, argv, (void **) &command_args) != ESP_OK) {
67 arg_print_errors(stderr, command_args.end, argv[0]);
68 return ESP_FAIL;
69 }
70
71 espnow_frame_head_t frame_head = {
72 .filter_adjacent_channel = true,
73 .security = CONFIG_ESPNOW_DEBUG_SECURITY,
74 };
75
76 esp_err_t ret = ESP_OK;
77 size_t addrs_num = 0;
78 espnow_addr_t *addr_list = ESP_MALLOC(sizeof(espnow_addr_t));
79
80 for (const char *tmp = command_args.addr->sval[0];; tmp++) {
81 if (*tmp == ',' || *tmp == ' ' || *tmp == '|' || *tmp == '.' || *tmp == '\0') {
82 espnow_mac_str2hex(tmp - 17, addr_list[addrs_num]);
83 addrs_num++;
84
85 if (*tmp == '\0' || *(tmp + 1) == '\0') {
86 break;
87 }
88
89 addr_list = ESP_REALLOC(addr_list, sizeof(espnow_addr_t) * (addrs_num + 1));
90 }
91 }
92
93 ESP_ERROR_RETURN(addrs_num <= 0, ESP_ERR_INVALID_ARG,
94 "The format of the address is incorrect. Please enter the format as xx:xx:xx:xx:xx:xx");
95
96 if (command_args.channel_all->count) {
97 frame_head.channel = ESPNOW_CHANNEL_ALL;
98 frame_head.filter_adjacent_channel = false;
99 }
100
101 if (addrs_num == 1 && ESPNOW_ADDR_IS_BROADCAST(addr_list[0])) {
102 frame_head.broadcast = true;
103 frame_head.retransmit_count = ESPNOW_RETRANSMIT_MAX_COUNT;
104 frame_head.forward_rssi = -25;
105 frame_head.forward_ttl = 1;
106
107 ret = espnow_send(ESPNOW_DATA_TYPE_DEBUG_COMMAND, addr_list[0], command_args.command->sval[0],
108 strlen(command_args.command->sval[0]) + 1, &frame_head, portMAX_DELAY);
109 ESP_ERROR_RETURN(ret != ESP_OK, ret, "espnow_send");
110 } else if(addrs_num < 8) {
111 for(int i = 0; i < addrs_num; ++i) {
112 espnow_add_peer(addr_list[i], NULL);
113 ret = espnow_send(ESPNOW_DATA_TYPE_DEBUG_COMMAND, addr_list[i], command_args.command->sval[0],
114 strlen(command_args.command->sval[0]) + 1, &frame_head, portMAX_DELAY);
115 ESP_ERROR_RETURN(ret != ESP_OK, ret, "espnow_send");
116 espnow_del_peer(addr_list[i]);
117 }
118 } else {
119 espnow_addr_t temp_group = {0x0};
120 frame_head.group = 1;
121 frame_head.broadcast = true;
122 frame_head.retransmit_count = ESPNOW_RETRANSMIT_MAX_COUNT;
123 frame_head.forward_rssi = -25;
124 frame_head.forward_ttl = 1;
125
126 esp_fill_random(temp_group, sizeof(espnow_addr_t));
127 espnow_set_group(addr_list, addrs_num, temp_group, &frame_head, true, portMAX_DELAY);
128 ret = espnow_send(ESPNOW_DATA_TYPE_DEBUG_COMMAND, temp_group, command_args.command->sval[0],
129 strlen(command_args.command->sval[0]) + 1, &frame_head, portMAX_DELAY);
130 ESP_ERROR_RETURN(ret != ESP_OK, ret, "espnow_send");
131 espnow_set_group(addr_list, addrs_num, temp_group, &frame_head, false, portMAX_DELAY);
132 }
133
134 ESP_FREE(addr_list);
135
136 return ESP_OK;
137}

181 .retransmit_count = ESPNOW_RETRANSMIT_MAX_COUNT,

199 uint8_t primary = 0;
200 wifi_second_chan_t second = 0;
201 static wifi_country_t country = {0};

204 esp_wifi_get_country(&country);
205
206 for (int i = 0; i < country.nchan; ++i) {
207 esp_wifi_set_channel(country.schan + i, WIFI_SECOND_CHAN_NONE);
208 frame_head.channel = country.schan + i;

211 esp_err_t ret = espnow_send(ESPNOW_DATA_TYPE_DEBUG_COMMAND, addr,
212 data, strlen(data) + 1, &frame_head, portMAX_DELAY);
213 ESP_ERROR_RETURN(ret != ESP_OK, ret, "espnow_send");
214 vTaskDelay(100);
215 }
216
217 /**< Waiting to receive the response message */
218 }
219
220 esp_wifi_set_channel(primary, second);
221 } else {
222 esp_err_t ret = espnow_send(ESPNOW_DATA_TYPE_DEBUG_COMMAND, addr,
223 data, strlen(data) + 1, &frame_head, portMAX_DELAY);
224 ESP_ERROR_RETURN(ret != ESP_OK, ret, "espnow_send");
225 }
226
227 return ESP_OK;
228}

425static int control_func(int argc, char **argv)
426{
427 if (arg_parse(argc, argv, (void **) &control_args) != ESP_OK) {
428 arg_print_errors(stderr, control_args.end, argv[0]);

440
441 if (control_args.ack->count) {
442 frame_head.ack = true;
443 }
444
445 if (control_args.broadcast->count) {
446 frame_head.broadcast = control_args.broadcast->ival[0];

448

461 if (control_args.forward_rssi->count) {
462 frame_head.forward_rssi = control_args.forward_rssi->ival[0];
463 }
464
465 if (control_args.command->count) {
466

474 .responder_attribute = control_args.responder_attribute->ival[0],

482
483 if (control_args.mac->count) {
484 espnow_mac_str2hex(control_args.mac->sval[0], dest_addr);

486

500 ESP_ERROR_RETURN(ret != ESP_OK, ret, "espnow_ctrl_initiator_bind");
501 }
502

505 espnow_ctrl_responder_get_bindlist(NULL, &size);
506
507 if (size > 0) {
508 espnow_ctrl_bind_info_t *list = ESP_CALLOC(size, sizeof(espnow_ctrl_bind_info_t));
509 espnow_ctrl_responder_get_bindlist(list, &size);

511 for (int i = 0; i < size; ++i) {
512 ESP_LOGI("control_func", "mac: "MACSTR", initiator_type: %d, initiator_value: %d",

518

520 }

522 return ESP_OK;
523}

599 /**
600 * @brief First, the firmware is obtained from the http server and stored on the root node.
601 */
602 do {
603 ret = esp_http_client_open(client, 0);
604
605 if (ret != ESP_OK) {
606 vTaskDelay(pdMS_TO_TICKS(1000));
607 ESP_LOGW(TAG, "<%s> Connection service failed", esp_err_to_name(ret));
608 }
609 } while (ret != ESP_OK);
610
611 total_size = esp_http_client_fetch_headers(client);
612 sscanf(url, "%*[^/]//%*[^/]/%[^.bin]", name);
613
614 if (total_size <= 0) {
615 ESP_LOGW(TAG, "Please check the address of the server");
616 ret = esp_http_client_read(client, (char *)data, OTA_DATA_PAYLOAD_LEN);
617 ESP_ERROR_GOTO(ret < 0, EXIT, "<%s> Read data from http stream", esp_err_to_name(ret));
618
619 ESP_LOGW(TAG, "Recv data: %.*s", ret, data);
620 goto EXIT;
621 }
622
623 /**< Get partition info of currently running app
624 Return the next OTA app partition which should be written with a new firmware.*/
625 const esp_partition_t *running = esp_ota_get_running_partition();
626 g_ota_data_partition = esp_ota_get_next_update_partition(NULL);
627
628 ESP_ERROR_RETURN(!running || !g_ota_data_partition, ESP_ERR_ESPNOW_OTA_FIRMWARE_PARTITION,
629 "No partition is found or flash read operation failed");
630
631 ESP_LOGI(TAG, "Running partition, label: %s, type: 0x%x, subtype: 0x%x, address: 0x%x",
632 running->label, running->type, running->subtype, running->address);
633 ESP_LOGI(TAG, "Update partition, label: %s, type: 0x%x, subtype: 0x%x, address: 0x%x",
634 g_ota_data_partition->label, g_ota_data_partition->type, g_ota_data_partition->subtype, g_ota_data_partition->address);
635
636 /**< Commence an OTA update writing to the specified partition. */
637 ret = esp_ota_begin(g_ota_data_partition, total_size, &ota_handle);
638 ESP_ERROR_GOTO(ret != ESP_OK, EXIT, "<%s> esp_ota_begin failed, total_size", esp_err_to_name(ret));
639
640 /**
641 * @brief 3. Read firmware from the server and write it to the flash of the root node
642 */
643 for (ssize_t size = 0, recv_size = 0, i = 0; recv_size < total_size; recv_size += size, ++i) {
644 size = esp_http_client_read(client, (char *)data, OTA_DATA_PAYLOAD_LEN);
645 ESP_ERROR_GOTO(size < 0, EXIT, "<%s> Read data from http stream", esp_err_to_name(ret));
646
647 if (size > 0) {
648 /**< Write OTA update data to partition */
649 ret = esp_ota_write(ota_handle, data, size);
650 ESP_ERROR_GOTO(ret != ESP_OK, EXIT, "<%s> Write firmware to flash, size: %d, data: %.*s",
651 esp_err_to_name(ret), size, size, data);
652
653 } else {
654 ESP_LOGW(TAG, "<%s> esp_http_client_read", esp_err_to_name(ret));
655 goto EXIT;
656 }
657
658 if (i % 100 == 0 || recv_size == total_size) {
659 ESP_LOGI(TAG, "Firmware download size: %d, progress rate: %d%%",
660 recv_size, recv_size * 100 / total_size);
661 }
662 }
663
664 ESP_LOGI(TAG, "The service download firmware is complete, total_size: %d Spend time: %ds",
665 total_size, (xTaskGetTickCount() - start_time) * portTICK_PERIOD_MS / 1000);
666
667 g_ota_size = total_size;
668 espnow_storage_set("binary_len", &total_size, sizeof(size_t));
669
670 ret = esp_ota_end(ota_handle);
671 ESP_ERROR_GOTO(ret != ESP_OK, EXIT, "<%s> esp_ota_end", esp_err_to_name(ret));
672
673EXIT:
674 ESP_FREE(data);
675 return ESP_OK;
676}
