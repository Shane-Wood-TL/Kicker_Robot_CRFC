29esp_err_t espnow_timesync_start()
30{
31 if (sntp_enabled()) {
32 ESP_LOGI(TAG, "SNTP already initialized.");
33 g_init_done = true;
34 return ESP_OK;
35 }
36
37 const char *sntp_server_name = CONFIG_ESP_SNTP_SERVER_NAME;
38
39 ESP_LOGI(TAG, "Initializing SNTP. Using the SNTP server: %s", sntp_server_name);
40 sntp_setoperatingmode(SNTP_OPMODE_POLL);
41 sntp_setservername(0, sntp_server_name);
42
43 sntp_init();
44 g_init_done = true;
45
46 return ESP_OK;
47}

49bool espnow_timesync_check(void)
50{
51 time_t now;
52 time(&now);
53
54 if (now > REF_TIME) {
55 return true;
56 }

61esp_err_t espnow_timesync_wait(TickType_t ticks_to_wait)
62{
63 if (!g_init_done) {
64 ESP_LOGW(TAG, "Time sync not initialised using 'espnow_timesync_start'");
65 }
66
67 ESP_LOGW(TAG, "Waiting for time to be synchronized. This may take time.");
68 uint32_t ticks_remaining = ticks_to_wait;
69 uint32_t ticks = DEFAULT_TICKS;
70
71 while (ticks_remaining > 0) {
72 if (espnow_timesync_check() == true) {
73 break;
74 }
75
76 ESP_LOGD(TAG, "Time not synchronized yet. Retrying...");
77 ticks = ticks_remaining < DEFAULT_TICKS ? ticks_remaining : DEFAULT_TICKS;
78 ticks_remaining -= ticks;
79 vTaskDelay(ticks);
80 }
81
82 /* Check if ticks_to_wait expired and time is not synchronized yet. */
83 if (espnow_timesync_check() == false) {
84 ESP_LOGE(TAG, "Time not synchronized within the provided ticks: %u", ticks_to_wait);
85 return ESP_FAIL;
86 }
87
88 /* Get current time */
89 struct tm timeinfo;
90 char strftime_buf[64];
91 time_t now;
92 time(&now);
93 localtime_r(&now, &timeinfo);
94 strftime(strftime_buf, sizeof(strftime_buf), "%c", &timeinfo);
95 ESP_LOGI(TAG, "The current UTC time is: %s", strftime_buf);
96 return ESP_OK;
97}
