49static esp_err_t validate_image_header(const esp_partition_t *update)
50{
51 esp_app_desc_t new_app_info;
52 esp_app_desc_t running_app_info;
53 const esp_partition_t *running = esp_ota_get_running_partition();
54
55 if (esp_ota_get_partition_description(running, &running_app_info) == ESP_OK) {
56 ESP_LOGI(TAG, "Running firmware version: %s", running_app_info.version);
57 }
58
59 if (esp_ota_get_partition_description(update, &new_app_info) == ESP_OK) {
60 ESP_LOGI(TAG, "Running firmware version: %s", new_app_info.version);
61 }
62
63 if (!g_espnow_ota_config->skip_version_check
64 && !memcmp(new_app_info.version, running_app_info.version, sizeof(new_app_info.version))) {
65 ESP_LOGW(TAG, "Current running version is the same as a new. We will not continue the update.");
66 return ESP_FAIL;
67 }
68
69#ifdef CONFIG_BOOTLOADER_APP_ANTI_ROLLBACK
70 /**
71 * Secure version check from firmware image header prevents subsequent download and flash write of
72 * entire firmware image. However this is optional because it is also taken care in API
73 * esp_https_ota_finish at the end of OTA update procedure.
74 */
75 const uint32_t hw_sec_version = esp_efuse_read_secure_version();
76
77 if (new_app_info.secure_version < hw_sec_version) {
78 ESP_LOGW(TAG, "New firmware security version is less than eFuse programmed, %d < %d", new_app_info.secure_version, hw_sec_version);
79 return ESP_FAIL;
80 }
81
82#endif
83
84 return ESP_OK;
85}

88{
89 esp_err_t ret = ESP_OK;
90 /**< Remove useless data, sha256 of elf file (32 Byte) + reserv2 (20 Byte)*/
91 size_t size = sizeof(espnow_ota_info_t) - 32 - 20;
92 espnow_ota_info_t *info = ESP_MALLOC(sizeof(espnow_ota_info_t));

118static esp_err_t espnow_ota_status_handle(const espnow_addr_t src_addr, const espnow_ota_status_t *status, size_t size)
119{
120 ESP_PARAM_CHECK(src_addr);
121 ESP_PARAM_CHECK(status);
122 ESP_PARAM_CHECK(size);
123
124 esp_err_t ret = ESP_ERR_NO_MEM;
125 size_t response_size = sizeof(espnow_ota_status_t);
126 uint8_t running_sha_256[32] = {0};
127
128 if (!g_ota_config) {
129 size_t config_size = sizeof(ota_config_t) + ESPNOW_OTA_PROGRESS_MAX_SIZE * 10;
130 g_ota_config = ESP_CALLOC(1, config_size);
131 ESP_ERROR_GOTO(!g_ota_config, EXIT, "<ESP_ERR_NO_MEM> g_ota_config");
132
133 espnow_storage_get(ESPNOW_OTA_STORE_CONFIG_KEY, g_ota_config, 0);
134
135 g_ota_config->start_time = xTaskGetTickCount();
136 g_ota_config->partition = esp_ota_get_next_update_partition(NULL);
137 }
138
139 g_ota_config->status.type = ESPNOW_OTA_TYPE_STATUS;
140 g_ota_config->status.error_code = ESP_OK;
141
142 ret = esp_partition_get_sha256(esp_ota_get_running_partition(), running_sha_256);
143 ESP_ERROR_RETURN(ret != ESP_OK, ret, "esp_partition_get_sha256");
144
145 if (!memcmp(running_sha_256, status->sha_256, ESPNOW_OTA_HASH_LEN)) {
146 ESP_LOGW(TAG, "The firmware to be upgraded is the same as the currently running firmware, so no upgrade");
147 g_ota_config->status.error_code = ESP_ERR_ESPNOW_OTA_FINISH;
148 g_ota_config->status.written_size = 0;
149 goto EXIT;
150 }
151
152 /**< If g_ota_config->status has been created and
153 once again upgrade the same name bin, just return ESP_OK */
154 if (!memcmp(g_ota_config->status.sha_256, status->sha_256, ESPNOW_OTA_HASH_LEN)
155 && g_ota_config->status.total_size == status->total_size) {
156 ret = ESP_OK;
157 goto EXIT;
158 }
159
160 memset(g_ota_config, 0, sizeof(ota_config_t));
161 memcpy(&g_ota_config->status, status, sizeof(espnow_ota_status_t));
162 memset(g_ota_config->status.progress_array, 0, status->packet_num / 8 + 1);
163 g_ota_config->status.written_size = 0;
164 g_ota_config->status.error_code = ESP_ERR_ESPNOW_OTA_FIRMWARE_NOT_INIT;
165
166 ret = espnow_send(ESPNOW_DATA_TYPE_OTA_STATUS, src_addr, &g_ota_config->status,
167 sizeof(espnow_ota_status_t), &g_frame_config, portMAX_DELAY);
168 ESP_ERROR_RETURN(ret != ESP_OK, ret, "espnow_write");
169
170 ESP_LOGI(TAG, "The device starts to upgrade");
171
172 g_ota_finished_flag = false;
173 /**< Get partition info of currently running app
174 Return the next OTA app partition which should be written with a new firmware.*/
175 const esp_partition_t *running = esp_ota_get_running_partition();
176 const esp_partition_t *update = esp_ota_get_next_update_partition(NULL);
177
178 ret = ESP_ERR_ESPNOW_OTA_FIRMWARE_PARTITION;
179 ESP_ERROR_GOTO(!running || !update, EXIT,
180 "No partition is found or flash read operation failed");
181 ESP_LOGD(TAG, "Running partition, label: %s, type: 0x%x, subtype: 0x%x, address: 0x%x",
182 running->label, running->type, running->subtype, running->address);
183 ESP_LOGD(TAG, "Update partition, label: %s, type: 0x%x, subtype: 0x%x, address: 0x%x",
184 update->label, update->type, update->subtype, update->address);
185
186 g_ota_config->partition = update;
187 g_ota_config->start_time = xTaskGetTickCount();
188
189 /**< Commence an OTA update writing to the specified partition. */
190 ret = esp_ota_begin(update, g_ota_config->status.total_size, &g_ota_config->handle);
191 ESP_ERROR_GOTO(ret != ESP_OK, EXIT, "esp_ota_begin failed");
192
193 /**< Save upgrade infomation to flash. */
194 ret = espnow_storage_set(ESPNOW_OTA_STORE_CONFIG_KEY, g_ota_config,
195 sizeof(ota_config_t) + g_ota_config->status.packet_num / 8 + 1);
196 ESP_ERROR_GOTO(ret != ESP_OK, EXIT, "info_store_save, ret: %d", ret);
197
198 /**< Send ESP_EVENT_ESPNOW_OTA_STARTED event to the event handler */
199 esp_event_post(ESP_EVENT_ESPNOW, ESP_EVENT_ESPNOW_OTA_STARTED, NULL, 0, 0);
200
201 g_ota_config->status.error_code = ESP_OK;
202
203EXIT:
204
205 /**< Update g_ota_config->status */
206 if (g_ota_config->status.written_size
207 && g_ota_config->status.written_size != g_ota_config->status.total_size) {
208 espnow_ota_status_t *tmp_status = ESP_MALLOC(sizeof(espnow_ota_status_t) + ESPNOW_OTA_PROGRESS_MAX_SIZE);
209 memcpy(tmp_status, &g_ota_config->status, sizeof(espnow_ota_status_t));
210 espnow_frame_head_t frame_head = ESPNOW_FRAME_CONFIG_DEFAULT();
211 frame_head.security = CONFIG_ESPNOW_OTA_SECURITY;
212
213 for (int seq = 0; seq < tmp_status->packet_num; ++seq) {
214 if (!ESPNOW_OTA_GET_BITS(g_ota_config->status.progress_array, seq)) {
215 tmp_status->progress_index = seq / (ESPNOW_OTA_PROGRESS_MAX_SIZE * 8);
216 memcpy(tmp_status->progress_array[0], g_ota_config->status.progress_array[tmp_status->progress_index], ESPNOW_OTA_PROGRESS_MAX_SIZE);
217 ret = espnow_send(ESPNOW_DATA_TYPE_OTA_STATUS, src_addr, tmp_status,
218 sizeof(espnow_ota_status_t) + ESPNOW_OTA_PROGRESS_MAX_SIZE, &frame_head, portMAX_DELAY);
219 ESP_LOG_BUFFER_HEXDUMP(TAG, tmp_status->progress_array, ESPNOW_OTA_PROGRESS_MAX_SIZE, ESP_LOG_DEBUG);
220 ESP_ERROR_BREAK(ret != ESP_OK, "espnow_send");
221
222 ESP_LOGD(TAG, "status, total_size: %d, written_size: %d, progress_index: %d",
223 tmp_status->total_size, tmp_status->written_size, tmp_status->progress_index);
224 break;
225 }
226 }
227
228 ESP_FREE(tmp_status);
229 return ret;
230 }
231
232 if (ret != ESP_OK) {
233 g_ota_config->status.error_code = ret;
234 }
235
236 ESP_LOGD(TAG, "Response upgrade status, written_size: %d, response_size: %d, addr: " MACSTR,
237 g_ota_config->status.written_size, response_size, MAC2STR(src_addr));
238 ret = espnow_send(ESPNOW_DATA_TYPE_OTA_STATUS, src_addr, &g_ota_config->status,
239 sizeof(espnow_ota_status_t), &g_frame_config, portMAX_DELAY);
240 ESP_ERROR_RETURN(ret != ESP_OK, ret, "espnow_write");
241
242 return ESP_OK;
243}

245static esp_err_t espnow_ota_write(const espnow_addr_t src_addr, const espnow_ota_packet_t *packet, size_t size)
246{
247 ESP_PARAM_CHECK(src_addr);
248 ESP_PARAM_CHECK(packet);
249 ESP_PARAM_CHECK(size);
250
251 esp_err_t ret = ESP_OK;
252
253 /* Return if not recv status or firmware has been upgraded */
254 if (!g_ota_config || g_ota_config->status.error_code == ESP_ERR_ESPNOW_OTA_FINISH) {
255 return ESP_OK;
256 }
257
258 if (g_ota_config->status.error_code == ESP_ERR_ESPNOW_OTA_STOP) {
259 g_ota_config->status.type = ESPNOW_OTA_TYPE_STATUS;
260 g_ota_config->status.written_size = 0;
261 memset(g_ota_config->status.progress_array, 0, g_ota_config->status.packet_num / 8 + 1);
262 espnow_storage_erase(ESPNOW_OTA_STORE_CONFIG_KEY);
263
264 ret = espnow_send(ESPNOW_DATA_TYPE_OTA_STATUS, src_addr, &g_ota_config->status,
265 sizeof(espnow_ota_status_t), &g_frame_config, portMAX_DELAY);
266 ESP_ERROR_RETURN(ret != ESP_OK, ret, "espnow_write");
267
268 return ESP_OK;
269 }
270
271 ESP_ERROR_RETURN(packet->seq * ESPNOW_OTA_PACKET_MAX_SIZE > g_ota_config->status.total_size,
272 ESP_ERR_INVALID_ARG, "packet->seq: %d", packet->seq);
273
274 /**< Received a duplicate packet */
275 if (ESPNOW_OTA_GET_BITS(g_ota_config->status.progress_array, packet->seq)) {
276 ESP_LOGD(TAG, "Received a duplicate packet, packet_seq: %d", packet->seq);
277 return ESP_OK;
278 }
279
280 /**< Write firmware data to the update partition */
281 ret = esp_partition_write(g_ota_config->partition, packet->seq * ESPNOW_OTA_PACKET_MAX_SIZE,
282 packet->data, packet->size);
283 ESP_ERROR_RETURN(ret != ESP_OK, ESP_ERR_ESPNOW_OTA_FIRMWARE_DOWNLOAD,
284 "esp_partition_write %s", esp_err_to_name(ret));
285
286 ESPNOW_OTA_SET_BITS(g_ota_config->status.progress_array, packet->seq);
287 g_ota_config->status.written_size += packet->size;
288
289 /**< Save OTA status periodically, it can be used to
290 resumable data transfers from breakpoint after system reset */
291 static uint32_t s_next_written_percentage = 0;
292 if (g_espnow_ota_config->progress_report_interval) {
293 uint32_t written_percentage = g_ota_config->status.written_size * 100 / g_ota_config->status.total_size;
294
295 if (!s_next_written_percentage) {
296 s_next_written_percentage = g_espnow_ota_config->progress_report_interval;
297 }
298
299 ESP_LOGD(TAG, "packet_seq: %d, packet_size: %d, written_size: %d, progress: %03d%%, next_percentage: %03d%%",
300 packet->seq, packet->size, g_ota_config->status.written_size, written_percentage, s_next_written_percentage);
301
302 if (written_percentage == s_next_written_percentage) {
303 ESP_LOGD(TAG, "Save the data of upgrade status to flash");
304 s_next_written_percentage += g_espnow_ota_config->progress_report_interval;
305
306 espnow_storage_set(ESPNOW_OTA_STORE_CONFIG_KEY, g_ota_config,
307 sizeof(ota_config_t) + g_ota_config->status.packet_num / 8 + 1);
308
309 /**< Send ESP_EVENT_ESPNOW_OTA_STATUS event to the event handler */
310 esp_event_post(ESP_EVENT_ESPNOW, ESP_EVENT_ESPNOW_OTA_STATUS, &written_percentage, sizeof(uint32_t), 0);
311
312 ESP_LOGD(TAG, "packet_seq: %d, packet_size: %d, written_size: %d, progress: %d%%",
313 packet->seq, packet->size, g_ota_config->status.written_size, written_percentage);
314 } else if (written_percentage > s_next_written_percentage) {
315 s_next_written_percentage = (written_percentage / g_espnow_ota_config->progress_report_interval + 1) * g_espnow_ota_config->progress_report_interval;
316 }
317 }
318
319 if (g_ota_config->status.written_size == g_ota_config->status.total_size) {
320 s_next_written_percentage = 0;
321 ESP_LOG_BUFFER_CHAR_LEVEL(TAG, g_ota_config->status.progress_array,
322 ESPNOW_OTA_PROGRESS_MAX_SIZE, ESP_LOG_VERBOSE);
323 ESP_LOGI(TAG, "Write total_size: %d, written_size: %d, spend time: %ds",
324 g_ota_config->status.total_size, g_ota_config->status.written_size,
325 (xTaskGetTickCount() - g_ota_config->start_time) * portTICK_PERIOD_MS / 1000);
326
327 /**< If ESP32 was reset duration OTA, and after restart, the update_handle will be invalid,
328 but it still can switch boot partition and reboot successful */
329 esp_ota_end(g_ota_config->handle);
330 espnow_storage_erase(ESPNOW_OTA_STORE_CONFIG_KEY);
331
332 const esp_partition_t *update_partition = esp_ota_get_next_update_partition(NULL);
333
334 ret = validate_image_header(update_partition);
335 ESP_ERROR_RETURN(ret != ESP_OK, ret, "validate_image_header");
336
337 ret = esp_ota_set_boot_partition(update_partition);
338
339 if (ret != ESP_OK) {
340 g_ota_config->status.written_size = 0;
341 g_ota_config->status.error_code = ESP_ERR_ESPNOW_OTA_STOP;
342 ESP_LOGW(TAG, "<%s> esp_ota_set_boot_partition", esp_err_to_name(ret));
343 return ret;
344 }
345
346 /**< Send ESP_EVENT_ESPNOW_OTA_FINISH event to the event handler */
347 g_ota_finished_flag = true;
348 g_ota_config->status.type = ESPNOW_OTA_TYPE_STATUS;
349
350 /**< Response firmware upgrade status to root node. */
351 ret = espnow_send(ESPNOW_DATA_TYPE_OTA_STATUS, src_addr, &g_ota_config->status,
352 sizeof(espnow_ota_status_t), &g_frame_config, portMAX_DELAY);
353 ESP_ERROR_RETURN(ret != ESP_OK, ret, "espnow_send");
354
355 esp_event_post(ESP_EVENT_ESPNOW, ESP_EVENT_ESPNOW_OTA_FINISH, NULL, 0, 0);
356 }
357
358 return ESP_OK;
359}

371esp_err_t espnow_ota_responder_stop()
372{
373 esp_err_t ret = ESP_OK;
374
375 if (!g_ota_config) {
376 return ESP_OK;
377 }
378
379 if (g_ota_finished_flag) {
380 const esp_partition_t *running = esp_ota_get_running_partition();
381 ret = esp_ota_set_boot_partition(running);
382 ESP_ERROR_RETURN(ret != ESP_OK, ret, "esp_ota_set_boot_partition");
383 }
384
385 g_ota_config->status.type = ESPNOW_OTA_TYPE_STATUS;
386 g_ota_config->status.error_code = ESP_ERR_ESPNOW_OTA_STOP;
387 g_ota_config->status.written_size = 0;
388 memset(g_ota_config->status.progress_array, 0, g_ota_config->status.packet_num / 8 + 1);
389 espnow_storage_erase(ESPNOW_OTA_STORE_CONFIG_KEY);
390 espnow_frame_head_t frame_head = ESPNOW_FRAME_CONFIG_DEFAULT();
391 frame_head.security = CONFIG_ESPNOW_OTA_SECURITY;
392
393 ret = espnow_send(ESPNOW_DATA_TYPE_OTA_STATUS, ESPNOW_ADDR_BROADCAST,
394 &g_ota_config->status, sizeof(espnow_ota_status_t), &frame_head, portMAX_DELAY);
395 ESP_ERROR_RETURN(ret != ESP_OK, ret, "espnow_write");
396
397 return ESP_OK;

400static esp_err_t espnow_ota_responder_data_process(uint8_t *src_addr, void *data,
401 size_t size, wifi_pkt_rx_ctrl_t *rx_ctrl)
402{
403 ESP_PARAM_CHECK(src_addr);
404 ESP_PARAM_CHECK(data);
405 ESP_PARAM_CHECK(size);
406 ESP_PARAM_CHECK(rx_ctrl);
407
408 esp_err_t ret = ESP_OK;
409 uint8_t data_type = ((uint8_t *)data)[0];
410 espnow_add_peer(src_addr, NULL);
411
412 switch (data_type) {
413 case ESPNOW_OTA_TYPE_REQUEST:
414 ESP_LOGD(TAG, "ESPNOW_OTA_TYPE_INFO");
415 ret = espnow_ota_info(src_addr);
416 break;
417
418 case ESPNOW_OTA_TYPE_STATUS:
419 ESP_LOGD(TAG, "ESPNOW_OTA_TYPE_STATUS");
420 ret = espnow_ota_status_handle(src_addr, (espnow_ota_status_t *)data, size);
421 break;
422
423 case ESPNOW_OTA_TYPE_DATA:
424 ESP_LOGD(TAG, "ESPNOW_OTA_TYPE_DATA");
425 ret = espnow_ota_write(src_addr, (espnow_ota_packet_t *)data, size);
426 break;
427
428 default:
429 break;
430 }
431
432 espnow_del_peer(src_addr);
433 ESP_ERROR_RETURN(ret != ESP_OK, ret, "espnow_ota_handle");
434 return ret;
435}
