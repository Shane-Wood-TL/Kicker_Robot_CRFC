42 if (motors_enabled == ENABLED) {

139void menu_handler::options_pressed() {
140 current_state++;
141 if (current_state > cyclic_menu_count) {
142 current_state = STATUS_SCREEN_STATE;
143 }
144 switch (current_state) {
145 case (STATUS_SCREEN_STATE): {
146 display->clear_buffer();
147 if (xSemaphoreTake(main_menu_values_mutex, portMAX_DELAY)) {
148 status_screen->draw_to_display();
149 xSemaphoreGive(main_menu_values_mutex);
150 }
151 display->write_buffer_SSD1306();
152 break;
153 }
154 case (MOTOR_SPEEDS_STATE): {
155 display->clear_buffer();
156 if (xSemaphoreTake(motor_speeds_mutex, portMAX_DELAY)) {
157 motor_speed_menu->draw_to_display();
158 xSemaphoreGive(motor_speeds_mutex);
159 }
160 display->write_buffer_SSD1306();
161 break;
162 }
163 case (RAMPED_VELOCITY_STATE): {
164 display->clear_buffer();
165 if (xSemaphoreTake(ramped_velocity_mutex, portMAX_DELAY)) {
166 ramped_velocity_menu->draw_to_display();
167 xSemaphoreGive(ramped_velocity_mutex);
168 }
169 display->write_buffer_SSD1306();
170 break;
171 }
172 case (NETWORK_CHAN): {
173 display->clear_buffer();
174 if (xSemaphoreTake(network_channel_mutex, portMAX_DELAY)) {
175 network_channel_selector->draw_to_display();
176 xSemaphoreGive(network_channel_mutex);
177 }
178 display->write_buffer_SSD1306();
179 break;
180 }
181 default: {
182 current_state = STATUS_SCREEN_STATE;
183 break;
184 }
185 }
186}

284void menu_handler::update() {
285 if (xSemaphoreTake(update_main_display_mutex, portMAX_DELAY)) {
286 if (update_main_display == true && current_state == STATUS_SCREEN_STATE) {
287 update_main_display = false;
288 xSemaphoreGive(update_main_display_mutex);
289 if (xSemaphoreTake(main_menu_values_mutex, portMAX_DELAY)) {
290 draw_status_display();
291 xSemaphoreGive(main_menu_values_mutex);
292 }
293
294 } else {
295 xSemaphoreGive(update_main_display_mutex);
296 }
297 }
298 if (xQueueReceive(other_controller_data_queue, &current_data, pdMS_TO_TICKS(10)) == pdTRUE) {
299 xQueueReset(other_controller_data_queue); //prevents multiple inputs
300 bool current_options_state = current_data.options_triggers & (1 << processed_options_bit);
301
302 bool current_r2_state = current_data.options_triggers & (1 << processed_r2_bit);
303 bool current_l2_state = current_data.options_triggers & (1 << processed_l2_bit);
304
305 bool current_square_state = (current_data.buttons & (1 << processed_square_bit));
306 bool current_triangle_state = (current_data.buttons & (1 << processed_triangle_bit));
307 bool current_circle_state = (current_data.buttons & (1 << processed_circle_bit));
308 bool current_x_state = (current_data.buttons & (1 << processed_x_bit));
309
310 bool current_dpad_up_state = current_data.buttons & (1 << processed_dpad_up_bit);
311 bool current_dpad_down_state = current_data.buttons & (1 << processed_dpad_down_bit);
312 bool current_dpad_left_state = current_data.buttons & (1 << processed_dpad_left_bit);
313 bool current_dpad_right_state = current_data.buttons & (1 << processed_dpad_right_bit);
314
315
316 if (triangle_state.update(current_triangle_state)) {
317 triangle_pressed();
318 } else if (square_state.update(current_square_state)) {
319 square_pressed();
320 } else if (circle_state.update(current_circle_state)) {
321 circle_pressed();
322 } else if ((r2_state.update(current_r2_state)) and (current_x_state)) {
323 servo_released();
324 } else if (l2_state.update(current_l2_state)) {
325 l2_pressed();
326 } else if (option_state.update(current_options_state)) {
327 options_pressed();
328 } else if (dpad_up_state.update(current_dpad_up_state)) {
329 dpad_up_pressed();
330 } else if (dpad_down_state.update(current_dpad_down_state)) {
331 dpad_down_pressed();
332 } else if (dpad_right_state.update(current_dpad_right_state)) {
333 dpad_right_pressed();
334 } else if (dpad_left_state.update(current_dpad_left_state)) {
335 dpad_left_pressed();
336 }
337 }
