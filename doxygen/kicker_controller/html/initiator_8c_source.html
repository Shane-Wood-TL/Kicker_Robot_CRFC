59 int32_t event_id, void *event_data)
60{

118esp_err_t app_espnow_prov_beacon_start(int32_t sec)
119{
120 wifi_config_t wifi_config = { 0 };
121 espnow_prov_wifi_t prov_wifi_config = { 0 };
122 espnow_prov_responder_t responder_info = {
123 .product_id = "responder_test"
124 };
125
126 esp_err_t ret = esp_wifi_get_config(WIFI_IF_STA, &wifi_config);
127 if (ret != ESP_OK) {

140 if (ret != ESP_OK) {
141 ESP_LOGE(TAG, "espnow_prov_responder_start failed, %d", ret);
142 }
143
144 return ret;

147static void app_espnow_prov_responder_task(void *arg)
148{
149 for (;;) {
150 vTaskDelay(pdMS_TO_TICKS(1000));
151
152 /* Waiting until either the connection is established (WIFI_CONNECTED_BIT) or connection failed for the maximum
153 * number of re-tries (WIFI_FAIL_BIT). The bits are set by event_handler() (see above) */
154 EventBits_t bits = xEventGroupWaitBits(s_wifi_event_group,
155 WIFI_CONNECTED_BIT | WIFI_FAIL_BIT,
156 pdFALSE,
157 pdFALSE,
158 portMAX_DELAY);
159
160 /* xEventGroupWaitBits() returns the bits before the call returned, hence we can test which event actually
161 * happened. */
162 if (bits & WIFI_CONNECTED_BIT) {
163 ESP_LOGD(TAG, "connected to ap");
164 } else if (bits & WIFI_FAIL_BIT) {
165 ESP_LOGD(TAG, "Failed to connect to ap");
166 continue;
167 } else {
168 ESP_LOGE(TAG, "UNEXPECTED EVENT");
169 continue;
170 }
171
172 if (app_espnow_prov_beacon_start(30) != ESP_OK) {
173 break;
174 }
175
176 // Clear WIFI_CONNECTED_BIT to restart responder when reconnection
177 xEventGroupClearBits(s_wifi_event_group, WIFI_CONNECTED_BIT);
178 }
179
180 ESP_LOGI(TAG, "provisioning responder exit");
181 vTaskDelete(NULL);
182}

189static void app_espnow_initiator_sec_task(void *arg)
190{
191 espnow_sec_result_t espnow_sec_result = {0};
192 espnow_sec_responder_t *info_list = NULL;
193 espnow_addr_t *dest_addr_list = NULL;
194 size_t num = 0;
195 uint8_t key_info[APP_KEY_LEN];
196
197 if (espnow_get_key(key_info) != ESP_OK) {
198 esp_fill_random(key_info, APP_KEY_LEN);
199 }
200 espnow_set_key(key_info);
201 espnow_set_dec_key(key_info);
202
203 uint32_t start_time1 = xTaskGetTickCount();
204 espnow_sec_initiator_scan(&info_list, &num, pdMS_TO_TICKS(3000));
205 ESP_LOGW(TAG, "espnow wait security num: %u", num);
206
207 if (num == 0) {
208 goto EXIT;
209 }
210
211 dest_addr_list = ESP_MALLOC(num * ESPNOW_ADDR_LEN);
212
213 for (size_t i = 0; i < num; i++) {
214 memcpy(dest_addr_list[i], info_list[i].mac, ESPNOW_ADDR_LEN);
215 }
216
217 espnow_sec_initiator_scan_result_free();
218
219 uint32_t start_time2 = xTaskGetTickCount();
220 esp_err_t ret = espnow_sec_initiator_start(key_info, pop_data, dest_addr_list, num, &espnow_sec_result);
221 ESP_ERROR_GOTO(ret != ESP_OK, EXIT, "<%s> espnow_sec_initiator_start", esp_err_to_name(ret));
222
223 ESP_LOGI(TAG, "App key is sent to the device to complete, Spend time: %" PRId32" ms, Scan time: %" PRId32 "ms",
224 (xTaskGetTickCount() - start_time1) * portTICK_PERIOD_MS,
225 (start_time2 - start_time1) * portTICK_PERIOD_MS);
226 ESP_LOGI(TAG, "Devices security completed, successed_num: %u, unfinished_num: %u",
227 espnow_sec_result.successed_num, espnow_sec_result.unfinished_num);
228
229EXIT:
230 ESP_FREE(dest_addr_list);
231 espnow_sec_initiator_result_free(&espnow_sec_result);
232
233 vTaskDelete(NULL);
234 s_sec_task = NULL;
235}
