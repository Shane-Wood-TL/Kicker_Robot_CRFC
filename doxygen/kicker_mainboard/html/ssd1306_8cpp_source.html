18void ssd1306::send_data_SSD1306(uint8_t *data, size_t len)
19{
20 uint8_t data_to_send[len+1] = {0};

56void ssd1306::clear_SSD1306()
57{
58 uint8_t emptyBuffer[SSD1306_horizontal_resolution] = {0}; // each page is 128 x 8
59 // display is 128 x 64, each page is 8 pixels high
60 for (uint8_t i = 0; i < (SSD1306_page_count); i++)
61 {
62 send_command_SSD1306(SSD1306_PAGE_OFFSET + i); // 0B0 - 0B7 are the 8 vertical pages
63
64 // sets the column address to 0
65 send_command_SSD1306(SSD1306_LOWER_COLUMN_START);
66 send_command_SSD1306(SSD1306_UPPER_COLUMN_START);
67
68 // write the empty buffer to the display
69 send_data_SSD1306(emptyBuffer, sizeof(emptyBuffer));
70 }
71}

75void ssd1306::draw_pixel_SSD1306(uint8_t x, uint8_t y)
76{
77 if ((x >= SSD1306_horizontal_resolution) || (y >= SSD1306_vertical_resolution)){
78 return; // Out of bounds
79 }
80 uint8_t pixel_in_page = (1 << (y % (SSD1306_page_height))); //limit the pixel from 0 to 7 with a 1 in the position 0000_0001, 0000_0010, etc
81 //buffer is indexed as such
82 // x straight across 0-127
83 //each one of these is 8 pixels bits tall, or 1 page
84 //repeat for pages 0-8
85 //leading to a total size of 128*8 = 1024 page "strips" each containing the data for 8 pixels
86 uint16_t buffer_index = uint16_t(x + ((y / (SSD1306_page_height)) * SSD1306_horizontal_resolution));
87 buffer[buffer_index] |= pixel_in_page; //write to the buffer based on page and column
88}

92void ssd1306::write_buffer_SSD1306()
93{
94 for (uint8_t i = 0; i < SSD1306_page_count; i++)
95 { // Send each page
96 send_command_SSD1306(SSD1306_PAGE_OFFSET + i); // Set page start address
97 send_command_SSD1306(SSD1306_LOWER_COLUMN_START); // Set lower column address
98 send_command_SSD1306(SSD1306_UPPER_COLUMN_START); // Set higher column address
99
100 send_data_SSD1306(&buffer[SSD1306_horizontal_resolution * i], SSD1306_horizontal_resolution);
101 }
102}

106void ssd1306::write_letter_SSD1306(uint8_t letter, uint8_t x_pos, uint8_t y_pos)
107{
108 const uint16_t *matrix = font_instance.get_letter(letter);//get the pixel data for one font char
