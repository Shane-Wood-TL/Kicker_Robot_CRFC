19void oDrive::send_message(const uint8_t message_ID, const uint8_t *message_contents, uint8_t message_length, const bool data_returned)
20{
21 // max data size is 8 bytes
22 if (message_length > max_message_size)
23 {
24 return;
25 }
26 if (message_contents == nullptr)
27 {
28 message_length = 0;
29 }
30
31 // setup can node + message ID with some bit manipulation to ensure the messageID is the least sigificant node_id_offset bits
32 uint32_t can_ID = (node_ID << node_id_offset) | (message_ID & lower_five_bits);
33
34 // setup the message to transmit
35 twai_message_t tx_message;
36 tx_message.identifier = can_ID;
37 (void)memcpy(tx_message.data, message_contents, message_length); // copy the message data
38 tx_message.extd = 0; // not a extended frame
39 tx_message.rtr = 0; // no remote transmission request by default
40 tx_message.data_length_code = message_length; // set the number of bytes in the message
41
42 twai_message_t rx_message;
43 if(data_returned){
44 //clear the receive buffer before sending the message
45 tx_message.rtr = 1; //data will be returned
46 while(twai_receive(&rx_message,0)== ESP_OK){
47
48 }
49 }
50 esp_err_t result = twai_transmit(&tx_message, pdMS_TO_TICKS(200));
51
52 // if the data was sent successfully, and a response is expected
53 if (data_returned && (result == ESP_OK))
54 {
55 receive_message(message_ID);
56 }
57 else if (data_returned && (result != ESP_OK))
58 {
59 // if the transmit failed, and data is expected, set the battery voltage to 0
60 // if the transmit failed the, then the receive will not be called
61 if (message_ID == Get_Bus_Voltage_Current)
62 {
63 clear_battery_voltage();
64 }
65 }
66}
