169template <typename T>
170void menu<T>::change_selected(int8_t direction){
171 if(direction != 1 && direction != -1 && direction != 0){
172 return;
173 }
174
175 if(!has_selectable_values){
176 return;
177 }
178 currently_selected -= direction;
179 if(currently_selected >= lines){
180 currently_selected = 0;
181 }else if(currently_selected < 0){
182 currently_selected = lines-1;
183 }
184 while(selectable_values[currently_selected] == false){
185 currently_selected -= direction;
186 if(currently_selected >= lines){
187 currently_selected = 0;
188 }else if(currently_selected < 0){
189 currently_selected = lines-1;
190 }
191 }
192 //draw an X denoting currently selected
193 display->write_string_SSD1306("X",0,font_rows[currently_selected]);
194}
