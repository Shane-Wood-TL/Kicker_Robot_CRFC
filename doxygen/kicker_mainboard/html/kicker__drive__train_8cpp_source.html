76void kicker_drive_train::motor_status_updater()
77{
78 if (xSemaphoreTake(motor_status_mutex, portMAX_DELAY))
79 {
80 if (motor_status != last_motor_state)
81 {
82 uint8_t current_motor_state = motor_status;
83 xSemaphoreGive(motor_status_mutex);
84 switch (current_motor_state)
85 {
86 case (ENABLED):
87 {
88 if (last_motor_state != ENABLED)
89 {
90 enable_motors();
91 }
92 break;
93 }
94 case (DISABLED):
95 {
96 if (last_motor_state != DISABLED)
97 {
98 break_motors();
99 //estop();
100 //disable_motors();
101 }
102 break;
103 }
104 case (ERRORLESS):
105 {
106 if (last_motor_state != ERRORLESS)
107 {
108 clear_errors();
109 }
110 break;
111 }
112 case (CALIBRATING):
113 {
114 if (last_motor_state != CALIBRATING)
115 {
116 calibrating();
117 disable_motors();
118 last_motor_state = DISABLED;
119 }
120 break;
121 }
122 }
123 last_motor_state = current_motor_state;
124 }
125 else
126 {
127 xSemaphoreGive(motor_status_mutex);
128 }
129 }

234void kicker_drive_train::drive_motors(bool boosted){
235 //create variables to hold the values to be sent to the motors
236 float current_drive_speed = 0.0f;
237 float current_turning_speed = 0.0f;
238 if(!boosted){
239 //normal operation
240 current_drive_speed = map(last_driving_speed, eight_bit_minimum, eight_bit_maximum, (-last_driving_speed_mult * motor_drive_speed_multiplier_value), (last_driving_speed_mult * motor_drive_speed_multiplier_value));
241
242 }else{
243 //boosted operation
244 current_drive_speed = map(last_driving_speed, eight_bit_minimum, eight_bit_maximum, ((-boost_amount-last_driving_speed_mult) * motor_drive_speed_multiplier_value), ((boost_amount+last_driving_speed_mult) * motor_drive_speed_multiplier_value));
245 }
246
247 //turning speed is always the same, regardless of boosted or not
248 current_turning_speed = map(last_turning_speed, eight_bit_minimum, eight_bit_maximum, (-last_turning_speed_mult * motor_turn_speed_multiplier_value), (last_turning_speed_mult * motor_turn_speed_multiplier_value));
249
250 //apply deadzone to the drive and turning speeds
251 if (abs(current_drive_speed) < (input_velocity_deadzone * last_driving_speed_mult))
252 {
253 current_drive_speed = 0;
254 }
255 if ((abs(current_turning_speed) < (input_velocity_deadzone * last_turning_speed_mult)))
256 {
257 current_turning_speed = 0;
258 }
259
260 //combine the turn and drive speeds
261 float current_left_motor_speed = -current_drive_speed + current_turning_speed;
262 float current_right_motor_speed = current_drive_speed + current_turning_speed; //right motor is inverted
263
264 // set the torque feedforward value to the same value for both motors
265 float Input_Torque_FF = odrive_motor_torque;
266 uint8_t vel_left_as_int[full_message_size] = {0};
267 uint8_t vel_right_as_int[full_message_size] = {0};
268
269 //set up the message to be sent to the motors
270 (void)memcpy(&vel_left_as_int[full_message_start_index], &current_left_motor_speed, sizeof(float));
271 (void)memcpy(&vel_left_as_int[full_message_middle_index], &Input_Torque_FF, sizeof(float));
272
273 (void)memcpy(&vel_right_as_int[full_message_start_index], &current_right_motor_speed, sizeof(float));
274 (void)memcpy(&vel_right_as_int[full_message_middle_index], &Input_Torque_FF, sizeof(float));
275
276 //send the message to the motors
277 left_drive->send_message(Set_Input_Vel, vel_left_as_int, eight_bytes, false);
278 right_drive->send_message(Set_Input_Vel, vel_right_as_int, eight_bytes, false);
279}
