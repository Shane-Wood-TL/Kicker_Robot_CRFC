65/**
66 * @enum motor_status_list
67 * @brief Contains the different types of states for the Odrive motor controllers
68 *
69 * @details motor_status_list has the following options: ENABLED, DISABLED, ERRORLESS, CALIBRATING, IDLE, BOOSTED
70 * ENABLED = motor working as normal
71 * DISABLED = motor set to idle (not by choice, this state happens when something wrong has occured, such as controller disconnect)
72 * ERRORLESS = clears motor errors
73 * CALIBRATING = motor is running it's calibration sequence
74 * IDLE = motor is not moving / has no power going through it
75 * BOOSTED = motor is in a boosted state, driving faster
76 */
77enum motor_status_list{ENABLED, DISABLED, ERRORLESS,CALIBRATING, IDLE, BOOSTED};

82 *
83 * @details servo_status_list has the following options: LATCHED, RELEASED, DETACHED
84 * LATCHED = servo is in position to keep the kicking leg from moving
85 * RELEASED = servo is in position to release the kicking leg
86 * DETACHED = state servo is in on boot, will update to the expected state when controller connects
87 */
88enum servo_status_list{LATCHED, RELEASED, DETACHED};
