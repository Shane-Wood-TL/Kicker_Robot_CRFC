114extern "C" void app_main(void) {
115 #ifdef TIME_BETWEEN_MESSAGES
116 gpio_set_direction(C0_4_pin, GPIO_MODE_OUTPUT);
117 #endif
118 battery_voltage_mutex = xSemaphoreCreateMutex();
119 controller_connected_mutex = xSemaphoreCreateMutex();
120 motor_status_mutex = xSemaphoreCreateMutex();
121 servo_status_mutex = xSemaphoreCreateMutex();
122 ramped_values_updated = xSemaphoreCreateMutex();
123 motor_speeds_settings_mutex = xSemaphoreCreateMutex();
124 update_main_display_mutex = xSemaphoreCreateMutex();
125 main_menu_values_mutex = xSemaphoreCreateMutex();
126 motor_speeds = xSemaphoreCreateMutex();
127 ramped_mutex = xSemaphoreCreateMutex();
128 network_channel_mutex = xSemaphoreCreateMutex();
129
130 right_drive_velocity_estimate_mutex = xSemaphoreCreateMutex();
131 left_drive_velocity_estimate_mutex = xSemaphoreCreateMutex();
132
133 esp_now_setup();
134 i2c_setup();
135 led_setup();
136
137 (void)xTaskCreate(display_driver_task, "display_driver_task", display_driver_stack_size, NULL, display_diver_priority, NULL);
138 (void)xTaskCreatePinnedToCore(kicker_drive_train_task, "kicker_drive_train_task", kicker_drive_train_stack_size, NULL, kicker_drive_train_priority, NULL, 0);
139 (void)xTaskCreate(servo_driver_task, "servo_driver_task", servo_driver_stack_size, NULL, servo_driver_priority, NULL);
140 (void)xTaskCreate(send_data_task, "send_data_task", send_data_stack_size, NULL, send_data_priority, NULL);
141
142 while (1) {
143 vTaskDelay(pdMS_TO_TICKS(main_task_wait_time));
144 }
145}
